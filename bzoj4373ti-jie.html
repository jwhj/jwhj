bzoj4373题解

Sun 11 December 2016

可以使用一种神奇的hash，维护区间的最小值和平方和，对于每个询问，先找出区间的最小值a，然后看看以a为首项，k为公差的等差数列的平方和是否等于这个区间的平方和。

平方和比较大，让它自然溢出就可以了。

为了速度写了zkw。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 300010
#define INF 1000000000
const ele size=1<<20;
const ele M=size>>1;
ele n,m,m1[size],m2[size];
inline void get(ele&a){
    char c;
    while ((c=getchar())<'0' || c>'9');
    a=c-'0';
    while ((c=getchar())>='0' && c<='9')
        a*=10,a+=c-'0';
}
inline void maintain(ele i){
    if (i<M){
        m1[i]=min(m1[i<<1],m1[i<<1|1]);
        m2[i]=m2[i<<1]+m2[i<<1|1];
    }
}
inline void upd(ele i,ele x){
    i+=M;
    m1[i]=x;
    m2[i]=x*x;
    while (i>1)
        maintain(i>>=1);
}
inline ele q1(ele l,ele r){
    ele ans=INF;
    for (l=l+M-1,r=r+M+1; l^r^1; l>>=1,r>>=1){
        if (~l&1)
            ans=min(ans,m1[l^1]);
        if (r&1)
            ans=min(ans,m1[r^1]);
    }
    return ans;
}
inline ele q2(ele l,ele r){
    ele ans=0;
    for (l=l+M-1,r=r+M+1; l^r^1; l>>=1,r>>=1){
        if (~l&1)
            ans+=m2[l^1];
        if (r&1)
            ans+=m2[r^1];
    }
    return ans;
}
inline ele calc(ele a,ele k,ele n){
    ele t1=a*a*n;
    ele t2=n*(n-1)*k*a;
    ele t3=(n-1)*n*(2*n-1)*k*k;
    return t1*6+t2*6+t3;
}
int main(){
    get(n); get(m);
    for (int i=0; i<n; ++i){
        ele a;
        get(a);
        m1[i+M+1]=a;
        m2[i+M+1]=a*a;
    }
    for (int i=M-1; i>0; --i)
        maintain(i);
    ele cnt=0;
    while (m--){
        ele op,u,v,w;
        get(op); get(u); get(v);
        if (op==2)
            get(w);
        u^=cnt; v^=cnt; w^=cnt;
        if (op==1)
            upd(u,v);
        else{
            ele a=q1(u,v),b=q2(u,v);
            ele ss=calc(a,w,v-u+1);
            if (b*6==ss){
                ++cnt;
                puts("Yes");
            }
            else puts("No");
        }
    }
    return 0;
}
