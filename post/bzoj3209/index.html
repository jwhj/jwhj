bzoj3209题解

Oct 22, 2016 10:02 · 90 words · 1 minute read 题解 bzoj 数位dp DP

用数位dp统计出sum(i)=1,2,…的i分别有多少个，然后用快速幂求出答案。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxk 53
#define MOD 10000007
ele n,m,c[maxk],f[maxk][maxk],g[maxk][maxk];
inline ele pow(ele a,ele x){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp%MOD)
		if (x&1)
			ans=ans*tmp%MOD;
	return ans;
}
int main(){
	scanf("%lld",&n);
	for (m=0; n; n>>=1,++m)
		c[m]=n&1;
	f[0][0]=1;
	if (c[0])
		f[0][1]=1;
	else f[0][1]=0;
	g[0][0]=g[0][1]=1;
	for (int i=2; i<=m; ++i)
		f[0][i]=g[0][i]=0;
	for (int i=1; i<m; ++i){
		f[i][0]=g[i][0]=1;
		for (int j=1; j<=m; ++j){
			g[i][j]=g[i-1][j]+g[i-1][j-1];
			if (c[i])
				f[i][j]=g[i-1][j]+f[i-1][j-1];
			else f[i][j]=f[i-1][j];
		}
	}
	ele ans=1;
	for (int i=1; i<=m; ++i)
		ans*=pow(i,f[m-1][i]),ans%=MOD;
	printf("%lld\n",ans);
	return 0;
}
