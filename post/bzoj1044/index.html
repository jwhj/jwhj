bzoj1044题解

Aug 12, 2017 10:02 · 134 words · 1 minute read 题解 bzoj dp 二分

本来是道水题的，可是智商下线走了好多弯路。

一开始我通过dp+二分优化来计算最大长度的最小值，并且顺便求出方案数，后来发现写起来非常麻烦，调了半天调不动了。

接下来发现分两次dp，第一次求出最大长度的最小值，第二次求出方案数，写起来会清晰很多。这回大概是写对了，可是$O(nm\log n)$的复杂度实在是过不了。不过分两次dp来降低代码复杂度这个思想应该还是很有用的。

后来我发现求出最大长度最小值只需要二分+贪心就可以了，而且跟NOIP2015D2T1长得几乎一样。计算方案数的dp可以用双指针来优化。

这道题我一上来就在想dp，这种习惯是不对的，因为实际上有的时候可能还有更简单、更高效的方法。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 50010
#define MOD 10007
const ele INF=1e9;
ele n,m,a[maxn],s[maxn],s1[maxn],f[maxn],g[maxn];
inline bool check(ele x){
	ele r=m,now=0;
	for (int i=0; i<n; ++i){
		if (now+a[i]>x){
			if (!r) return false;
			--r;
			now=0;
		}
		now+=a[i];
	}
	if (now>x && !r) return false;
	return true;
}
int main(){
	scanf("%d%d",&n,&m);
	s[0]=0;
	ele L=0;
	for (int i=0; i<n; ++i) scanf("%d",a+i),L=max(L,a[i]),s[i+1]=s[i]+a[i];
	--L;
	ele R=s[n];
	while (R-L>1){
		ele mid=(L+R)>>1;
		if (check(mid)) R=mid; else L=mid;
	}
	ele ans=R,ans1=0;
	s1[0]=g[0]=1;
	for (int i=1; i<=n; ++i) g[i]=(s[i]<=ans),s1[i]=(s1[i-1]+g[i])%MOD;
	ans1+=g[n];
	for (int j=1; j<=m; ++j){
		for (int i=0; i<=j; ++i) f[i]=0;
		ele p=1;
		for (int i=j+1; i<=n; ++i){
			while (s[i]-s[p]>ans) ++p;
			f[i]=((s1[i-1]-s1[p-1])%MOD+MOD)%MOD;
		}
		memcpy(g,f,sizeof(ele)*(n+1));
		(ans1+=g[n])%=MOD;
		s1[0]=g[0];
		for (int i=1; i<=n; ++i) s1[i]=(s1[i-1]+g[i])%MOD;
	}
	printf("%d %d\n",ans,ans1);
	return 0;
}
