Nov 6, 2016 10:02 · 191 words · 1 minute read 题解 bzoj 树分治

点剖裸题……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 20010
#define func(x) ((x)*((x)-1)/2)
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxn<<1];
ele n,ecnt;
bool flag[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
ele gcd(ele a,ele b){
	if (!b) return a;
	return gcd(b,a%b);
}
ele getsize(ele p,ele i){
	ele ans=1;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v])
			ans+=getsize(i,j->v);
	return ans;
}
ele dfs1(ele p,ele i,ele s,ele&k){
	ele ans=1;
	bool f=true;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v]){
			ele tmp=dfs1(i,j->v,s,k);
			if (tmp<<1>s)
				f=false;
			ans+=tmp;
		}
	if (s-ans<<1>s)
		f=false;
	if (f)
		k=i;
	return ans;
}
void dfs2(ele p,ele i,ele d,ele *c){
	++c[d%3];
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v])
			dfs2(i,j->v,d+j->w,c);
}
ele divide_and_conquer(ele p,ele i){
	ele s=getsize(-1,i);
	ele k; dfs1(-1,i,s,k);
	ele c[3],ans=0;
	memset(c,0,sizeof(c));
	flag[k]=true;
	for (edge *j=h[k]; j; j=j->nxt)
		if (!flag[j->v]){
			ele c1[3];
			memset(c1,0,sizeof(c1));
			dfs2(k,j->v,j->w,c1);
			for (int w=0; w<3; ++w)
				c[w]+=c1[w];
			ans-=func(c1[0])+c1[1]*c1[2];
			ans+=divide_and_conquer(k,j->v);
		}
	++c[0];
	ans+=func(c[0])+c[1]*c[2];
	return ans;
}
int main(){
	scanf("%lld",&n);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		--u,--v;
		addedge(u,v,w); addedge(v,u,w);
	}
	memset(flag,0,sizeof(flag));
	ele t1=n+2*divide_and_conquer(-1,0),t2=n*n,t3=gcd(t1,t2);
	printf("%lld/%lld\n",t1/t3,t2/t3);
	return 0;
}
