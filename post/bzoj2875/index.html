bzoj2875题解

Jul 9, 2016 10:02 · 117 words · 1 minute read 题解 bzoj 矩阵

为什么题目不说清楚线性伪随机的生成公式……

$$X_{i+1}=(aX_i+c) \mod m$$

所以用矩阵快速幂就可以做了。

还有一个坑点，就是longlong乘longlong模longlong。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
struct MAT
{
    ele n,m,a[10][10];
};
ele m,a,c,x0,n,g;
MAT ans,C;
inline ele mul(ele a,ele b)
{
    ele tmp=a,ans=0;
    for (; b; tmp=tmp*2%m,b>>=1)
        if (b&1) ans=(ans+tmp)%m;
    return ans;
}
inline MAT operator*(MAT a,MAT b)
{
    MAT ans;
    ans.n=a.n; ans.m=b.m;
    for (int i=0; i<a.n; ++i)
        for (int j=0; j<b.m; ++j)
        {
            ans.a[i][j]=0;
            for (int k=0; k<a.m; ++k)
                ans.a[i][j]+=mul(a.a[i][k],b.a[k][j]),
                ans.a[i][j]%=m;
        }
    return ans;
}
inline MAT pow(MAT a,ele x)
{
    MAT tmp=a,ans;
    ans.n=ans.m=a.n;
    memset(ans.a,0,sizeof(ans.a));
    for (int i=0; i<a.n; ++i) ans.a[i][i]=1;
    for (; x; tmp=tmp*tmp,x>>=1)
        if (x&1) ans=ans*tmp;
    return ans;
}
int main()
{
    scanf("%lld%lld%lld%lld%lld%lld",&m,&a,&c,&x0,&n,&g);
    ans=(MAT){2,1,{{x0},{1}}};
    C=(MAT){2,2,{{a,c},{0,1}}};
    C=pow(C,n);
    ans=C*ans;
    printf("%lld\n",ans.a[0][0]%g);
    return 0;
}
