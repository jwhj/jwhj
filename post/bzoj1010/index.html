bzoj1010题解

Dec 13, 2016 10:02 · 79 words · 1 minute read 题解 bzoj DP 斜率优化

没什么好说的……直接斜率优化就可以了。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 50010
ele n,p,l,r,c[maxn],bt[maxn],bf[maxn],q[maxn];
ele *t=bt+1,*f=bf+1;
inline double slop(ele j,ele k){
	return (double)(f[j]-f[k]+t[j]*t[j]-t[k]*t[k])/(t[j]-t[k])/2;
}
int main(){
	scanf("%lld%lld",&n,&p);
	++p;
	for (int i=0; i<n; ++i)
		scanf("%lld",c+i);
	t[0]=c[0];
	for (int i=1; i<n; ++i)
		t[i]=t[i-1]+c[i];
	for (int i=0; i<n; ++i)
		t[i]+=i;
	t[-1]=-1; f[-1]=0;
	l=r=0; q[0]=-1;
	for (int i=0; i<n; ++i){
		while (l<r && slop(q[l],q[l+1])<=t[i]-p)
			++l;
		ele j=q[l];
		f[i]=f[j]+(t[i]-t[j]-p)*(t[i]-t[j]-p);
		while (l<r && slop(q[r-1],q[r])>=slop(q[r],i))
			--r;
		q[++r]=i;
	}
	printf("%lld\n",f[n-1]);
	return 0;
}
