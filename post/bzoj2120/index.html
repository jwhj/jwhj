bzoj2120题解

Jun 1, 2016 10:02 · 220 words · 2 minutes read 题解 bzoj 统计 莫队

五月刚过，然而BGM歌名的直译貌似是“十一月过了”……

跟bzoj1878很像，但区别在于这道题有修改。

没有修改的话可以用莫队来做，其实有修改也可以。

没有修改的时候，提出询问的时间对询问的结果没有影响，但是现在有修改，所以我们要考虑询问的时间，每个询问要用一个三元组$(l,r,t)$来表示。

没有修改的时候，我们会移动左端点和右端点，其实时间也是可以像移动左右端点那样移动的。

把整个序列每S个分成一块，以左端点所在块为第一关键字，右端点所在块为第二关键字，时间为第三关键字排序，然后跑莫队。

方便起见，我们假设n和q同阶，可得时间复杂度为$O(n(S+(\frac{n}{S})^2))$，所以我们应该让$S=n^\frac{2}{3}$，此时算法的时间复杂度是$O(n^\frac{5}{3})$。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 10010
#define maxa 1000010
ele S;
struct ASK
{
	ele l,r,t;
	inline bool operator<(ASK b)const
	{
		if (l/S<b.l/S) return true;
		if (l/S>b.l/S) return false;
		if (r/S<b.r/S) return true;
		if (r/S>b.r/S) return false;
		return t<b.t;
	}
};
struct EVENT
{
	ele i,p,n;
};
ele n,m,ans,res[maxn],a[maxn],b[maxa],tmp[maxn];
ASK q[maxn];
EVENT e[maxn];
inline void add(ele x)
{
	if (!b[x]) ++ans;
	++b[x];
}
inline void rem(ele x)
{
	if (b[x]==1) --ans;
	--b[x];
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i) scanf("%d",a+i),tmp[i]=a[i];
	char s[10];
	ele x,y;
	for (int i=0; i<m; ++i)
	{
		scanf("%s%d%d",s,&x,&y);
		q[i].t=i;
		if (s[0]=='Q')
		{
			q[i].l=x-1; q[i].r=y-1;
			e[i].i=-1;
		}
		else
		{
			q[i].l=-1;
			--x;
			e[i].i=x;
			e[i].p=tmp[x];
			e[i].n=y;
			tmp[x]=y;
		}
	}
	S=exp(log(n)*2/3);
	sort(q,q+m);
	memset(b,0,sizeof(b));
	ele l=0,r=0,t=-1;
	b[a[0]]=1; ans=1;
	memset(res,0,sizeof(res));
	for (int i=0; i<m; ++i)
	{
		if (q[i].l==-1) continue;
		for (int j=l; j<q[i].l; ++j) rem(a[j]);
		for (int j=l-1; j>=q[i].l; --j) add(a[j]);
		for (int j=r; j>q[i].r; --j) rem(a[j]);
		for (int j=r+1; j<=q[i].r; ++j) add(a[j]);
		l=q[i].l; r=q[i].r;
		for (int j=t+1; j<=q[i].t; ++j)
		{
			if (e[j].i==-1) continue;
			a[e[j].i]=e[j].n;
			if (l<=e[j].i && e[j].i<=r) rem(e[j].p),add(e[j].n);
		}
		for (int j=t; j>q[i].t; --j)
		{
			if (e[j].i==-1) continue;
			a[e[j].i]=e[j].p;
			if (l<=e[j].i && e[j].i<=r) rem(e[j].n),add(e[j].p);
		}
		res[q[i].t]=ans;
		t=q[i].t;
	}
	for (int i=0; i<m; ++i)
		if (res[i]) printf("%d\n",res[i]);
	return 0;
}
