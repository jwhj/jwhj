bzoj2654题解

Dec 19, 2016 10:02 · 121 words · 1 minute read 题解 bzoj 乱搞 二分

假设给图中每条白边边权加上delta，然后直接跑一次kruskal，如果白边数量比较少，就可以把delta调小一点，如果白边比较多，就可以把delta调大一点。

所以二分这个delta，就能得到恰好有need条白边的权值和最小的生成树了。

有一点要注意一下，排序的时候，除了要比较边权之外，还要比较边的颜色，同权值的情况下，一定要指定白边优先或者黑边优先。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 50010
#define maxm 100010
struct edge{
	ele u,v,w,c;
	inline bool operator<(edge b)const{
		if (w!=b.w) return w<b.w;
		return c<b.c;
	}
};
ele n,m,need,ws,cnt,u[maxn];
edge e[maxm];
ele getf(ele *u,ele x){
	return (x==u[x])?(x):(u[x]=getf(u,u[x]));
}
inline void kruskal(ele delta){
	for (int i=0; i<m; ++i)
		if (!e[i].c)
			e[i].w+=delta;
	sort(e,e+m);
	for (int i=0; i<n; ++i)
		u[i]=i;
	ws=cnt=0;
	for (int i=0; i<m; ++i){
		ele a=getf(u,e[i].u),b=getf(u,e[i].v);
		if (a==b) continue;
		u[a]=b;
		ws+=e[i].w;
		if (!e[i].c)
			++cnt;
	}
	for (int i=0; i<m; ++i)
		if (!e[i].c)
			e[i].w-=delta;
}
int main(){
	scanf("%d%d%d",&n,&m,&need);
	for (int i=0; i<m; ++i)
		scanf("%d%d%d%d",&e[i].u,&e[i].v,&e[i].w,&e[i].c);
	ele l=-105,r=105;
	while (r-l>1){
		ele mid=(l+r)/2;
		kruskal(mid);
		if (cnt<need)
			r=mid;
		else l=mid;
	}
	kruskal(l);
	printf("%d\n",ws-l*need);
	return 0;
}
