UOJ34题解（快速傅立叶变换）

Apr 19, 2016 10:02 · 200 words · 1 minute read 题解 uoj FFT

题意大概就是给两个多项式，把它们乘起来。

暴力肯定是不行的。

假设多项式共有$n$项，$n$是2的整数次幂：

$$A(x)=a_0+a_1 x+a2 x^2+\ldots+a{n-1}x^{n-1}$$

我们称向量$a$为多项式$A(x)$的系数表示。下面定义向量$y$（$\omega_n$为n次单位根）：

$$y_k=A(\omega_n^{k})=\Sigma a_j\omega_n^{jk}$$

那么称$y$为$A(x)$的点值表示，由$a$求$y$的过程称为离散傅立叶变换，即DFT。

直接计算显然是$O(n^2)$的。

定义两个多项式$A^{[0]}(x)$和$A^{[1]}(x)$，它们的系数表示分别为$a^{[0]}$和$a^{[1]}$。有：

$$a^{[0]}=(a_0,a_2,\ldots)$$

$$a^{[1]}=(a_1,a_3,\ldots)$$

$$A(x)=A^{[0]}(x^2)+xA^{[1]}(x^2)$$

递归地对$a^{[0]}$和$a^{[1]}$进行FFT，得到$y^{[0]}$和$y^{[1]}$。

那么对于$0\le k\le\frac{n}{2}-1$，就有

$$y_k=A^{[0]}(\omega_n^{2k})+\omega_n^{k}A^{[1]}(\omegan^{2k})=A^{[0]}(\omega{n/2}^{k})+\omegan^{k}A^{[1]}(\omega{n/2}^{k})=y^{[0]}_k+\omega_n^{k}y^{[1]}_k$$

$$y_{k+\frac{n}{2}}=y^{[0]}_k-\omega_n^{k}y^{[1]}_k$$

于是就可以高效地完成DFT了。

DFT的逆变换称为IDFT，有：

$$a_k=\frac{1}{n} \Sigma y_j\omega_n^{-jk}$$

最后一个问题：如何求n次单位根？有以下的式子：

$$\omega_n=e^{\frac{2\pi i}{n}}=cos(\frac{2\pi}{n})+isin(\frac{2\pi}{n})$$

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <complex>
#define ele int
using namespace std;
#define maxn 400010
#define pn complex<double>
const double pi=3.1415926535897;
struct POOL{
	pn a[10000000];
	ele cnt;
	inline pn* alloc(ele n){
		cnt+=n;
		return a+cnt-n;
	}
};
ele n,m,a[maxn],b[maxn];
pn a1[maxn],a2[maxn],b1[maxn],b2[maxn];
POOL p;
pn* DFT(ele n,pn *a,pn *b){
	if (n==1){
		b[0]=a[0];
		return b;
	}
	pn *a1=p.alloc(n>>1),*a2=p.alloc(n>>1);
	for (int i=0; i<(n>>1); ++i)
		a1[i]=a[i<<1],a2[i]=a[i<<1|1];
	pn *b1=p.alloc(n>>1),*b2=p.alloc(n>>1);
	DFT(n>>1,a1,b1); DFT(n>>1,a2,b2);
	pn o=1,o1=(pn){cos(pi*2/n),sin(pi*2/n)};
	for (int i=0; i<(n>>1); ++i){
		b[i]=b1[i]+o*b2[i];
		b[i+(n>>1)]=b1[i]-o*b2[i];
		o*=o1;
	}
	return b;
}
pn* IDFT(ele n,pn *a,pn *b){
	if (n==1){
		b[0]=a[0];
		return b;
	}
	pn *a1=p.alloc(n>>1),*a2=p.alloc(n>>1);
	for (int i=0; i<(n>>1); ++i)
		a1[i]=a[i<<1],a2[i]=a[i<<1|1];
	pn *b1=p.alloc(n>>1),*b2=p.alloc(n>>1);
	IDFT(n>>1,a1,b1); IDFT(n>>1,a2,b2);
	pn o=1,o1=(pn){cos(-pi*2/n),sin(-pi*2/n)};
	for (int i=0; i<(n>>1); ++i){
		b[i]=b1[i]+o*b2[i];
		b[i+(n>>1)]=b1[i]-o*b2[i];
		o*=o1;
	}
	return b;
}
int main(){
	scanf("%d%d",&n,&m);
	ele tmp=1;
	while (tmp<=n || tmp<=m) tmp<<=1;
	tmp<<=1;
	ele x;
	memset(a,0,sizeof(a));
	for (int i=0; i<=n; ++i) scanf("%d",&x),a1[i]=x;
	memset(b,0,sizeof(b));
	for (int i=0; i<=m; ++i) scanf("%d",&x),a2[i]=x;
	ele tmp1=n+m+1;
	n=tmp;
	p.cnt=0;
	DFT(n,a1,b1);
	p.cnt=0;
	DFT(n,a2,b2);
	for (int i=0; i<n; ++i) b1[i]*=b2[i];
	p.cnt=0;
	IDFT(n,b1,a1);
	for (int i=0; i<tmp1; ++i) printf("%d ",(ele)(a1[i].real()/n+0.5));
	puts("");
	return 0;
}
