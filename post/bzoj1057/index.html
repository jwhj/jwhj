bzoj1057题解

Aug 10, 2017 10:02 · 98 words · 1 minute read 题解 bzoj dp

从左上到右下枚举正方形/矩形的右下角。维护f[j]代表当前行第j列之多向上多少格满足黑白相间。若存在if[j]，那么这个f[i]实际上是毫无意义的。维护f[i]向左严格递减的序列，对每个右下角，在每个最大高度降低的点计算一次最大面积就可以了。

这个dp看上去非常暴力，时间复杂度像是$O(nm^2)$，然而它却A了，原因是，一个长度为k的序列的递增/递减子序列长度的期望是$O(\log k)$，也就是说，算法的期望复杂度是$O(nm\log m)$。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 2010
ele n,m,a[maxn][maxn],f[maxn];
pair<ele,ele> Q[maxn];
inline ele _sqr(ele x){
	return x*x;
}
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j) scanf("%d",&a[i][j]);
	memset(f,0,sizeof(f));
	ele ans1=0,ans2=0;
	for (int i=0; i<n; ++i){
		ele qh=0,qt=0;
		for (int j=0; j<m; ++j){
			if (!i || a[i-1][j]!=a[i][j]) ++f[j]; else f[j]=1;
			pair<ele,ele> tmp=make_pair(j,f[j]);
			if (!j || a[i][j-1]!=a[i][j])
				while (qh<qt && Q[qt-1].second>f[j]) tmp.first=Q[qt-1].first,--qt;
			else qt=qh;
			if (qh==qt || Q[qt-1].second<f[j]) Q[qt++]=tmp;
			for (int k=qh; k<qt; ++k)
				ans1=max(ans1,_sqr(min(j-Q[k].first+1,Q[k].second))),
				ans2=max(ans2,(j-Q[k].first+1)*Q[k].second);
		}
	}
	printf("%d\n%d\n",ans1,ans2);
	return 0;
}
