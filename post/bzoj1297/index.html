bzoj1297题解

May 7, 2016 10:02 · 127 words · 1 minute read 题解 bzoj 矩阵

首先假设边权全都是1。

定义$Ci$为一个矩阵，且满足$C{ijk}$为j到k长度为i的路径条数。

那么显然$C_1$就是邻接矩阵，可以证明$C_i=C_1^i$。

现在边权有大于1的，但都是整数，而且不超过9，所以可以把每个点拆成9个点来做。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 110
#define MOD 2009
struct mat{
	ele n,m,a[maxn][maxn];
};
inline mat operator*(mat a,mat b){
	mat ans;
	ans.n=a.n; ans.m=b.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<b.m; ++j){
			ans.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				ans.a[i][j]+=a.a[i][k]*b.a[k][j]%MOD,
				ans.a[i][j]%=MOD;
		}
	return ans;
}
ele n,t;
mat g,ans;
char s[20];
inline mat pow(mat a,ele x){
	mat ans,tmp=a;
	ans.n=a.n; ans.m=a.m;
	memset(ans.a,0,sizeof(ans.a));
	for (int i=0; i<ans.n; ++i)
		ans.a[i][i]=1;
	for (; x>0; x>>=1){
		if (x&1) ans=ans*tmp;
		tmp=tmp*tmp;
	}
	return ans;
}
int main(){
	scanf("%d%d",&n,&t);
	memset(g.a,0,sizeof(g.a));
	g.n=g.m=n*9;
	for (int i=0; i<n; ++i)
		for (int j=1; j<9; ++j)
			g.a[i*9+j][i*9+j-1]=1;
	for (int i=0; i<n; ++i){
		scanf("%s",s);
		for (int j=0; j<n; ++j)
			if (s[j]>'0')
				g.a[i*9][j*9+s[j]-'0'-1]=1;
	}
	ans=pow(g,t);
	printf("%d\n",ans.a[0][(n-1)*9]);
	return 0;
}
