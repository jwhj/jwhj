hdu5412题解

Oct 7, 2016 10:02 · 210 words · 1 minute read 题解 hdu 整体二分

题目没有强制在线，用整体二分就可以做了，详细的题解晚点再补。

估计很多人写的是树套树，所以我的程序跑到了第一版。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
struct BIT{
	inline void alloc(ele n){
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele k){
		while (i<=n){
			a[i]+=k;
			i+=i&(-i);
		}
	}
	inline ele prefix(ele i){
		ele ans=0;
		while (i>0){
			ans+=a[i];
			i-=i&(-i);
		}
		return ans;
	}
	inline ele sum(ele l,ele r){ return prefix(r)-prefix(l-1);}
	ele n,a[maxn];
};
struct ASK{
	ele ty,u,v,w,id;
};
ele n,m,acnt,len,res[maxn],arr[maxn<<1],a[maxn];
ASK q[maxn*3],q1[maxn*3];
BIT b;
inline void solve(ASK *q,ASK *q1,ele ql,ele qr,ele l,ele r){
	if (l+1>=r){
		for (int i=ql; i<=qr; ++i)
			if (q[i].ty==0)
				res[q[i].id]=l;
		return;
	}
	ele j=ql,k=qr,tmp,mid=(l+r)>>1;
	for (int i=ql; i<=qr; ++i)
		if (q[i].ty==0)
			if (q[i].w<=(tmp=b.sum(q[i].u,q[i].v)))
				q1[j++]=q[i];
			else q[i].w-=tmp,q1[k--]=q[i];
		else
			if (q[i].v<mid){
				b.add(q[i].u,(q[i].ty==1)?(-1):(1));
				q1[j++]=q[i];
			}
			else q1[k--]=q[i];
	--j,++k;
	for (int i=k; i<qr+k-i; ++i) swap(q1[i],q1[qr+k-i]);
	for (int i=ql; i<=qr; ++i) q[i]=q1[i];
	for (int i=ql; i<=j; ++i)
		if (q[i].ty) b.add(q[i].u,(q[i].ty==1)?(1):(-1));
	solve(q,q1,ql,j,l,mid);
	solve(q,q1,k,qr,mid,r);
}
inline ele idx(ele x){
	return lower_bound(arr,arr+len,x)-arr+1;
}
int main(){
	while (~scanf("%d",&n)){
		acnt=0;
		for (int i=0; i<n; ++i){
			scanf("%d",a+i);
			arr[i]=a[i];
			q[i]={2,i+1,a[i],0,0};
		}
		scanf("%d",&m);
		ele mcnt=n;
		len=n;
		for (int i=0; i<m; ++i){
			ele ty,u,v,w;
			scanf("%d%d%d",&ty,&u,&v);
			if (ty==1){
				q[mcnt++]={1,u,a[u-1],0,i};
				arr[len++]=v;
				q[mcnt++]={2,u,v,0,i};
				a[u-1]=v;
			}
			else{
				scanf("%d",&w);
				q[mcnt++]={0,u,v,w,i};
			}
		}
		sort(arr,arr+len);
		len=unique(arr,arr+len)-arr;
		for (int i=0; i<mcnt; ++i)
			if (q[i].ty)
				q[i].v=idx(q[i].v);
		b.alloc(n);
		memset(res,-1,sizeof(res));
		solve(q,q1,0,mcnt-1,1,len+1);
		for (int i=0; i<m; ++i)
			if (res[i]>=0) printf("%d\n",arr[res[i]-1]);
	}
	return 0;
}
