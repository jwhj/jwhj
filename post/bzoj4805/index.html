bzoj4805题解

Apr 3, 2017 10:02 · 88 words · 1 minute read 题解 bzoj 数学 杜教筛

非常简单的杜教筛……

刚刚发现杜教筛的不用散列表和map的写法，效率比我以前写的那种高很多。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxs 3000010
ele n,pcnt,S,plst[maxs],phi[maxs],ans[maxs];
bool flag[maxs];
int main(){
	S=3000000;
	pcnt=0;
	memset(flag,0,sizeof(flag));
	phi[1]=1;
	for (int i=2; i<maxs; ++i){
		if (!flag[i]) plst[pcnt++]=i,phi[i]=i-1;
		for (int j=0; j<pcnt && i*plst[j]<maxs; ++j){
			flag[i*plst[j]]=true;
			if (i%plst[j]) phi[i*plst[j]]=phi[i]*phi[plst[j]];
			else{
				phi[i*plst[j]]=phi[i]*plst[j];
				break;
			}
		}
	}
	for (int i=2; i<maxs; ++i) phi[i]+=phi[i-1];
	scanf("%lld",&n);
	ele k;
	for (k=1; n/k>=S; ++k);
	for (--k; k; --k){
		ele m=n/k;
		ans[k]=m*(m+1)/2;
		for (int i=m; i>1;){
			ele j=m/(m/i+1);
			ele tmp=(m/i<S)?phi[m/i]:ans[i*k];
			ans[k]-=tmp*(i-j);
			i=j;
		}
	}
	printf("%lld\n",(n<S)?phi[n]:ans[1]);
	return 0;
}
