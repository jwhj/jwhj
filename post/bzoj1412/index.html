bzoj1412题解

Mar 3, 2017 10:02 · 216 words · 2 minutes read 题解 bzoj 网络流

就是一个最小割，但建图的时候还是有一些细节的，看代码吧……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 10010
#define INF 1000000000
const ele di[]={-1,1,0,0};
const ele dj[]={0,0,-1,1};
struct edge{
	ele v,r;
	edge *nxt,*rev;
}ep[4000000];
ele n,m,ecnt,dep[maxn],a[110][110];
edge *g[maxn],*cur[maxn];
inline void addedge1(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->r=w; p->nxt=g[u]; p->rev=ep+((ecnt-1)^1);
	g[u]=p;
}
inline void addedge(ele u,ele v,ele w){
	addedge1(u,v,w); addedge1(v,u,0);
}
inline bool bfs(ele s,ele t){
	static ele Q[maxn];
	ele qh=0,qt=1;
	Q[0]=s;
	memset(dep,0,sizeof(dep));
	dep[s]=1;
	while (qh<qt){
		for (edge *j=g[Q[qh]]; j; j=j->nxt)
			if (!dep[j->v] && j->r){
				dep[j->v]=dep[Q[qh]]+1;
				Q[qt++]=j->v;
				if (j->v==t) return true;
			}
		++qh;
	}
	return false;
}
ele dfs(ele i,ele t,ele p){
	if (i==t || !p) return p;
	ele f=0;
	for (edge*&j=cur[i]; j; j=j->nxt)
		if (dep[i]+1==dep[j->v] && j->r){
			ele tmp=dfs(j->v,t,min(p,j->r));
			if (tmp){
				j->r-=tmp;
				j->rev->r+=tmp;
				f+=tmp;
				p-=tmp;
				if (!p) break;
			}
		}
	if (!p) dep[i]=-1;
	return f;
}
inline ele dinic(ele s,ele t){
	ele ans=0;
	while (bfs(s,t)){
		memcpy(cur,g,sizeof(edge*)*(2+n*m));
		ans+=dfs(s,t,INF);
	}
	return ans;
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=0;
	memset(g,0,sizeof(g));
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j){
			scanf("%d",&a[i][j]);
			if (a[i][j]==1) addedge(0,2+i*m+j,INF);
			else if (a[i][j]==2) addedge(2+i*m+j,1,INF);
		}
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			if (a[i][j]!=2)
				for (int w=0; w<4; ++w){
					ele ni=i+di[w],nj=j+dj[w];
					if (0<=ni && ni<n && 0<=nj && nj<m && (a[i][j]!=1 || a[ni][nj]!=1))
						addedge(2+i*m+j,2+ni*m+nj,1);
				}
	printf("%d\n",dinic(0,1));
	return 0;
}
