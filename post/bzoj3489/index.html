bzoj3489题解

Sep 2, 2016 10:02 · 266 words · 2 minutes read 题解 bzoj KD树 统计

最快的解法似乎是可持久化树套树，然而我智商太低只会用kd树来做。

记pi,qi为ai上一次出现的位置和下一次出现的位置，那么要求的答案就是$$\max_{l\le i\le r且p_i\lt l且q_i\gt r}{a_i}$$

把每个ai看成一个三位空间中的点(i,pi,qi)，就可以用kd树来进行查询了。

kd树查询最大值的时候有一个优化的技巧：如果当前子树的最大值比已经找到的最大值要小，那么这个子树就不用查询了。在本题中这个优化的效果非常明显。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
#define K 3
struct node
{
	ele c[K],mn[K],mx[K],max,s,x;
	node *l,*r;
}pool[maxn];
ele D,ans;
inline bool cmp(node *a,node *b)
{
	return a->c[D]<b->c[D];
}
inline void maintain(node *x)
{
	x->s=1; x->max=x->x;
	for (int i=0; i<K; ++i) x->mn[i]=x->mx[i]=x->c[i];
	if (x->l)
	{
		x->s+=x->l->s; x->max=max(x->max,x->l->max);
		for (int i=0; i<K; ++i)
			x->mn[i]=min(x->mn[i],x->l->mn[i]),
			x->mx[i]=max(x->mx[i],x->l->mx[i]);
	}
	if (x->r)
	{
		x->s+=x->r->s; x->max=max(x->max,x->r->max);
		for (int i=0; i<K; ++i)
			x->mn[i]=min(x->mn[i],x->r->mn[i]),
			x->mx[i]=max(x->mx[i],x->r->mx[i]);
	}
}
node* build(node **a,ele l,ele r,ele d)
{
	if (l>r) return NULL;
	if (l==r)
	{
		a[l]->l=a[l]->r=NULL;
		maintain(a[l]);
		return a[l];
	}
	ele mid=l+r>>1;
	D=d;
	nth_element(a+l,a+mid,a+r+1,cmp);
	a[mid]->l=build(a,l,mid-1,(d+1)%K);
	a[mid]->r=build(a,mid+1,r,(d+1)%K);
	maintain(a[mid]);
	return a[mid];
}
inline bool in(ele a[],ele b[],ele c[],ele d[])
{
	for (int i=0; i<K; ++i)
		if (a[i]<c[i] || b[i]>d[i]) return false;
	return true;
}
inline bool out(ele a[],ele b[],ele c[],ele d[])
{
	for (int i=0; i<K; ++i)
		if (b[i]<c[i] || a[i]>d[i]) return true;
	return false;
}
ele query(node *x,ele q1[],ele q2[])
{
	if (!x) return 0;
	if (ans>x->max) return 0;
	if (in(x->mn,x->mx,q1,q2)) return ans=max(ans,x->max);
	if (out(x->mn,x->mx,q1,q2)) return 0;
	if (in(x->c,x->c,q1,q2)) ans=max(ans,x->x);
	ans=max(ans,query(x->l,q1,q2));
	ans=max(ans,query(x->r,q1,q2));
	return ans;
}
ele n,m,a[maxn],pos[maxn],p[maxn],q[maxn];
node *root,*tmp[maxn];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i) scanf("%d",a+i);
	for (int i=1; i<=n; ++i) pos[i]=-1;
	for (int i=0; i<n; ++i) p[i]=pos[a[i]],pos[a[i]]=i;
	for (int i=1; i<=n; ++i) pos[i]=n;
	for (int i=n-1; i>=0; --i) q[i]=pos[a[i]],pos[a[i]]=i;
	for (int i=0; i<n; ++i)
	{
		node *x=pool+i;
		x->c[0]=i; x->c[1]=p[i]; x->c[2]=q[i];
		x->x=a[i];
		x->l=x->r=NULL;
		maintain(x);
		tmp[i]=pool+i;
	}
	root=build(tmp,0,n-1,0);
	ele lastans=0;
	while (m--)
	{
		ele u,v;
		scanf("%d%d",&u,&v);
		u=(u+lastans)%n;
		v=(v+lastans)%n;
		if (u>v) swap(u,v);
		ele q1[K]={u,-1,v+1},q2[K]={v,u-1,n};
		ans=0;
		lastans=query(root,q1,q2);
		printf("%d\n",lastans);
	}
	return 0;
}
