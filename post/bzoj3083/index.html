bzoj3083题解

Nov 2, 2016 10:02 · 375 words · 2 minutes read 题解 bzoj 树剖

这题又有子树操作，又有换根，用估计做不了。

注意到虽然根换了，树的形态不会变，所以可以用树剖来做。

换根对链上操作没有影响，子树操作分三种情况讨论换根的影响，详细的看代码。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define ll long long
using namespace std;
#define maxn 100010
#define maxk 19
#define INF 1e10
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<1];
ele ecnt;
edge *h[maxn];
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
struct node{
	ele min,set;
	node *l,*r;
}np[maxn<<2];
ele ncnt,seq[maxn],w[maxn];
inline void pushdown(node *x){
	if (!x) return;
	if (x->set==-1) return;
	if (x->l) x->l->set=x->l->min=x->set;
	if (x->r) x->r->set=x->r->min=x->set;
	x->set=-1;
}
inline void maintain(node *x){
	if (!x) return;
	if (x->set!=-1)
		x->min=x->set;
	else x->min=min(x->l->min,x->r->min);
}
node* build(ele l,ele r){
	node *p=&np[ncnt++];
	if (l==r){
		p->l=p->r=NULL;
		p->min=p->set=w[seq[l]];
	}
	else{
		ele mid=l+r>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		p->set=-1;
		maintain(p);
	}
	return p;
}
void supd(node *x,ele a,ele b,ele l,ele r,ele k){
	pushdown(x);
	if (l<=a && b<=r)
		x->set=k;
	else{
		ele mid=a+b>>1;
		if (l<=mid) supd(x->l,a,mid,l,r,k);
		if (mid<r) supd(x->r,mid+1,b,l,r,k);
	}
	maintain(x);
}
ele squery(node *x,ele a,ele b,ele l,ele r){
	pushdown(x);
	if (l<=a && b<=r)
		return x->min;
	ele mid=a+b>>1;
	ll ans=INF;
	if (l<=mid) ans=min(ans,(ll)squery(x->l,a,mid,l,r));
	if (mid<r) ans=min(ans,(ll)squery(x->r,mid+1,b,l,r));
	return ans;
}
ele dep[maxn],size[maxn],fa[maxn][maxk];
edge *hc[maxn];
void dfs1(ele p,ele i){
	size[i]=1;
	fa[i][0]=p;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p){
			dep[j->v]=dep[i]+1;
			dfs1(i,j->v);
			size[i]+=size[j->v];
		}
	hc[i]=NULL;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p)
			if (hc[i]==NULL || size[j->v]>size[hc[i]->v])
				hc[i]=j;
}
ele scnt,ac[maxn],pos[maxn];
void dfs2(ele p,ele i){
	pos[i]=scnt;
	seq[scnt++]=i;
	if (hc[i])
		ac[hc[i]->v]=ac[i],dfs2(i,hc[i]->v);
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && j!=hc[i])
			ac[j->v]=j->v,dfs2(i,j->v);
}
inline ele lca(ele u,ele v){
	if (dep[u]>dep[v])
		swap(u,v);
	ele k=dep[v]-dep[u];
	for (int i=0; k; ++i,k>>=1)
		if (k&1)
			v=fa[v][i];
	if (u==v) return u;
	for (int i=maxk-1; i>=0; --i)
		if (fa[u][i]!=fa[v][i])
			u=fa[u][i],v=fa[v][i];
	return fa[u][0];
}
ele n,m,r;
node *root;
inline void upd(ele u,ele v,ele k){
	while (u!=-1 && pos[u]>=pos[v]){
		ele i=ac[u];
		supd(root,0,n-1,max(pos[v],pos[i]),pos[u],k);
		u=fa[seq[max(pos[v],pos[i])]][0];
	}
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v);
		--u,--v;
		addedge(u,v); addedge(v,u);
	}
	for (int i=0; i<n; ++i)
		scanf("%d",w+i);
	scanf("%d",&r);
	--r;
	dep[0]=0;
	dfs1(-1,0);
	for (int i=1; i<maxk; ++i)
		for (int j=0; j<n; ++j)
			if (fa[j][i-1]!=-1)
				fa[j][i]=fa[fa[j][i-1]][i-1];
			else fa[j][i]=-1;
	scnt=0; ac[0]=0;
	dfs2(-1,0);
	ncnt=0;
	root=build(0,n-1);
	while (m--){
		ele opt,u,v,k;
		scanf("%d%d",&opt,&u);
		--u;
		if (opt==2)
			scanf("%d%d",&v,&k),--v;
		if (opt==1)
			r=u;
		else if (opt==2){
			ele w=lca(u,v);
			upd(u,w,k); upd(v,w,k);
		}
		else if (opt==3){
			ele w=lca(r,u),ans;
			if (r==u)
				ans=squery(root,0,n-1,0,n-1);
			else if (w==u){
				k=dep[r]-dep[u]-1;
				v=r;
				for (int i=0; k; ++i,k>>=1)
					if (k&1)
						v=fa[v][i];
				ans=squery(root,0,n-1,0,pos[v]-1);
				ans=min(ans,squery(root,0,n-1,pos[v]+size[v],n-1));
			}
			else
				ans=squery(root,0,n-1,pos[u],pos[u]+size[u]-1);
			printf("%d\n",ans);
		}
	}
	return 0;
}
