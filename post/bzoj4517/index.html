bzoj4517题解

Sep 8, 2016 10:02 · 96 words · 1 minute read 题解 bzoj 组合 数学

令$f_n$为n个数错排的个数，那么$fn=(n-1)(f{n-1}+f{n-2})$，且有$\textrm{ans}=\textrm{C}_n^mf{n-m}$。

预处理出$x!和(x!)^{-1}(\mod 10^9+7)$，就可以快速地算出组合数。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 1000010
#define MOD 1000000007
ele n,m,fac[maxn],ifac[maxn],f[maxn];
void exgcd(ele a,ele b,ele&x,ele&y)
{
    if (!b)
    {
        x=1; y=0;
        return;
    }
    exgcd(b,a%b,y,x);
    y-=a/b*x;
}
inline ele inv(ele x)
{
    ele u,v;
    exgcd(x,MOD,u,v);
    return (u+MOD)%MOD;
}
inline ele C(ele m,ele n)
{
    return (fac[n]*ifac[m]%MOD)*ifac[n-m]%MOD;
}
int main()
{
    ele T;
    fac[0]=ifac[0]=1;
    for (int i=1; i<maxn; ++i)
    {
        fac[i]=fac[i-1]*i%MOD;
        ifac[i]=inv(fac[i]);
    }
    f[0]=1; f[1]=0; f[2]=1;
    for (int i=3; i<maxn; ++i)
        f[i]=(i-1)*(f[i-1]+f[i-2])%MOD;
    scanf("%lld",&T);
    while (T--)
    {
        scanf("%lld%lld",&n,&m);
        if (m<=n)
            printf("%lld\n",C(m,n)*f[n-m]%MOD);
        else puts("0");
    }
    return 0;
}
