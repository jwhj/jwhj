bzoj4066题解

Sep 1, 2016 10:02 · 290 words · 2 minutes read 题解 bzoj KD树

直接用KD树就能做了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxm 200010
#define alpha 0.7
#define dc(x) (x)^=lastans
struct node
{
    ele c[2],mn[2],mx[2],x,s,sum;
    node *l,*r;
}pool[maxm];
ele n,pcnt,K;
inline bool cmp(node *a,node *b)
{
    return a->c[K]<b->c[K];
}
inline void maintain(node *x)
{
    x->s=1; x->sum=x->x;
    for (int i=0; i<2; ++i) x->mn[i]=x->mx[i]=x->c[i];
    if (x->l)
    {
        x->s+=x->l->s,x->sum+=x->l->sum;
        for (int i=0; i<2; ++i)
            x->mn[i]=min(x->mn[i],x->l->mn[i]),
            x->mx[i]=max(x->mx[i],x->l->mx[i]);
    }
    if (x->r)
    {
        x->s+=x->r->s,x->sum+=x->r->sum;
        for (int i=0; i<2; ++i)
            x->mn[i]=min(x->mn[i],x->r->mn[i]),
            x->mx[i]=max(x->mx[i],x->r->mx[i]);
    }
}
node* build(node **a,ele l,ele r,ele k)
{
    if (l>r) return NULL;
    if (l==r)
    {
        a[l]->l=a[l]->r=NULL;
        maintain(a[l]);
        return a[l];
    }
    ele mid=l+r>>1;
    K=k;
    nth_element(a+l,a+mid,a+r+1,cmp);
    a[mid]->l=build(a,l,mid-1,k^1);
    a[mid]->r=build(a,mid+1,r,k^1);
    maintain(a[mid]);
    return a[mid];
}
void expt(node *x,node **res,ele i)
{
    if (!x) return;
    expt(x->l,res,i);
    if (x->l) i+=x->l->s;
    res[i++]=x;
    expt(x->r,res,i);
}
node* ins(node *x,node *y,ele k)
{
    if (!x) return y;
    K=k;
    if (cmp(y,x))
        x->l=ins(x->l,y,k^1);
    else x->r=ins(x->r,y,k^1);
    maintain(x);
    if (x->l && x->l->s>x->s*alpha || x->r && x->r->s>x->s*alpha)
    {
        static node* tmp[maxm];
        expt(x,tmp,0);
        x=build(tmp,0,x->s-1,k);
    }
    return x;
}
inline bool init(ele a[],ele b[],ele c[],ele d[])
{
    for (int i=0; i<2; ++i)
        if (a[i]<c[i] || b[i]>d[i]) return false;
    return true;
}
inline bool outit(ele a[],ele b[],ele c[],ele d[])
{
    for (int i=0; i<2; ++i)
        if (b[i]<c[i] || a[i]>d[i]) return true;
    return false;
}
ele query(node *x,node q1,node q2)
{
    if (!x) return 0;
    if (init(x->mn,x->mx,q1.c,q2.c)) return x->sum;
    if (outit(x->mn,x->mx,q1.c,q2.c)) return 0;
    ele ans=0;
    if (init(x->c,x->c,q1.c,q2.c)) ans+=x->x;
    ans+=query(x->l,q1,q2);
    ans+=query(x->r,q1,q2);
    return ans;
}
inline node pt(ele x,ele y)
{
    node ans=(node){{x,y},{},{},0,0,0,NULL,NULL};
    return ans;
}
int main()
{
    scanf("%d",&n);
    ele lastans=0;
    node *root=NULL;
    pcnt=0;
    while (1)
    {
        ele cmd,a,b,c,d;
        scanf("%d",&cmd);
        if (cmd==3) break;
        if (cmd==1)
        {
            scanf("%d%d%d",&a,&b,&c);
            dc(a); dc(b); dc(c);
            node *p=&pool[pcnt++];
            p->l=p->r=NULL;
            p->c[0]=a; p->c[1]=b;
            p->x=c;
            maintain(p);
            root=ins(root,p,0);
        }
        else
        {
            scanf("%d%d%d%d",&a,&b,&c,&d);
            dc(a); dc(b); dc(c); dc(d);
            printf("%d\n",lastans=query(root,pt(a,b),pt(c,d)));
        }
    }
    return 0;
}
