bzoj3289题解

Jun 26, 2016 10:02 · 184 words · 1 minute read 题解 bzoj 莫队

先把文件的大小离散化，再用一个树状数组来存储当前询问区间中每个数出现的次数，就可以用莫队算法来做了。

时间复杂度是$O(n\sqrt n\log n)$。

代码：

#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 50010
ele B;
struct query
{
	ele l,r,id;
	inline bool operator<(query b)const
	{
		if (l/B!=b.l/B) return l/B<b.l/B;
		return r<b.r;
	}
};
struct BIT
{
	inline void alloc(ele n)
	{
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele x)
	{
		while (i<=n)
		{
			a[i]+=x;
			i+=i&(-i);
		}
	}
	inline ele prefix(ele i)
	{
		ele ans=0;
		while (i>0)
		{
			ans+=a[i];
			i-=i&(-i);
		}
		return ans;
	}
	inline ele sum(ele l,ele r)
	{
		if (l>r) return 0;
		return prefix(r)-prefix(l-1);
	}
	ele n,a[maxn];
};
ele n,m,len,a[maxn],arr[maxn],res[maxn],ans;
query q[maxn];
BIT s;
int main()
{
	scanf("%lld",&n);
	for (int i=0; i<n; ++i) scanf("%lld",a+i),arr[i]=a[i];
	sort(arr,arr+n);
	len=unique(arr,arr+n)-arr;
	for (int i=0; i<n; ++i)
		a[i]=lower_bound(arr,arr+len,a[i])-arr+1;
	B=sqrt(n);
	scanf("%lld",&m);
	for (int i=0; i<m; ++i)
	{
		scanf("%lld%lld",&q[i].l,&q[i].r);
		--q[i].l,--q[i].r;
		q[i].id=i;
	}
	sort(q,q+m);
	ele l=0,r=0;
	ans=0;
	s.alloc(len);
	s.add(a[0],1);
	for (int i=0; i<m; ++i)
	{
		for (int j=l; j<q[i].l; ++j)
			s.add(a[j],-1),ans-=s.sum(1,a[j]-1);
		for (int j=l-1; j>=q[i].l; --j)
			ans+=s.sum(1,a[j]-1),s.add(a[j],1);
		for (int j=r; j>q[i].r; --j)
			s.add(a[j],-1),ans-=s.sum(a[j]+1,len);
		for (int j=r+1; j<=q[i].r; ++j)
			ans+=s.sum(a[j]+1,len),s.add(a[j],1);
		res[q[i].id]=ans;
		l=q[i].l; r=q[i].r;
	}
	for (int i=0; i<m; ++i) printf("%lld\n",res[i]);
	return 0;
}
