bzoj4803题解

Jul 8, 2017 10:02 · 200 words · 1 minute read 题解 bzoj 数学

因为phi(n)<=10^14，所以n至多有一个超过10^7+1的素因数。

先假设n的素因数均不超过10^7，那么只需预处理出不超过10^7且满足(p-1)|phi(n)的素数p，然后枚举它们在n中的指数就可以了。

现在考虑那个超过10^7的因数。枚举每个phi(n)的超过10^7的因数p，如果p+1是一个质数，那就说明n有可能有因数p+1，使用类似上面那种情况的方法搜出n剩下的因数即可。

判素数的时候我用了Miller-Rabin，不知道直接暴力判能不能过。

用Miller-Rabin的时候要注意long long乘long long模long long

代码：

#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <vector>
#define ele long long
using namespace std;
#define maxn 10000010
ele K,p,k,m,pcnt,plst[maxn],tp[maxn];
bool flag[maxn];
vector<ele> ans;
inline ele mul(ele a,ele b,ele p){
	ele ans=0,tmp=a%p;
	for (; b; b>>=1,tmp=(tmp<<1)%p)
		if (b&1) ans=(ans+tmp)%p;
	return ans;
}
inline ele pw(ele a,ele x,ele p){
	ele ans=1,tmp=a%p;
	for (; x; x>>=1,tmp=mul(tmp,tmp,p))
		if (x&1) ans=mul(ans,tmp,p);
	return ans;
}
inline bool mr(ele n){
	ele T=3;
	while (T--){
		ele a=0;
		while (!a) a=rand()%n;
		if (pw(a,n-1,n)!=1) return false;
		for (ele k=(n-1)/2; !k&1; k>>=1){
			ele p=pw(a,k,n);
			if (p==n-1) break;
			if (p!=1) return false;
		}
	}
	return true;
}
void dfs(ele i,ele p,ele tmp){
	if (i==m || tp[i]-1>p){
		if (p==1) ans.push_back(tmp*K);
		return;
	}
	dfs(i+1,p,tmp);
	if (p%(tp[i]-1)==0){
		p/=(tp[i]-1); tmp*=tp[i];
		dfs(i+1,p,tmp);
		while (p%tp[i]==0){
			p/=tp[i]; tmp*=tp[i];
			dfs(i+1,p,tmp);
		}
	}
}
int main(){ 
	srand(2333);
	pcnt=0;
	memset(flag,0,sizeof(flag));
	for (int i=2; i<maxn; ++i){
		if (!flag[i]) plst[pcnt++]=i;
		for (int j=0; j<pcnt && plst[j]*i<maxn; ++j){
			flag[plst[j]*i]=true;
			if (i%plst[j]==0) break;
		}
	}
	scanf("%lld%lld",&p,&k);
	m=0;
	for (int i=0; i<pcnt; ++i)
		if (p%(plst[i]-1)==0) tp[m++]=plst[i];
	for (ele i=1; i*i<=p; ++i)
		if (p%i==0){
			K=p/i+1;
			if (K<maxn) break;
			if (mr(K)) dfs(0,i,1);
		}
	K=1;
	dfs(0,p,1);
	sort(ans.begin(),ans.end());
	for (int i=0; i<k-1; ++i) printf("%lld ",ans[i]);
	printf("%lld\n",ans[k-1]);
	return 0;
}
