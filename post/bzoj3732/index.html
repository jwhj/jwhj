bzoj3732题解

Dec 24, 2016 10:02 · 131 words · 1 minute read 题解 bzoj 图论 并查集 乱搞

题目其实就是要求最小瓶颈路，而最小瓶颈路一定在最小生成树上面，所以就可以转换为求树上两点间的最大边权。

求树上两点间的最大边权有很多种做法，比如倍增、lct（我不明白为什么会有人用这个）等。

然而其实这题有一种更简单的方法。

考虑kruskal算法求最小生成树的过程，每次加入一条边，连接两个联通块A、B。因为这条边的边权比已加入的任何一条边都要大，所以可以改变这条边的两个端点，只要端点仍然分别在A、B中，对结果就没有影响。

于是可以考虑像按大小合并、不路径压缩的并查集那样连边，这样最后得到的树高是$O(\log n)$的，所以暴力查询就可以了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 15010
#define maxm 30010
#define INF 1000000000
struct edge{
	ele u,v,w;
	inline bool operator<(edge b)const{
		return w<b.w;
	}
};
ele n,m,q,u[maxn],size[maxn],f[maxn],dep[maxn];
edge e[maxm];
ele getf(ele *u,ele x){
	return (x==u[x])?(x):(getf(u,u[x]));
}
ele d(ele *u,ele x){
	if (x==u[x])
		return dep[x]=0;
	if (dep[x]!=-1)
		return dep[x];
	return dep[x]=d(u,u[x])+1;
}
int main(){
	scanf("%d%d%d",&n,&m,&q);
	for (int i=0; i<m; ++i){
		scanf("%d%d%d",&e[i].u,&e[i].v,&e[i].w);
		--e[i].u,--e[i].v;
	}
	sort(e,e+m);
	for (int i=0; i<n; ++i)
		u[i]=i,size[i]=1;
	for (int i=0; i<m; ++i){
		ele a=getf(u,e[i].u),b=getf(u,e[i].v);
		if (a==b) continue;
		if (size[a]>size[b])
			swap(a,b);
		u[a]=b;
		size[b]+=size[a];
		f[a]=e[i].w;
	}
	memset(dep,-1,sizeof(dep));
	while (q--){
		ele x,y;
		scanf("%d%d",&x,&y);
		--x,--y;
		ele ans=0;
		if (d(u,x)>d(u,y))
			swap(x,y);
		while (dep[x]<dep[y])
			ans=max(ans,f[y]),y=u[y];
		while (x!=y){
			ans=max(ans,f[x]);
			ans=max(ans,f[y]);
			x=u[x],y=u[y];
		}
		printf("%d\n",ans);
	}
	return 0;
}
