bzoj1038题解

Aug 23, 2017 10:02 · 114 words · 1 minute read 题解 bzoj 三分

容易证明，在任意一段山坡内，瞭望塔需要的最低海拔是单峰的，另一方面，山坡的海拔是单峰的，而两个单峰函数加起来还是单峰函数，所以在任意一段山坡内，瞭望塔的高度是单峰的。这样的话就可以对每一段山坡进行三分了。

精度是个问题……我用了long double并把eps调到了$10^{-9}$才AC。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 310
#define eps 1e-9
ele n,cnt;
long double x[maxn],y[maxn];
inline long double ght(long double x1,long double y1,long double x2,long double y2,long double p){
	long double k=(y1-y2)/(x1-x2);
	return k*(p-x1)+y1;
}
inline long double f(long double p){
	long double ans=0;
	for (int i=0; i<n-1; ++i)
		ans=max(ans,ght(x[i],y[i],x[i+1],y[i+1],p));
	return ans-ght(x[cnt],y[cnt],x[cnt+1],y[cnt+1],p);
}
long double calc(long double l,long double r){
	if (r-l<eps) return f((l+r)/2);
	long double L=(r-l)/3,m1=f(l+L),m2=f(r-L);
	if (m1<m2) return calc(l,r-L); else return calc(l+L,r);
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%Lf",x+i);
	for (int i=0; i<n; ++i) scanf("%Lf",y+i);
	long double ans=1e12;
	for (cnt=0; cnt<n-1; ++cnt) ans=min(ans,calc(x[cnt],x[cnt+1]));
	printf("%.3Lf\n",ans);
	return 0;
}
