bzoj3744题解

Dec 1, 2016 10:02 · 240 words · 2 minutes read 题解 bzoj 分块 树状数组

如果这道题没有强制在线，那么用莫队算法就能水过去了。

不过莫队也是可以在线的。

先把整个序列离散化一下。

把整个序列分成$O(\sqrt{n})$块，维护一个f[i][j]，代表第i块到第j块的答案，一会再说这个怎么求。

对于每个询问[l,r]，考虑两种情况：

  • l，r在同一块或相邻的块中，暴力求答案就可以了

  • 否则，l，r中间整块的答案可以通过f数组直接知道，只需暴力考虑两边不超过$2\sqrt{n}$个数的贡献。

如果已经知道某个区间的答案，现在只需考虑如何在区间边上再加一个数。加上一个数x，对答案的贡献为原区间中大于/小于x的数的个数。于是可以维护$\sqrt{n}$个树状数组，f1[i]代表一个树状数组，储存0~i这i个块中每个元素出现多少次。每次加一个数x，中间块里面比x大/小的数的个数就可以$O(\log n)$求出来。再用一个树状数组维护边上的数的出现次数就可以了。

我之前的写法内存消耗太大，结果在bzoj上总是0ms TLE，调了一整晚才发现是内存问题。以后得记住bzoj上0ms TLE是MLE导致的。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 50010
#define maxs 250
struct BIT{
	inline void alloc(ele n){
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele k){
		for (int j=i; j<=n; j+=j&(-j))
			a[j]+=k;
	}
	inline ele query(ele i){
		ele ans=0;
		for (int j=i; j>0; j-=j&(-j))
			ans+=a[j];
		return ans;
	}
	inline ele query(ele l,ele r){
		if (l>r) return 0;
		return query(r)-query(l-1);
	}
	ele n,a[maxn];
};
ele n,m,len,S,a[maxn],arr[maxn],f[maxs][maxs];
BIT b,f1[maxs];
inline void get(ele&a){
	char c;
	while ((c=getchar())<'0' || c>'9')
		;
	a=c-'0';
	while ((c=getchar())>='0' && c<='9')
		a*=10,a+=c-'0';
}
int main(){
	get(n);
	for (int i=0; i<n; ++i)
		get(a[i]),arr[i]=a[i];
	get(m);
	sort(arr,arr+n);
	len=unique(arr,arr+n)-arr;
	for (int i=0; i<n; ++i)
		a[i]=lower_bound(arr,arr+len,a[i])-arr+1;
	S=sqrt(n);
	for (int i=0; i*S<n; ++i){
		ele j=i*S;
		ele tmp=0;
		b.alloc(n);
		for (int k=j; k<n; ++k){
			b.add(a[k],1);
			tmp+=b.query(a[k]+1,n);
			if ((k+1)%S==0 || k==n-1)
				f[i][k/S]=tmp;
		}
		f1[i].alloc(n);
		for (int k=j; k<n && k<j+S; ++k)
			f1[i].add(a[k],1);
	}
	for (int i=1; i*S<n; ++i)
		for (int j=1; j<=n; ++j)
			f1[i].a[j]+=f1[i-1].a[j];
	ele ans=0;
	b.alloc(n);
	while (m--){
		ele l,r;
		get(l); get(r);
		l^=ans; r^=ans;
		--l,--r;
		ans=0;
		if (l/S+1>=r/S){
			for (int i=l; i<=r; ++i){
				b.add(a[i],1);
				ans+=b.query(a[i]+1,n);
			}
			for (int i=l; i<=r; ++i)
				b.add(a[i],-1);
		}
		else{
			ele i1=l/S,i2=r/S;
			ans+=f[i1+1][i2-1];
			for (int i=i2*S; i<=r; ++i){
				ans+=f1[i2-1].query(a[i]+1,n);
				ans-=f1[i1].query(a[i]+1,n);
				b.add(a[i],1);
				ans+=b.query(a[i]+1,n);
			}
			for (int i=i1*S+S-1; i>=l; --i){
				ans+=f1[i2-1].query(1,a[i]-1);
				ans-=f1[i1].query(1,a[i]-1);
				b.add(a[i],1);
				ans+=b.query(1,a[i]-1);
			}
			for (int i=i2*S; i<=r; ++i)
				b.add(a[i],-1);
			for (int i=i1*S+S-1; i>=l; --i)
				b.add(a[i],-1);
		}
		printf("%d\n",ans);
	}
	return 0;
}
