bzoj1064题解

Aug 27, 2017 10:02 · 149 words · 1 minute read 题解 bzoj 图论

注意到，虽然题目中给出a能看到b的关系是有向的，但因为面具种类可以顺着这个边正推和倒推，所以其实具有一些无向的边的性质。

把(a,b)拆成a->b，边权为1的边和b->a，边权为-1的边，那么如果图中有环，k一定是环长的因数。

如果整个图都没有环，那么k最大值就是每个联通块最长链的长度之和，这里一条链的长度定义为边权和+1。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
#define ele int
using namespace std;
#define maxn 100010
#define maxm 1000010
#define fi first
#define se second
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxm<<1],*ecnt;
ele n,m,ans,d[maxn],mx,mn;
bool vis[maxn];
edge *h[maxn];
set<pair<ele,ele> > S;
inline void addedge(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
inline ele _abs(ele x){
	return (x<0)?-x:x;
}
ele gcd(ele a,ele b){
	return b?gcd(b,a%b):a;
}
void dfs(ele p,ele i){
	vis[i]=true;
	mx=max(mx,d[i]); mn=min(mn,d[i]);
	for (edge *j=h[i]; j; j=j->nxt)
		if (!vis[j->v]){
			d[j->v]=d[i]+j->w;
			dfs(i,j->v);
		}
	else ans=gcd(ans,_abs(d[i]-d[j->v]+j->w));
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=ep; memset(h,0,sizeof(h));
	while (m--){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		if (S.count(make_pair(u,v))) continue;
		S.insert(make_pair(u,v));
		addedge(u,v,1); addedge(v,u,-1);
	}
	ans=0;
	memset(vis,0,sizeof(vis));
	ele s=0;
	for (int i=0; i<n; ++i)
		if (!vis[i]){
			d[i]=0; mx=mn=0;
			dfs(-1,i);
			s+=mx-mn+1;
		}
	ele ans1=3;
	if (ans)
		for (; ans1<=ans && ans%ans1; ++ans1);
	else ans=s;
	if (ans>=3) printf("%d %d\n",ans,ans1);
	else puts("-1 -1");
	return 0;
}
