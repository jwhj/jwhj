bzoj2962题解

Dec 8, 2016 10:02 · 300 words · 2 minutes read 题解 bzoj 线段树

操作3看上去很可怕，但是其实这个操作是可以区间合并的。用$f_i$表示选i个数的所有乘积的和，那么有$f_i=\sum \mathrm{lc.}fj\times\mathrm{rc.}f{i-j}$。

操作2也很容易，对于所有奇的i，给$f_i$取相反数就可以了。

操作1自己推一下可以得到，如果加上x，那么$f’_i=\sum f_{i-j}C_{\mathrm{len}-i+j}^{j}x^j$这里len代表区间的长度。

这么一道题我调了一晚上，最近代码能力大幅下降啊。以后一定要记住，给线段树打标记的时候一定要弄清楚各个标记之间、标记和维护的信息之间的先后顺序。

另外数组好像要开大一点才能过？

代码：

#include <cstdio>
#include <cstring>
#define ele int
#define ll long long
using namespace std;
#define MOD 19940417
#define maxn 60010
#define maxc 21
struct ANS{
	ele f[maxc];
};
struct node{
	ANS a;
	ele add;
	bool neg;
	node *l,*r;
}np[maxn<<2];
ele n,q,ncnt,a[maxn],C[maxn][maxc];
node *root;
inline ANS operator+(ANS a,ANS b){
	ANS c;
	for (int i=0; i<maxc; ++i){
		c.f[i]=0;
		for (int j=0; j<=i; ++j){
			c.f[i]+=(ll)a.f[j]*b.f[i-j]%MOD;
			c.f[i]%=MOD;
		}
	}
	return c;
}
inline void pushdown(node *x,ele a,ele b){
	if (x->neg){
		for (int i=1; i<maxc; i+=2)
			x->a.f[i]=(MOD-x->a.f[i])%MOD;
		if (x->l)
			x->l->neg=!x->l->neg,x->r->neg=!x->r->neg,
			x->l->add=(MOD-x->l->add)%MOD,
			x->r->add=(MOD-x->r->add)%MOD;
		x->neg=false;
	}
	if (x->add){
		for (int i=maxc-1; i>=0; --i){
			ele tmp=x->add;
			for (int j=1; j<=i; ++j,tmp=(ll)tmp*x->add%MOD){
				x->a.f[i]+=(ll)tmp*x->a.f[i-j]%MOD*C[b-a+1-i+j][j]%MOD;
				x->a.f[i]%=MOD;
			}
		}
		if (x->l)
			x->l->add+=x->add,x->r->add+=x->add,
			x->l->add%=MOD,x->r->add%=MOD;
		x->add=0;
	}
}
inline void maintain(node *x,ele a,ele b){
	if (x->l){
		ele mid=a+b>>1;
		pushdown(x,a,b);
		pushdown(x->l,a,mid);
		pushdown(x->r,mid+1,b);
		x->a=x->l->a+x->r->a;
	}
}
node* build(ele l,ele r){
	node *p=&np[ncnt++];
	p->add=0; p->neg=false;
	if (l==r){
		memset(p->a.f,0,sizeof(p->a.f));
		p->a.f[0]=1; p->a.f[1]=(a[l-1]%MOD+MOD)%MOD;
		p->l=p->r=NULL;
	}
	else{
		ele mid=l+r>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		maintain(p,l,r);
	}
	return p;
}
void upd1(node *x,ele a,ele b,ele l,ele r,ele k){
	pushdown(x,a,b);
	if (l<=a && b<=r)
		x->add=k,pushdown(x,a,b);
	else{
		ele mid=a+b>>1;
		if (l<=mid)
			upd1(x->l,a,mid,l,r,k);
		if (mid<r)
			upd1(x->r,mid+1,b,l,r,k);
		maintain(x,a,b);
	}
}
void upd2(node *x,ele a,ele b,ele l,ele r){
	pushdown(x,a,b);
	if (l<=a && b<=r)
		x->neg=true,pushdown(x,a,b);
	else{
		ele mid=a+b>>1;
		if (l<=mid)
			upd2(x->l,a,mid,l,r);
		if (mid<r)
			upd2(x->r,mid+1,b,l,r);
		maintain(x,a,b);
	}
}
ANS query(node *x,ele a,ele b,ele l,ele r){
	pushdown(x,a,b);
	if (l<=a && b<=r)
		return x->a;
	else{
		ele mid=a+b>>1;
		if (r<=mid)
			return query(x->l,a,mid,l,r);
		if (mid<l)
			return query(x->r,mid+1,b,l,r);
		ANS lans=query(x->l,a,mid,l,r);
		ANS rans=query(x->r,mid+1,b,l,r);
		return lans+rans;
	}
}
int main(){
	scanf("%d%d",&n,&q);
	for (int i=0; i<n; ++i)
		scanf("%d",a+i);
	memset(C,0,sizeof(C));
	for (int i=0; i<=n; ++i){
		C[i][0]=C[i][i]=1;
		for (int j=1; j<i && j<maxc; ++j)
			C[i][j]=(C[i-1][j-1]+C[i-1][j])%MOD;
	}
	ncnt=0;
	root=build(1,n);
	while (q--){
		char s[10];
		ele u,v,w;
		scanf("%s%d%d",s,&u,&v);
		if (s[0]!='R')
			scanf("%d",&w);
		if (s[0]=='I')
			upd1(root,1,n,u,v,(w%MOD+MOD)%MOD);
		else if (s[0]=='R')
			upd2(root,1,n,u,v);
		else if (s[0]=='Q'){
			printf("%d\n",query(root,1,n,u,v).f[w]%MOD);
		}
	}
	return 0;
}
