bzoj4540题解

Feb 12, 2017 10:02 · 217 words · 2 minutes read 题解 bzoj 统计 莫队 分块 线段树

记a[i]左边第一个比a[i]小的数的位置为prv[i]。

考虑莫队，将[l,r-1]变为[l,r]的时候，对于$i\in [\mathrm{prv}[r]+1,r]$，[i,r]的最小值为a[r]。对于$i\in [\mathrm{prv}[\mathrm{prv}[r]]+1,\mathrm{prv}[r]]$，[i,r]的最小值为a[prv[r]]……

在左边加入一个数的情况类似。

我比较懒就写了个在线莫队，不需要考虑删除的情况。

为了快速算出这个东西，我一开始写了一个倍增，结果复杂度比较劲，是$O(n\sqrt{n}\log n)$，我写了一个交上去TLE了。接下来我玩了很长一段时间的常数，感觉不大可能卡过去，就开始想怎么把那个log消掉。

注意到，从r开始，按照prv往前跳，设跳到l前面之前的最后一个位置为i，那么a[i]为[l,r]上的最小值。

这样处理一个类似前缀和的东西，再用st表把i找出来，就可以做到$O(n\sqrt{n})$。

据说有线段树的做法，晚点去研究一下。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <cctype>
#include <algorithm>
#define ele int
#define ll long long
using namespace std;
#define maxn 100010
#define maxk 17
#define maxs 320
ele n,q,S,top,a[maxn],prv[maxn],nxt[maxn],stk[maxn],st[maxn][maxk],LOG[maxn];
ll f[maxs][maxn],s1[maxn],s2[maxn];
inline void get(ele&a){
	char c;
	while (c!='-' && !isdigit(c=getchar()));
	bool flag=(c=='-');
	a=(flag)?(0):(c-'0');
	while (isdigit(c=getchar())) a=(a<<3)+(a<<1)+c-'0';
	a*=(flag)?(-1):(1);
}
inline ele rmq(ele l,ele r){
	ele L=LOG[r-l+1];
	return (a[st[l][L]]<a[st[r-(1<<L)+1][L]])?(st[l][L]):(st[r-(1<<L)+1][L]);
}
inline ll addl(ele i,ele j){
	ele k=rmq(j,i);
	return s2[j]-s2[k]+(ll)a[k]*(i-k+1);
}
inline ll addr(ele i,ele j){
	ele k=rmq(i,j);
	return s1[j]-s1[k]+(ll)a[k]*(k-i+1);
}
int main(){
	get(n); get(q);
	S=sqrt(n);
	for (int i=0; i<n; ++i) get(a[i]);
	stk[0]=0; top=1; prv[0]=-1; s1[0]=a[0];
	for (int i=1; i<n; ++i){
		while (top && a[stk[top-1]]>a[i]) --top;
		prv[i]=(top)?(stk[top-1]):(-1);
		s1[i]=(ll)a[i]*(i-prv[i]);
		s1[i]+=(top)?(s1[prv[i]]):(0);
		stk[top++]=i;
	}	
	stk[0]=n-1; top=1; nxt[n-1]=n; s2[n-1]=a[n-1];
	for (int i=n-2; i>=0; --i){
		while (top && a[stk[top-1]]>a[i]) --top;
		nxt[i]=(top)?(stk[top-1]):(n);
		s2[i]=(ll)a[i]*(nxt[i]-i);
		s2[i]+=(top)?(s2[nxt[i]]):(0);
		stk[top++]=i;
	}
	LOG[1]=0;
	for (int i=2; i<=n; ++i) LOG[i]=LOG[i>>1]+1;
	for (int i=0; i<n; ++i) st[i][0]=i;
	for (int j=1; j<maxk; ++j)
		for (int i=0; i<n; ++i)
			if (i+(1<<j-1)>=n || a[st[i][j-1]]<a[st[i+(1<<j-1)][j-1]])
				st[i][j]=st[i][j-1];
			else st[i][j]=st[i+(1<<j-1)][j-1];
	for (int i=0,j=0; i<n; i+=S,++j){
		f[j][i]=a[i];
		for (int k=i+1; k<n; ++k)
			f[j][k]=f[j][k-1]+addr(i,k);
	}
	while (q--){
		ele l,r;
		ll ans;
		get(l); get(r);
		--l,--r;
		if (l/S==r/S){
			ans=a[l];
			for (int i=l+1; i<=r; ++i)
				ans+=addr(l,i);
		}
		else{
			ans=f[l/S+1][r];
			for (int i=(l/S+1)*S-1; i>=l; --i)
				ans+=addl(r,i);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
