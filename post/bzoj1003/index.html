bzoj1003题解

Dec 14, 2016 10:02 · 170 words · 1 minute read 题解 bzoj spfa DP

用SPFA预处理出t[i][j]，表示第i天到第j天走同一条路的话这条路最短是多少，然后就可以DP了。

要注意INF不能开得太大，因为后面可能要把t[i][j]乘上一个数，如果开得太大INF乘一个数会溢出。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#define ele long long
using namespace std;
#define maxn 110
#define maxm 30
#define INF 1000000000000LL
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxm*maxm*2];
ele n,m,K,e,ecnt,f[maxn],t[maxn][maxn];
bool flag[maxn][maxm];
edge *g[maxm];
inline void addedge(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->w=w; p->nxt=g[u];
	g[u]=p;
}
inline ele spfa(ele a,ele b){
	static bool oc[maxm];
	memset(oc,0,sizeof(oc));
	for (int i=a; i<=b; ++i)
		for (int j=0; j<m; ++j)
			oc[j]=oc[j] || flag[i][j];
	static ele d[maxm];
	for (int i=1; i<m; ++i)
		d[i]=INF;
	d[0]=0;
	static queue<ele> q;
	q.push(0);
	while (!q.empty()){
		ele k=q.front(); q.pop();
		for (edge *j=g[k]; j; j=j->nxt)
			if (!oc[j->v] && d[j->v]>d[k]+j->w){
				d[j->v]=d[k]+j->w;
				q.push(j->v);
			}
	}
	return d[m-1];
}
int main(){
	scanf("%lld%lld%lld%lld",&n,&m,&K,&e);
	ecnt=0;
	memset(g,0,sizeof(g));
	for (int i=0; i<e; ++i){
		ele a,b,c;
		scanf("%lld%lld%lld",&a,&b,&c);
		--a,--b;
		addedge(a,b,c); addedge(b,a,c);
	}
	ele D;
	scanf("%lld",&D);
	memset(flag,0,sizeof(flag));
	while (D--){
		ele a,b,p;
		scanf("%lld%lld%lld",&p,&a,&b);
		--p,--a,--b;
		for (int i=a; i<=b; ++i)
			flag[i][p]=true;
	}
	for (int i=0; i<n; ++i)
		for (int j=i; j<n; ++j)
			t[i][j]=spfa(i,j);
	for (int i=0; i<n; ++i){
		f[i]=t[0][i]*(i+1);
		for (int j=i-1; j>=0; --j)
			f[i]=min(f[i],f[j]+t[j+1][i]*(i-j)+K);
	}
	printf("%lld\n",f[n-1]);
	return 0;
}
