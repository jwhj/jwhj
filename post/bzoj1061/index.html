bzoj1061题解

Jun 3, 2016 10:02 · 159 words · 1 minute read 题解 bzoj 网络流 线性规划

正解是网络流，然而我智商下线不会建图，所以就写了线性规划。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#define ele int
using namespace std;
#define maxn 1010
#define maxm 10010
#define eps 1e-7
#define inf 1e19
namespace LP
{
	double a[maxm][maxn],b[maxm],c[maxn],v;
	ele n,m;
	inline void pivot(ele p,ele k)
	{
		double tmp=a[p][k]; a[p][k]=1;
		b[p]/=tmp;
		for (int i=0; i<m; ++i) a[p][i]/=tmp;
		tmp=c[k]; c[k]=0;
		v+=tmp*b[p];
		for (int i=0; i<m; ++i) c[i]-=tmp*a[p][i];
		for (int i=0; i<n; ++i)
			if (i!=p && fabs(a[i][k])>eps)
			{
				tmp=a[i][k]; a[i][k]=0;
				b[i]-=tmp*b[p];
				for (int j=0; j<m; ++j) a[i][j]-=tmp*a[p][j];
			}
	}
	inline double simplex()
	{
		while (1)
		{
			ele k;
			for (k=0; k<m; ++k)
				if (c[k]>eps) break;
			if (k==m) return v;
			double tmp=inf;
			ele p;
			for (int i=0; i<n; ++i)
				if (a[i][k]>eps && tmp>b[i]/a[i][k])
					tmp=b[i]/a[i][k],p=i;
			if (tmp+eps>=inf) return inf;
			pivot(p,k);
		}
	}
}
ele n,m,a[maxn],s[maxm],t[maxm],c[maxm];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i) scanf("%d",a+i),LP::c[i]=a[i];
	for (int i=0; i<m; ++i)
		scanf("%d%d%d",s+i,t+i,c+i),--s[i],--t[i],LP::b[i]=c[i];
	LP::n=m; LP::m=n; LP::v=0;
	memset(LP::a,0,sizeof(LP::a));
	for (int i=0; i<m; ++i)
		for (int j=s[i]; j<=t[i]; ++j) LP::a[i][j]=1;
	printf("%lld\n",(long long)(LP::simplex()+0.5));
	return 0;
}
