bzoj1093题解

Aug 16, 2017 10:02 · 199 words · 1 minute read 题解 bzoj tarjan 半联通图

首先要知道一个性质：任何一个半联通的DAG一定存在一条链遍历所有点。

把原图强联通缩点之后，就是要在DAG上求带全最长链及其条数，dp就可以了。

缩点完了要注意处理重边的问题。注意seq[i]不要打成i。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
#define ele int
using namespace std;
#define maxn 100010
#define maxm 1000010
struct edge{
	ele v;
	edge *nxt;
}ep[maxm<<1],*ecnt;
ele n,m,MOD,bcnt,top,tcnt,scnt,blg[maxn],stk[maxn],dfn[maxn],low[maxn],
seq[maxn],w[maxn],f[maxn],g[maxn];
bool instack[maxn],vis[maxn];
edge *h[maxn],*h1[maxn];
set<pair<ele,ele> > S;
inline void addedge(edge **h,ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void tarjan(ele p,ele i){
	dfn[i]=low[i]=tcnt++;
	instack[i]=true; stk[top++]=i;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!~dfn[j->v]){
			tarjan(i,j->v);
			low[i]=min(low[i],low[j->v]);
		}
		else if (instack[j->v]) low[i]=min(low[i],dfn[j->v]);
	if (dfn[i]==low[i]){
		do{
			blg[stk[top-1]]=bcnt;
			instack[stk[--top]]=false;
		}while (stk[top]!=i);
		++bcnt;
	}
}
void dfs(ele p,ele i){
	vis[i]=true;
	for (edge *j=h1[i]; j; j=j->nxt)
		if (!vis[j->v]) dfs(i,j->v);
	seq[scnt--]=i;
}
int main(){
	scanf("%d%d%d",&n,&m,&MOD);
	ecnt=ep; memset(h,0,sizeof(h));
	for (int i=0; i<m; ++i){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		addedge(h,u,v);
	}
	top=tcnt=bcnt=0; memset(dfn,-1,sizeof(dfn)); memset(instack,0,sizeof(instack));
	for (int i=0; i<n; ++i)
		if (!~dfn[i]) tarjan(-1,i);
	memset(h1,0,sizeof(h1)); memset(w,0,sizeof(w));
	for (int i=0; i<n; ++i){
		++w[blg[i]];
		for (edge *j=h[i]; j; j=j->nxt)
			if (blg[i]!=blg[j->v] && !S.count(make_pair(blg[i],blg[j->v])))
				addedge(h1,blg[i],blg[j->v]),S.insert(make_pair(blg[i],blg[j->v]));
	}
	memset(vis,0,sizeof(vis));
	scnt=bcnt-1;
	for (int i=0; i<bcnt; ++i)
		if (!vis[i]) dfs(-1,i);
	for (int i=bcnt-1; ~i; --i){
		f[seq[i]]=0; S.clear();
		for (edge *j=h1[seq[i]]; j; j=j->nxt)
			f[seq[i]]=max(f[seq[i]],f[j->v]);
		g[seq[i]]=0; S.clear();
		for (edge *j=h1[seq[i]]; j; j=j->nxt)
			if (f[seq[i]]==f[j->v]) (g[seq[i]]+=g[j->v])%=MOD;
		f[seq[i]]+=w[seq[i]];
		(g[seq[i]]+=(!h1[seq[i]]))%=MOD;
	}
	ele mx=0,cnt=0;
	for (int i=0; i<bcnt; ++i)
		if (f[i]>mx) mx=f[i],cnt=g[i];
		else if (f[i]==mx) (cnt+=g[i])%=MOD;
	printf("%d\n%d\n",mx,cnt);
	return 0;
}
