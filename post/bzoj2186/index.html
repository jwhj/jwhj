bzoj2186题解

May 5, 2016 10:02 · 135 words · 1 minute read 题解 bzoj 数学

当$n=m$时，答案显然是$\phi(m!)$。

由于gcd(a+b,b)=gcd(a,b)，当$n>m$时，由于$m!|n!$，我们可以把$n!$分成若干组： $$1,2,\ldots,m!$$ $$m!+1,m!+2,\ldots,2\times m!$$ $$\ldots$$

每一组中，显然都有$\phi(m!)$种真货币。

所以答案为$\phi(m!)\times\frac{n!}{m!}=n!\times \Pi\frac{p_i-1}{p_i}$，这里$p_i$是小于$m$的质数。

这道题相当卡常，我用了读入优化，没用long long才过。

似乎求逆元ex_gcd比快速幂更快。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 10000010
ele T,R,n,m,fac[maxn],pcnt,plist[maxn],a[maxn];
bool flag[maxn];
inline void get(ele &a){
    char c;
    while ((c=getchar())<'0' || c>'9');
    a=c-'0';
    while ((c=getchar())>='0' && c<='9')
        a*=10,a+=c-'0';
}
void exgcd(ele a,ele b,ele &x,ele &y){
    if(b==0){x=1;y=0;return;}
    exgcd(b,a%b,x,y);
    ele t=x;x=y;y=t-a/b*y;
}
inline ele inv(ele t){
    ele x,y;
    exgcd(t,R,x,y);
    return (x%R+R)%R;
}
int main(){
    get(T); get(R);
    fac[0]=1;
    for (int i=1; i<maxn; ++i) fac[i]=(long long)fac[i-1]*i%R;
    pcnt=0;
    memset(flag,0,sizeof(flag));
    for (int i=2; i<maxn; ++i){
        if (!flag[i]) plist[pcnt++]=i;
        for (int j=0; j<pcnt && (long long)i*plist[j]<maxn; ++j){
            flag[(long long)i*plist[j]]=true;
            if (i%plist[j]==0) break;
        }
    }
    a[1]=1; a[2]=inv(2);
    for (int i=1; i<pcnt; ++i){
        ele k=plist[i];
        a[k]=((long long)a[k-1]*(k-1)%R)*inv(k)%R;
        ele r=(i==pcnt-1)?(maxn):(plist[i+1]);
        for (int j=k+1; j<r; ++j) a[j]=a[k];
    }
    while (T--){
        get(n); get(m);
        printf("%d\n",(long long)fac[n]*a[m]%R);
    }
    return 0;
}
