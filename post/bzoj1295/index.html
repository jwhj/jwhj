bzoj1295题解

Sep 17, 2016 10:02 · 146 words · 1 minute read 题解 bzoj spfa

搬走T个障碍后X能走到Y等价于存在一条X到Y的路径，上面有不超过T个障碍。

枚举X，用spfa求出X能通过不超过T个障碍走到的所有位置，并找出离X最远的那个，更新答案。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 40
#define INF 1e9
const ele di[]={-1,1,0,0};
const ele dj[]={0,0,-1,1};
struct crd{
	ele i,j;
};
ele n,m,T;
bool a[maxn][maxn];
inline ele bfs(ele x,ele y){
	static crd q[maxn];
	static ele d[maxn][maxn];
	ele qh=-1,qt=0,ans=0;
	q[0]=(crd){x,y};
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j) d[i][j]=INF;
	d[x][y]=(a[x][y])?(1):(0);
	while (qh<qt){
		++qh;
		ele i=q[qh].i,j=q[qh].j;
		if (d[i][j]>T) continue;
		for (int w=0; w<4; ++w){
			ele ni=i+di[w],nj=j+dj[w];
			if (0<=ni && ni<n && 0<=nj && nj<m){
				ele nd=(a[ni][nj])?(1):(0);
				nd+=d[i][j];
				if (nd<d[ni][nj]){
					d[ni][nj]=nd;
					q[++qt]=(crd){ni,nj};
					if (nd<=T){
						ele tmp=(x-ni)*(x-ni)+(y-nj)*(y-nj);
						ans=max(ans,tmp);
					}
				}
			}
		}
	}
	return ans;
}
int main(){
	scanf("%d%d%d",&n,&m,&T);
	for (int i=0; i<n; ++i){
		char s[maxn];
		scanf("%s",s);
		for (int j=0; j<m; ++j)
			if (s[j]=='0') a[i][j]=false;
			else a[i][j]=true;
	}
	ele ans=0;
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			ans=max(ans,bfs(i,j));
	printf("%.6f\n",sqrt(ans));
	return 0;
}
