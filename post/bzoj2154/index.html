bzoj2154题解

Jan 7, 2017 10:02 · 137 words · 1 minute read 题解 bzoj 数学 mobius

记$s(i,j)=\sum_{k=i}^j k$。

记$f(n,m)=\sum\sum ij[(i,j)=1]$，那么有 $$\begin{align*}f(n,m)&=\sum\sum ij\sum_{d|i,d|j}\mu(d)\\&=\sum d^2\mu(d)s(1,\lfloor\frac{n}{d}\rfloor)s(1,\lfloor\frac{m}{d}\rfloor)\end{align*}$$

答案为 $$\begin{align*}\sum\frac{ij}{(i,j)}&=\sum\frac{\sum ij[(i,j)=p]}{p}\\&=\sum\frac{\sum p^2\frac{i}{p}\frac{j}{p}[(i,j)=p]}{p}\\&=\sum p\sum\frac{i}{p}\frac{j}{p}[(i,j)=p]\\&=\sum p\sum\frac{i}{p}\frac{j}{p}[(\frac{i}{p},\frac{j}{p})=1]\\&=\sum p\times f(\lfloor\frac{n}{p}\rfloor,\lfloor\frac{m}{p}\rfloor)\end{align*}$$

然后再用mobius反演从$O(n)$优化到$O(\sqrt{n})$的方法优化一下就行了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define ll long long
using namespace std;
#define MOD 20101009
#define maxn 10000010
ele n,m,pcnt,plist[maxn],mu[maxn];
bool flag[maxn];
inline ele lsum(ele i,ele j){
	return (ll)(i+j)*(j-i+1)/2%MOD;
}
inline ele calc(ele n,ele m){
	ele ans=0;
	for (int i=min(n,m); i;){
		ele p=n/(n/i+1),p1=m/(m/i+1);
		p=max(p,p1);
		ele tmp=(ll)lsum(1,n/i)*lsum(1,m/i)%MOD;
		tmp=(ll)tmp*((mu[i]-mu[p])%MOD)%MOD;
		tmp%=MOD;
		ans+=tmp;
		ans%=MOD;
		i=p;
	}
	return ans;
}
int main(){
	pcnt=0;
	mu[0]=0;
	mu[1]=1;
	memset(flag,0,sizeof(flag));
	for (int i=2; i<maxn; ++i){
		if (!flag[i])
			plist[pcnt++]=i,mu[i]=-1;
		for (int j=0; j<pcnt && i*plist[j]<maxn; ++j){
			flag[i*plist[j]]=true;
			if (i%plist[j])
				mu[i*plist[j]]=-mu[i];
			else{
				mu[i*plist[j]]=0;
				break;
			}
		}
	}
	for (int i=2; i<maxn; ++i){
		mu[i]=(ll)mu[i]*i%MOD*i%MOD;
		mu[i]+=mu[i-1];
		mu[i]%=MOD;
	}
	scanf("%d%d",&n,&m);
	ele ans=0;
	for (int i=min(n,m); i;){
		ele p=n/(n/i+1),p1=m/(m/i+1);
		p=max(p,p1);
		ans+=(ll)calc(n/i,m/i)*lsum(p+1,i)%MOD;
		ans%=MOD;
		i=p;
	}
	if (ans<0)
		ans+=MOD;
	printf("%d\n",ans);
	return 0;
}
