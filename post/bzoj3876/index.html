bzoj3876题解

Apr 24, 2017 10:02 · 169 words · 1 minute read 题解 bzoj 网络流

关于带上下界的网络流，这里有一份比较好的教程。

今天才发现我的spfa写得过于naive……要用一个数组记录某个点是否在队列中……我以前为什么一直没发现……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
#define maxn 310
#define INF 1000000000
struct edge{
	ele v,c,r;
	edge *nxt,*rev;
}ep[10000000],*ecnt;
edge *g[maxn];
inline void addedge1(ele u,ele v,ele c,ele r){
	edge *p=ecnt++;
	p->v=v; p->c=c; p->r=r; p->nxt=g[u]; p->rev=ep+((ecnt-ep-1)^1);
	g[u]=p;
}
inline void addedge(ele u,ele v,ele c,ele r){
	addedge1(u,v,c,r); addedge1(v,u,-c,0);
}
ele n;
edge *pv[maxn];
ele dfs(ele i,ele&p){
	if (!pv[i]) return 0;
	p=min(p,pv[i]->rev->r);
	ele ans=dfs(pv[i]->v,p);
	ans+=pv[i]->rev->c*p;
	pv[i]->r+=p; pv[i]->rev->r-=p;
	return ans;
}
inline ele mincost(ele s,ele t){
	ele ans=0;
	static ele d[maxn];
	static queue<ele> Q;
	static bool vis[maxn];
	while (1){
		memset(d,-1,sizeof(d));
		d[s]=0;
		Q.push(s);
		pv[s]=NULL;
		memset(vis,0,sizeof(vis));
		while (!Q.empty()){
			ele k=Q.front(); Q.pop();
			vis[k]=false;
			for (edge *j=g[k]; j; j=j->nxt)
				if (j->r && (!~d[j->v] || d[k]+j->c<d[j->v])){
					d[j->v]=d[k]+j->c;
					pv[j->v]=j->rev;
					if (!vis[j->v]) vis[j->v]=1,Q.push(j->v);
				}
		}
		if (!~d[t]) break;
		ele p=INF; ans+=dfs(t,p);
	}
	return ans;
}
int main(){
	scanf("%d",&n);
	ecnt=ep; memset(g,0,sizeof(g));
	for (int i=0; i<n; ++i){
		ele c;
		scanf("%d",&c);
		for (int j=0; j<c; ++j){
			ele v,w;
			scanf("%d%d",&v,&w); --v;
			addedge(i,v,w,INF);
			addedge(n,v,w,1);
		}
		addedge(i,n+1,0,c);
		if (i) addedge(i,0,0,INF);
	}
	printf("%d\n",mincost(n,n+1));
	return 0;
}
