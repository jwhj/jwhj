bzoj1059题解

Dec 18, 2016 10:02 · 119 words · 1 minute read 题解 bzoj 二分图

两个同行的黑格，在变换之后还是同行。两个同列的黑格，在变换之后还是同列。

所以，如果不存在n个不同行不同列的黑格，则无解。

下证明如果存在n个不同行不同列的黑格则有解。

首先可以把其中一个黑格移到左上角。

接着只需要不移动第一行和第一列，就能把下一个黑格移到(2,2)。

接着只需要不移动前两行和前两列，就能把下一个黑格移到(3,3)。

以此类推。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 210
struct edge{
	ele v;
	edge *nxt;
}ep[maxn*maxn];
ele n,ecnt,link[maxn];
bool flag[maxn];
edge *g[maxn];
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=g[u];
	g[u]=p;
}
bool dfs(ele i){
	for (edge *j=g[i]; j; j=j->nxt)
		if (!flag[j->v]){
			flag[j->v]=true;
			if (link[j->v]==-1 || dfs(link[j->v])){
				link[j->v]=i;
				return true;
			}
		}
	return false;
}
int main(){
	ele T;
	scanf("%d",&T);
	while (T--){
		scanf("%d",&n);
		ecnt=0;
		memset(g,0,sizeof(g));
		for (int i=0; i<n; ++i)
			for (int j=0; j<n; ++j){
				ele x;
				scanf("%d",&x);
				if (x)
					addedge(i,j);
			}
		bool f=true;
		memset(link,-1,sizeof(link));
		for (int i=0; i<n; ++i){
			memset(flag,0,sizeof(flag));
			if (!dfs(i)){
				f=false;
				break;
			}
		}
		if (f)
			puts("Yes");
		else puts("No");
	}
	return 0;
}
