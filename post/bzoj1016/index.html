bzoj1016题解

Dec 17, 2016 10:02 · 190 words · 1 minute read 题解 bzoj 图论 乱搞

这里有很详细的题解。

感觉里面的“且一定有一条边$b_j$不在$B$中”要改为“且一定有一条边$b_j$不在$A$中”。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 110
#define maxm 1010
#define MOD 31011
struct edge{
    ele u,v,w;
    inline bool operator<(edge b)const{
        return w<b.w;
    }
};
ele n,m,len,arr[maxm],cnt[maxm],u[maxn];
edge e[maxm];
inline void get(ele&a){
    char c;
    while ((c=getchar())<'0' || c>'9');
    a=c-'0';
    while ((c=getchar())>='0' && c<='9')
        a*=10,a+=c-'0';
}
inline ele idx(ele x){
    return lower_bound(arr,arr+len,x)-arr;
}
ele getf(ele *u,ele x){
    return (x==u[x])?(x):(u[x]=getf(u,u[x]));
}
inline void uni(ele *u,ele x,ele y){
    ele a=getf(u,x),b=getf(u,y);
    if (a!=b) u[a]=b;
}
ele dfs(ele left,ele j,ele k){
    if (!left) return 1;
    if (j==k) return 0;
    ele ans=dfs(left,j+1,k);
    if (getf(u,e[j].u)!=getf(u,e[j].v)){
        ele tmp[maxn];
        memcpy(tmp,u,sizeof(ele)*n);
        uni(u,e[j].u,e[j].v);
        ans+=dfs(left-1,j+1,k);
        ans%=MOD;
        memcpy(u,tmp,sizeof(ele)*n);
    }
    return ans;
}
int main(){
    get(n); get(m);
    for (int i=0; i<m; ++i){
        get(e[i].u); get(e[i].v); get(e[i].w);
        --e[i].u,--e[i].v;
        arr[i]=e[i].w;
    }
    sort(arr,arr+m);
    len=unique(arr,arr+m)-arr;
    sort(e,e+m);
    memset(cnt,0,sizeof(cnt));
    ele j=0;
    for (int i=0; i<n; ++i)
        u[i]=i;
    for (int i=0; i<m; ++i){
        if (getf(u,e[i].u)==getf(u,e[i].v)) continue;
        ++j;
        uni(u,e[i].u,e[i].v);
        ++cnt[idx(e[i].w)];
    }
    if (j<n-1){
        puts("0");
        return 0;
    }
    ele ans=1;
    j=0;
    for (int i=0; i<n; ++i)
            u[i]=i;
    for (int i=0; i<len; ++i){
        ele k=j;
        while (k<m && e[k].w==arr[i])
            ++k;
        ans*=dfs(cnt[i],j,k);
        ans%=MOD;
        for (int p=j; p<k; ++p)
            uni(u,e[p].u,e[p].v);
        j=k;
    }
    printf("%d\n",ans);
    return 0;
}
