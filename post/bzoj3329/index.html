bzoj3329题解

Aug 27, 2016 10:02 · 148 words · 1 minute read 题解 bzoj DP 数位dp 矩阵

考虑到异或的逆运算还是异或，原式可以变形成$x\oplus 2x=3x$。

另一方面，显然有$x+2x=3x$，也就是说，$x$与$2x$相加的时候没有进位。

所以$x$的二进制表示中没有连续的1。

第一问数位dp，第二问数位dp+矩阵快速幂。

开数组的时候不小心开成了18调了好久……要开成$\log_2 n$，而不是$\lg n$……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxlogn 65
#define MOD 1000000007
struct MAT
{
	ele n,m,a[5][5];
};
inline MAT operator*(MAT a,MAT b)
{
	MAT ans;
	ans.n=a.n; ans.m=b.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<b.m; ++j)
		{
			ans.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				ans.a[i][j]+=a.a[i][k]*b.a[k][j],
				ans.a[i][j]%=MOD;
		}
	return ans;
}
inline MAT pow(MAT a,ele x)
{
	MAT ans,tmp=a;
	ans.n=ans.m=a.n;
	memset(ans.a,0,sizeof(ans.a));
	for (int i=0; i<a.n; ++i) ans.a[i][i]=1;
	for (; x; x>>=1,tmp=tmp*tmp)
		if (x&1) ans=ans*tmp;
	return ans;
}
ele n,m,c[maxlogn],f[maxlogn][2],g[maxlogn][2];
int main()
{
	ele T;
	scanf("%lld",&T);
	while (T--)
	{
		scanf("%lld",&n);
		MAT ans=(MAT){2,1,{{1},{1}}};
		MAT C=(MAT){2,2,{{1,1},{1,0}}};
		C=pow(C,n-1);
		ans=C*ans;
		m=0;
		while (n>0) c[m++]=n&1,n>>=1;
		g[0][0]=g[0][1]=1;
		f[0][0]=1;
		if (c[0]) f[0][1]=1; else f[0][1]=0;
		for (int i=1; i<m; ++i)
		{
			g[i][0]=g[i-1][0]+g[i-1][1];
			g[i][1]=g[i-1][0];
			if (!c[i])
			{
				f[i][0]=f[i-1][0]+f[i-1][1];
				f[i][1]=0;
			}
			else
			{
				f[i][0]=g[i-1][0]+g[i-1][1];
				f[i][1]=f[i-1][0];
			}
		}
		printf("%lld\n%lld\n",f[m-1][0]+f[m-1][1]-1,
		(ans.a[0][0]+ans.a[1][0])%MOD);
	}
	return 0;
}
