bzoj2333题解

Nov 14, 2016 10:02 · 230 words · 2 minutes read 题解 bzoj 并查集 优先队列 启发式合并

网上的题解很多都说用可并堆，可我不是很清楚可并堆打标记怎么合并，所以就用stl的priority_queue+启发式合并水过去了……

顺便还要用并查集维护一些东西。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#define ele int
using namespace std;
#define maxn 300010
struct hn{
	ele i,x;
	inline bool operator<(hn b)const{
		if (x!=b.x) return x<b.x;
		return i<b.i;
	}
	inline bool operator==(hn b)const{
		return i==b.i && x==b.x;
	}
};
struct heap{
	inline hn top(){
		while (!a.empty() && !b.empty() && a.top()==b.top())
			a.pop(),b.pop();
		return a.top();
	}
	inline void push(hn x){
		a.push(x);
	}
	inline void del(hn x){
		b.push(x);
	}
	inline ele size(){
		return a.size()-b.size();
	}
	priority_queue<hn> a,b;
};
ele n,q,gadd,w[maxn],u[maxn],add[maxn];
heap gh,h[maxn];
ele getf(ele *u,ele x){
	return (x==u[x])?(x):(u[x]=getf(u,u[x]));
}
inline void erase(ele z){
	hn tmp=h[z].top();
	gh.del({tmp.i,tmp.x+add[z]});
}
inline void push(ele z){
	hn tmp=h[z].top();
	gh.push({tmp.i,tmp.x+add[z]});
}
inline void merge(ele x,ele y){
	ele a=getf(u,x),b=getf(u,y);
	if (a==b) return;
	if (h[a].size()<h[b].size()) swap(a,b);
	erase(a); erase(b);
	while (h[b].size()){
		hn k=h[b].top(); h[b].del(k);
		w[k.i]+=add[b];
		w[k.i]-=add[a];
		k.x=w[k.i];
		h[a].push(k);
	}
	push(a);
	u[b]=a;
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i)
		scanf("%d",w+i);
	memset(add,0,sizeof(add));
	for (int i=0; i<n; ++i){
		u[i]=i;
		h[i].push({i,w[i]});
		push(i);
	}
	gadd=0;
	scanf("%d",&q);
	while (q--){
		char op[10];
		ele x,y;
		scanf("%s",op);
		if (op[0]=='U'){
			scanf("%d%d",&x,&y);
			--x,--y;
			merge(x,y);
		}
		else if (op[0]=='A'){
			if (op[1]=='1'){
				scanf("%d%d",&x,&y);
				--x;
				ele z=getf(u,x);
				erase(z);
				h[z].del({x,w[x]});
				w[x]+=y;
				h[z].push({x,w[x]});
				push(z);
			}
			else if (op[1]=='2'){
				scanf("%d%d",&x,&y);
				--x;
				ele z=getf(u,x);
				erase(z);
				add[z]+=y;
				push(z);
			}
			else if (op[1]=='3'){
				scanf("%d%d",&x);
				gadd+=x;
			}
		}
		else if (op[0]=='F'){
			if (op[1]=='1'){
				scanf("%d",&x);
				--x;
				ele z=getf(u,x);
				printf("%d\n",w[x]+add[z]+gadd);
			}
			else if (op[1]=='2'){
				scanf("%d",&x);
				--x;
				ele z=getf(u,x);
				printf("%d\n",h[z].top().x+add[z]+gadd);
			}
			else if (op[1]=='3')
				printf("%d\n",gh.top().x+gadd);
		}
	}
	return 0;
}
