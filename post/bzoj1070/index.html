bzoj1070题解

Aug 2, 2017 10:02 · 187 words · 1 minute read 题解 bzoj 网络流

单独考虑每辆车的等待时间较为困难，考虑每辆车对总等待时间的贡献，易知若i这辆车是j倒数第k个修的，那么贡献为k*time[i][j]。

用费用流来解决，把每个修理人员拆成n个点，代表倒数第1个，第2个，……第n个（不一定都存在）修的。从s到每辆车连容量为1，费用为0的边，从每辆车到每个修理人员拆出的点连容量为1，费用为k*time[i][j]的边，每个修理人员拆出的点到t连容量为1，费用为0的边。满流时的最小费用即为答案。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
const ele INF=1e9;
#define maxn 610
struct edge{
	ele v,r,c;
	edge *nxt,*rev;
}ep[1000000],*ecnt;
ele n,m,ti[maxn][maxn];
edge *h[maxn],*prv[maxn];
inline void addedge1(ele u,ele v,ele r,ele c){
	edge *p=ecnt++;
	p->v=v; p->r=r; p->c=c; p->nxt=h[u]; p->rev=ep+((ecnt-ep-1)^1);
	h[u]=p;
}
inline void addedge(ele u,ele v,ele r,ele c){
	addedge1(u,v,r,c); addedge1(v,u,0,-c);
}
void dfs(ele i,ele&p,ele&ans){
	if (!prv[i]) return;
	p=min(p,prv[i]->rev->r);
	dfs(prv[i]->v,p,ans);
	prv[i]->r+=p; prv[i]->rev->r-=p;
	ans+=p*prv[i]->rev->c;
}
inline ele mincost(ele s,ele t){
	static ele d[maxn],cnt[maxn];
	static queue<ele> Q;
	static bool vis[maxn];
	ele ans=0;
	while (1){
		memset(vis,0,sizeof(vis));
		memset(d,-1,sizeof(d));
		memset(cnt,0,sizeof(cnt));
		Q.push(s); vis[s]=true; d[s]=0; prv[s]=NULL; cnt[s]=1;
		while (!Q.empty()){
			ele k=Q.front(); Q.pop(); vis[k]=false;
			for (edge *j=h[k]; j; j=j->nxt)
				if (j->r && (!~d[j->v] || d[j->v]>d[k]+j->c)){
					d[j->v]=d[k]+j->c;
					prv[j->v]=j->rev;
					if (!vis[j->v]){
						vis[j->v]=true;
						Q.push(j->v);
					}
				}
		}
		if (!~d[t]) break;
		ele tmp=INF;
		dfs(t,tmp,ans);
	}
	return ans;
}
int main(){
	scanf("%d%d",&m,&n);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j) scanf("%d",&ti[i][j]);
	ecnt=ep;
	memset(h,0,sizeof(h));
	for (int i=0; i<n; ++i) addedge(0,2+i,1,0);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			for (int p=0; p<n; ++p)
				addedge(2+i,2+n+j*n+p,1,(p+1)*ti[i][j]);
	for (int j=0; j<m; ++j)
		for (int p=0; p<n; ++p)
			addedge(2+n+j*n+p,1,1,0);
	printf("%.2lf\n",(double)mincost(0,1)/n);
	return 0;
}
