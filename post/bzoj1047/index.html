bzoj1047题解

Aug 3, 2017 10:02 · 121 words · 1 minute read 题解 bzoj 单调栈/队列

用单调队列维护一些东西就可以做了。

知道这个单调队列方向，先考虑一下一维的情况，还是挺好想的，具体方法就不写了。

要维护一大堆队列，我还以为这么丑陋的代码只有别人写得出来……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 1010
const ele INF=1e9;
ele n,m,k,qh[2][maxn],qt[2][maxn],a[maxn][maxn],Q[2][maxn][maxn];
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j) scanf("%d",&a[i][j]);
	memset(qh,0,sizeof(qh)); memset(qt,0,sizeof(qt));
	ele ans=INF;
	for (int i=0; i<n; ++i){
		qh[0][m]=0; qt[0][m]=0;
		qh[1][m]=0; qt[1][m]=0;
		for (int j=0; j<m; ++j){
			while (qh[0][j]<qt[0][j] && Q[0][j][qh[0][j]]<=i-k) ++qh[0][j];
			while (qh[0][j]<qt[0][j] && a[Q[0][j][qt[0][j]-1]][j]>a[i][j]) --qt[0][j];
			Q[0][j][qt[0][j]++]=i;
			while (qh[1][j]<qt[1][j] && Q[1][j][qh[1][j]]<=i-k) ++qh[1][j];
			while (qh[1][j]<qt[1][j] && a[Q[1][j][qt[1][j]-1]][j]<a[i][j]) --qt[1][j];
			Q[1][j][qt[1][j]++]=i;
			while (qh[0][m]<qt[0][m] && Q[0][m][qh[0][m]]<=j-k) ++qh[0][m];
			while (qh[0][m]<qt[0][m]){
				ele p=Q[0][m][qt[0][m]-1];
				if (a[Q[0][p][qh[0][p]]][p]<=a[Q[0][j][qh[0][j]]][j]) break;
				--qt[0][m];
			}
			Q[0][m][qt[0][m]++]=j;
			while (qh[1][m]<qt[1][m] && Q[1][m][qh[1][m]]<=j-k) ++qh[1][m];
			while (qh[1][m]<qt[1][m]){
				ele p=Q[1][m][qt[1][m]-1];
				if (a[Q[1][p][qh[1][p]]][p]>=a[Q[1][j][qh[1][j]]][j]) break;
				--qt[1][m];
			}
			Q[1][m][qt[1][m]++]=j;
			ele p1=Q[0][m][qh[0][m]],p2=Q[1][m][qh[1][m]];
			if (i>=k-1 && j>=k-1)
				ans=min(ans,a[Q[1][p2][qh[1][p2]]][p2]-a[Q[0][p1][qh[0][p1]]][p1]);
		}
	}
	printf("%d\n",ans);
	return 0;
}
