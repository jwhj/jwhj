bzoj1027题解

May 27, 2017 10:02 · 157 words · 1 minute read 题解 bzoj 计算几何

首先注意到a+b+c=1，那么只需要管a和b就可以了。

把每种材料看成一个点(a,b)，那么两个材料(a1,b1)，(a2,b2)所合成的东西对应的点一定在两点的连线上。于是拿一堆材料进行合成，结果一定在它们构成的凸包内（包括边界），而且凸包内任何一个点都是可以达到的。

于是题目转化为，从点集A中选出最少的点，使点集B中所有点落在它们的凸包内。

顺时针绕凸包一圈，考虑每条边对应的向量，会发现凸包内所有点在其右侧。

对于A中任意两点i，j（i，j可相同），若B中所有点在向量i->j右侧，则连一条i到j的边长为1的边，然后跑最小环就能得到答案。

i和j可以相同是因为有可能A中只有一个点。

需要注意一个小细节，如果某个点与i，j共线，若它应该在i，j之间，连i到j的边，否则就不应该连。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#define ele int
#define ll long long
using namespace std;
#define maxn 510
#define eps 1e-7
const ele INF=1e9+10;
struct pt{
	double x,y;
};
#define vc pt
inline vc operator-(pt a,pt b){
	return (vc){a.x-b.x,a.y-b.y};
}
inline double dot(vc a,vc b){
	return a.x*b.x+a.y*b.y;
}
inline double cross(vc a,vc b){
	return a.x*b.y-a.y*b.x;
}
ele m,n,g[maxn][maxn];
pt a[maxn],b[maxn];
int main(){
	scanf("%d%d",&m,&n);
	double t;
	for (int i=0; i<m; ++i) scanf("%lf%lf%lf",&a[i].x,&a[i].y,&t);
	for (int i=0; i<n; ++i) scanf("%lf%lf%lf",&b[i].x,&b[i].y,&t);
	for (int i=0; i<m; ++i)
		for (int j=0; j<m; ++j) g[i][j]=INF;
	for (int i=0; i<m; ++i)
		for (int j=0; j<m; ++j){
			bool flag=true;
			for (int k=0; k<n; ++k){
				t=cross(a[i]-b[k],a[j]-b[k]);
				if (t>eps || (fabs(t)<eps && dot(a[i]-b[k],a[j]-b[k])>eps)) flag=false;
			}
			if (flag) g[i][j]=1;
		}
	ele ans=INF;
	for (int k=0; k<m; ++k)
		for (int i=0; i<m; ++i)
			for (int j=0; j<m; ++j) g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
	for (int i=0; i<m; ++i) ans=min(ans,g[i][i]);
	printf("%d\n",(ans<INF)?ans:-1);
	return 0;
}
