bzoj4552题解

Feb 11, 2017 10:02 · 253 words · 2 minutes read 题解 bzoj 统计 线段树 动态开点 值域线段树分裂合并

之前听同学讲过一个神奇的做法。

首先，值域线段树在动态开点的情况下是可以分裂合并的。

合并两棵树a，b，如果a，b中有一个为空，则返回另一个，否则递归合并左右子树，接到a上面，然后把b删掉。

分裂的话，随便想一想就知道了吧……

这样，序列一开始可以看成n个升序排列的块，每块1个数。每次排序可以看成分裂不超过2个块，并合并若干个块。用map维护每个块的位置，询问的两个端点所在的块的位置可以$O(\log n)$求出来。

时间复杂度的分析如下：

每次合并两棵线段树，总代价为删去的点数和。初始时点数为$O(n\log n)$，每次分裂至多增加$O(\log n)$个点。

据说复杂度比正解更优，跑到了第一版。

代码：

#include <cstdio>
#include <cstring>
#include <map>
#include <queue>
#define ele int
#define iter map<ele,node*>::iterator
using namespace std;
#define maxn 100010
struct node{
	ele sum;
	node *l,*r;
}np[10000000];
ele n,m,q,ncnt;
bool dc[maxn];
queue<node*> Q;
map<ele,node*> blk;
inline node* func(){
	node *p=Q.front();
	Q.pop();
	return p;
}
inline node* newnode(){
	return (Q.empty())?(&np[ncnt++]):(func());
}
inline void maintain(node *x){
	if (x){
		x->sum=0;
		if (x->l) x->sum+=x->l->sum;
		if (x->r) x->sum+=x->r->sum;
	}
}
inline node* build(ele l,ele r,ele k){
	node *p=newnode();
	p->sum=1;
	if (l==r)
		p->l=p->r=NULL;
	else{
		ele mid=(l+r)>>1;
		p->l=(k<=mid)?(build(l,mid,k)):(NULL);
		p->r=(mid<k)?(build(mid+1,r,k)):(NULL);
	}
	return p;
}
node* merge(node *a,node *b){
	if (!a) return b;
	if (!b) return a;
	a->l=merge(a->l,b->l);
	a->r=merge(a->r,b->r);
	maintain(a);
	Q.push(b);
	return a;
}
void split(node *x,ele k,node*&a,node*&b){
	if (!x){ a=b=NULL; return;}
	ele s=(x->l)?(x->l->sum):(0);
	node *c=newnode();
	if (k<s){
		split(x->l,k,a,b);
		c->l=b; c->r=x->r;
		maintain(c);
		x->l=a; x->r=NULL;
		maintain(x);
		a=x; b=c;
	}
	else if (k>s){
		split(x->r,k-s,a,b);
		c->l=NULL; c->r=b;
		maintain(c);
		x->r=a;
		maintain(x);
		a=x; b=c;
	}
	else{
		a=x; b=c;
		b->l=NULL;
		b->r=a->r; a->r=NULL;
		maintain(a); maintain(b);
	}
}
ele qry(node *x,ele l,ele r){
	if (l==r) return l;
	ele mid=(l+r)>>1;
	if (x->l && x->l->sum) return qry(x->l,l,mid);
	return qry(x->r,mid+1,r);
}
inline void split(ele i){
	if (i==-1 || blk[i]) return;
	blk[i]=NULL;
	iter it=blk.find(i),t=it;
	ele a=(--t)->first;
	t=it;
	ele b=(++t)->first;
	dc[i]=dc[b];
	if (dc[b]) split(blk[b],b-i,blk[b],blk[i]);
	else split(blk[b],i-a,blk[i],blk[b]);
}
int main(){
	scanf("%d%d",&n,&m);
	memset(dc,0,sizeof(dc));
	ncnt=0;
	blk[-1]=NULL;
	for (int i=0; i<n; ++i){
		ele x;
		scanf("%d",&x);
		blk[i]=build(1,n,x);
	}
	while (m--){
		ele op,u,v;
		scanf("%d%d%d",&op,&u,&v);
		--u,--v;
		split(u-1);
		split(v);
		dc[v]=op;
		iter it=blk.find(u-1),e=blk.find(v);
		++it;
		for (iter t=it; it!=e; it=t){
			++t;
			t->second=merge(it->second,t->second);
			blk.erase(it);
		}
	}
	scanf("%d",&q); --q;
	split(q);
	split(q-1);
	printf("%d\n",qry(blk[q],1,n));
	return 0;
}
