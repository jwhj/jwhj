CF725D题解

Oct 23, 2016 10:02 · 95 words · 1 minute read 题解 codeforces 贪心 优先队列

题目背景真是233

如果浮起来的队伍数量已经确定了，那么显然要选取排在前面的w-t最小的队伍，让他们浮起来。

从小到大枚举浮起来的队伍数，用一个优先队列维护排在前面的队伍的w-t值。每次选取最小的，让他们浮起来，然后看看会不会被后面的队伍超越，如果被超越就把那个队也加到优先队列里面。

代码：

#include <cstdio>
#include <queue>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 300010
struct team{
	ele t,w;
	inline bool operator<(team b)const{
		return w-t>b.w-b.t;
	}
};
inline bool cmp(team a,team b){
	return a.t>b.t;
}
ele n;
team a[maxn];
priority_queue<team> q;
int main(){
	scanf("%lld",&n);
	for (int i=0; i<n; ++i)
		scanf("%lld%lld",&a[i].t,&a[i].w);
	sort(a+1,a+n,cmp);
	ele i=0;
	while (i<n-1 && a[i].t<a[i+1].t){
		q.push(a[i+1]);
		swap(a[i],a[i+1]);
		++i;
	}
	ele ans=i+1;
	ele tmp=ans;
	while (!q.empty()){
		team k=q.top(); q.pop();
		if (k.w-k.t+1>a[i].t) break;
		--tmp;
		a[i].t-=k.w-k.t+1;
		while (i<n-1 && a[i].t<a[i+1].t){
			q.push(a[i+1]);
			swap(a[i],a[i+1]);
			++i;
			++tmp;
		}
		ans=min(ans,tmp);
	}
	printf("%lld\n",ans);
	return 0;
}
