bzoj3160题解

May 1, 2016 10:02 · 196 words · 1 minute read 题解 bzoj FFT 字符串 manacher

首先，答案=所有的-连续的。连续的可以用manacher解出来，只需考虑如何求所有的自序列个数。

对于每个对称中心，只需求出有多少对对称的字符即可，而对称中心确定以后，每对对称的字符，位置加起来的值是固定的，可以写个类似卷积的东西，然后用FFT来求。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <complex>
#define ele long long
using namespace std;
#define maxn 400010
#define MOD 1000000007
#define pn complex<double>
const double pi=3.1415926535897;
ele n,bin[maxn];
char s[maxn];
pn a1[maxn],a2[maxn],b1[maxn],b2[maxn];
inline ele manacher(ele n,char *s){
	static char s1[maxn];
	s1[0]='!';
	for (int i=0; i<n; ++i) s1[i<<1|1]='#',s1[i*2+2]=s[i];
	s1[n<<1|1]='#';
	s1[n*2+2]='@';
	n=n*2+3;
	static ele p[maxn];
	p[0]=0;
	ele mx=0,id=0;
	for (int i=1; i<n; ++i){
		if (i<=mx)
			p[i]=min(mx-i,p[id*2-i]);
		else
			p[i]=0;
		while (s1[i-p[i]-1]==s1[i+p[i]+1]) ++p[i];
		if (i+p[i]>mx){
			mx=i+p[i];
			id=i;
		}
	}
	ele ans=0;
	for (int i=0; i<n; ++i)
		ans+=(p[i]+1)>>1,ans%=MOD;
	return ans;
}
inline void FFT(ele K,ele n,pn *a,pn *b){
	static ele rev[maxn];
	rev[0]=0;
	for (int i=1; i<n; ++i)
		rev[i]=(rev[i>>1]>>1)+(i&1)*(n>>1);
	for (int i=0; i<n; ++i)
		b[rev[i]]=a[i];
	for (int i=1; i<n; i<<=1){
		pn o1=(pn){cos(pi/i*K),sin(pi/i*K)};
		for (int j=0; j<n; j+=(i<<1)){
			pn o=1;
			for (int k=j; k<j+i; ++k){
				pn u=b[k],t=o*b[k+i];
				b[k]=u+t;
				b[k+i]=u-t;
				o*=o1;
			}
		}
	}
	if (K==-1)
		for (int i=0; i<n; ++i)
			b[i]/=n;
}
int main(){
	scanf("%s",s);
	n=strlen(s);
	ele ans=-manacher(n,s);
	for (int i=0; i<n; ++i)
		a1[i]=s[i]-'a',b1[i]=1-s[i]+'a';
	ele tmp=1;
	while (tmp<n) tmp<<=1;
	tmp<<=1;
	FFT(1,tmp,a1,a2); FFT(1,tmp,b1,b2);
	for (int i=0; i<tmp; ++i)
		b1[i]=a2[i]*a2[i]+b2[i]*b2[i];
	FFT(-1,tmp,b1,a1);
	bin[0]=1;
	for (int i=1; i<n; ++i)
		bin[i]=bin[i-1]*2%MOD;
	for (int i=0; i<tmp; ++i)
		ans+=bin[((ele)(a1[i].real()+0.5)+1)>>1]-1,
		ans%=MOD;
	printf("%lld\n",ans);
	return 0;
}
