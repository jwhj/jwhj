bzoj1202题解

Jul 28, 2017 10:02 · 124 words · 1 minute read 题解 bzoj 并查集 乱搞

好久没碰过键盘了，做道水题练练手。

设b[i]为前i个月的总收入，那么每条信息(s,t,v)就相当于b[t]-b[s-1]=v。

于是题目转化为：给你n+1个变量，给定若干个他们之前差的关系，问是否有解。

用一个并查集来维护变量之间是否有关系。建一个图，每次得到一条新信息的时候，如果涉及的两个变量无关，则连一条s-1到t，权为v的边，和一条t-1到s，权为-v的边。如果涉及的变量有关，那么就要判断能不能成立。在图中找到s-1到t的路径，如果权值和恰为v，则成立。注意此时无论成不成立，均不需连边。

这个做法的思路跟差分约束比较类似，其实此题也可以直接用差分约束来做。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 110
#define maxm 1010
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxm<<1],*ecnt;
ele n,m,u[maxn],d[maxn];
edge *h[maxn];
bool vis[maxn];
inline void addedge(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w;
	p->nxt=h[u];
	h[u]=p;
}
ele getf(ele *u,ele x){
	return (x==u[x])?x:(u[x]=getf(u,u[x]));
}
void dfs(ele x){
	vis[x]=true;
	for (edge *j=h[x]; j; j=j->nxt)
		if (!vis[j->v]){
			d[j->v]=d[x]+j->w;
			dfs(j->v);
		}
}
int main(){
	ele w;
	scanf("%d",&w);
	while (w--){
		scanf("%d%d",&n,&m);
		ecnt=ep;
		memset(h,0,sizeof(h));
		for (int i=0; i<=n; ++i) u[i]=i;
		bool ans=true;
		while (m--){
			ele s,t,v;
			scanf("%d%d%d",&s,&t,&v);
			ele a=getf(u,s-1),b=getf(u,t);
			if (a!=b){
				u[a]=b;
				addedge(s-1,t,v); addedge(t,s-1,-v);
			}
			else{
				memset(vis,0,sizeof(vis));
				memset(d,-1,sizeof(d));
				d[s-1]=0;
				dfs(s-1);
				if (d[t]!=v) ans=false;
			}
		}
		if (ans) puts("true"); else puts("false");
	}
	return 0;
}
