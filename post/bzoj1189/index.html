bzoj1189题解

Jul 29, 2017 10:02 · 302 words · 2 minutes read 题解 bzoj 网络流 二分

利用bfs得出格子之间两两的最短路。

首先二分答案T，把每扇门拆成T个点，把每个有人的格子看成一个点，随便建一下图跑网络流，根据是否满流判断当前答案是否可行。

bfs都能写错……真是太久没碰oi了……

代码：

#include <cstdio>
#include <cstring>
#include <vector>
#define ele int
using namespace std;
#define maxn 25
#define maxnode 320010
const ele INF=1e9;
const ele di[]={-1,1,0,0};
const ele dj[]={0,0,-1,1};
struct edge{
	ele v,r;
	edge *nxt,*rev;
}ep[2500000],*ecnt;
ele n,m,dep[maxnode],d[maxn*maxn][maxn*maxn];
char s[maxn][maxn];
edge *h[maxnode],*cur[maxnode];
vector<ele> plst,dlst;
inline void addedge1(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->r=w; p->nxt=h[u]; p->rev=ep+((ecnt-1-ep)^1);
	h[u]=p;
}
inline void addedge(ele u,ele v,ele w){
	addedge1(u,v,w); addedge1(v,u,0);
}
inline bool bfs(ele s,ele t){
	memset(dep,-1,sizeof(dep));
	ele qh=0,qt=1;
	static ele Q[maxnode];
	Q[0]=s; dep[s]=0;
	while (qh<qt){
		for (edge *j=h[Q[qh]]; j; j=j->nxt)
			if (j->r && !~dep[j->v]){
				dep[j->v]=dep[Q[qh]]+1;
				Q[qt++]=j->v;
				if (j->v==t) return true;
			}
		++qh;
	}
	return false;
}
ele dfs(ele i,ele t,ele p){
	if (i==t || !p) return p;
	ele ans=0;
	for (edge*&j=cur[i]; j; j=j->nxt)
		if (j->r && dep[i]+1==dep[j->v]){
			ele tmp=dfs(j->v,t,min(p,j->r));
			if (tmp){
				j->r-=tmp;
				j->rev->r+=tmp;
				ans+=tmp;
				p-=tmp;
				if (!p) break;
			}
		}
	if (!p) dep[i]=-1;
	return ans;
}
inline ele dinic(ele s,ele t){
	ele ans=0;
	while (bfs(s,t)){
		memcpy(cur,h,sizeof(h));
		ans+=dfs(s,t,INF);
	}
	return ans;
}
inline void bfs1(ele p){
	static ele Q[maxn*maxn];
	static bool vis[maxn][maxn];
	memset(vis,0,sizeof(vis));
	ele qh=0,qt=1;
	Q[0]=p; d[p][p]=0;
	while (qh<qt){
		ele i=Q[qh]/m,j=Q[qh]%m;
		vis[i][j]=true;
		for (int w=0; w<4; ++w){
			ele ni=i+di[w],nj=j+dj[w];
			if (0<=ni && ni<n && 0<=nj && nj<m && !vis[ni][nj]){
				vis[ni][nj]=true;
				d[p][ni*m+nj]=d[p][Q[qh]]+1;
				if (s[ni][nj]=='.') Q[qt++]=ni*m+nj;
			}
		}
		++qh;
	}
}
inline bool test(ele T){
	ecnt=ep;
	memset(h,0,sizeof(h));
	for (int i=0; i<plst.size(); ++i){
		addedge(0,2+i,1);
		for (int j=0; j<dlst.size(); ++j){
			if (!~d[plst[i]][dlst[j]]) continue;
			for (int t=d[plst[i]][dlst[j]]; t<=T; ++t)
				addedge(2+i,2+plst.size()+(t-1)*dlst.size()+j,1);
		}
	}
	for (int t=0; t<T; ++t)
		for (int j=0; j<dlst.size(); ++j)
			addedge(2+plst.size()+t*dlst.size()+j,1,1);
	return dinic(0,1)>=plst.size();
}
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i) scanf("%s",s[i]);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			if (s[i][j]=='.') plst.push_back(i*m+j);
			else if (s[i][j]=='D') dlst.push_back(i*m+j);
	memset(d,-1,sizeof(d));
	for (int i=0; i<plst.size(); ++i) bfs1(plst[i]);
	ele L=0,R=n*m+1;
	while (R-L>1){
		ele mid=(L+R)>>1;
		if (test(mid)) R=mid; else L=mid;
	}
	if (R<n*m+1) printf("%d\n",R); else puts("impossible");
	return 0;
}
