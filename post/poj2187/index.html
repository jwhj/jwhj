poj2187题解

Aug 14, 2017 10:02 · 159 words · 1 minute read 题解 poj 计算几何 旋转卡壳

还不会旋转卡壳的可以看这里。

WA了一段时间，要注意的主要有以下几个：

  • 对点排序的时候，xy坐标都要比较，因为可能会有重合的点，要确保它们排完序之后在一起。
  • 求凸包的时候有一些细节，比方说要处理好点共线的情况

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 50010
#define vec pt
struct pt{
	ele x,y;
	inline bool operator<(pt b)const{
		if (x!=b.x) return x<b.x;
		return y<b.y;
	}
};
ele n,top;
pt p[maxn],a[maxn<<1];
inline ele _abs(ele x){ return (x<0)?-x:x; }
inline vec operator+(pt a,vec b){
	return {a.x+b.x,a.y+b.y};
}
inline vec operator-(pt a,vec b){
	return {a.x-b.x,a.y-b.y};
}
inline ele dot(vec a,vec b){
	return a.x*b.x+a.y*b.y;
}
inline ele cross(vec a,vec b){
	return a.x*b.y-a.y*b.x;
}
inline ele area(pt a,pt b,pt c){
	return _abs(cross(b-a,c-a));
}
inline ele dissqr(pt a,pt b){
	return (a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y);
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%d%d",&p[i].x,&p[i].y);
	sort(p,p+n);
	top=0;
	for (int i=0; i<n; ++i){
		while (top>1 && cross(a[top-1]-a[top-2],p[i]-a[top-1])<=0) --top;
		a[top++]=p[i];
	}
	ele tmp=top;
	for (int i=n-1; ~i; --i){
		while (top>tmp+1 && cross(a[top-1]-a[top-2],p[i]-a[top-1])<=0) --top;
		a[top++]=p[i];
	}
	for (int i=tmp; i<top; ++i) a[i]=a[i+1];
	--top;
	ele ans=0,j=2;
	for (int i=0; i<top-1; ++i){
		while (area(a[i],a[i+1],a[j])<area(a[i],a[i+1],a[j+1])) (j+=1)%=(top-1);
		ans=max(ans,dissqr(a[i],a[j]));
	}
	printf("%d\n",ans);
	return 0;
}
