bzoj1021题解

May 19, 2017 10:02 · 135 words · 1 minute read 题解 bzoj DP

容易发现，不可能会有A把一张钞票给B，B再将其给C的情况，因为这样的话直接由A交给C就可以了。

于是，对于每一种面额的钞票，总共有6种情况：A给BC，BC给A，B给AC，AC给B，C给AB，AB给C。

让f[i][j][k]代表只交换前i种钞票，使得A有j元，B有k元的最少交换次数，那么可以通过枚举上面的6中情况来进行转移。

由于很多状态实际上是达不到的，所以可以用类似bfs的方式，用一个队列保存状态并拓展。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
#define maxn 1010
const ele p[7]={0,1,5,10,20,50,100};
struct st{
	ele i,a[3];
};
ele x1,x2,x3,a[3],cnt[3][7],f[7][maxn][maxn];
queue<st> Q;
inline ele solve(){
	Q.push((st){0,{a[0],a[1],a[2]}});
	memset(f,-1,sizeof(f));
	f[0][a[0]][a[1]]=0;
	while (!Q.empty()){
		st s=Q.front(); Q.pop();
		if (s.i==6) continue;
		for (int i=0; i<3; ++i){
			ele j=(i+1)%3,k=(i+2)%3;
			for (int u=0; u<=cnt[i][s.i+1]; ++u)
				for (int v=0; v<=u; ++v){
					st tmp=s;
					++tmp.i;
					tmp.a[i]-=p[tmp.i]*u;
					tmp.a[j]+=p[tmp.i]*v;
					tmp.a[k]+=p[tmp.i]*(u-v);
					if (!~f[tmp.i][tmp.a[0]][tmp.a[1]] ||
					f[s.i][s.a[0]][s.a[1]]+u<f[tmp.i][tmp.a[0]][tmp.a[1]]){
						f[tmp.i][tmp.a[0]][tmp.a[1]]=f[s.i][s.a[0]][s.a[1]]+u;
						Q.push(tmp);
					}
				}
			for (int u=0; u<=cnt[j][s.i+1]; ++u)
				for (int v=0; v<=cnt[k][s.i+1]; ++v){
					st tmp=s;
					++tmp.i;
					tmp.a[i]+=p[tmp.i]*(u+v);
					tmp.a[j]-=p[tmp.i]*u;
					tmp.a[k]-=p[tmp.i]*v;
					if (!~f[tmp.i][tmp.a[0]][tmp.a[1]] ||
					f[s.i][s.a[0]][s.a[1]]+u+v<f[tmp.i][tmp.a[0]][tmp.a[1]]){
						f[tmp.i][tmp.a[0]][tmp.a[1]]=f[s.i][s.a[0]][s.a[1]]+u+v;
						Q.push(tmp);
					}
				}
		}
	}
	return f[6][a[0]-x1+x3][a[1]+x1-x2];
}
int main(){
	scanf("%d%d%d",&x1,&x2,&x3);
	for (int i=0; i<3; ++i){
		for (int j=6; j; --j){
			scanf("%d",&cnt[i][j]);
			a[i]+=cnt[i][j]*p[j];
		}
	}
	ele t=solve();
	if (~t) printf("%d\n",t);
	else puts("impossible");
	return 0;
}
