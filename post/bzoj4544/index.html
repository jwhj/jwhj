bzoj4544题解

Nov 2, 2016 10:02 · 112 words · 1 minute read 题解 bzoj 数学

先设$(x,y)=1$，那么$x$，$y$，$z$两两互质。

由题，有$3y^2=(n-x)(n+x)$。

如果x，n均为奇数，那么$(n-x)$与$(n+x)$的最大公约数为2，于是两者可以表示为$6a^2$和$2b^2$。

所以$x=|3a^2-b^2|,y=2ab,n=3a^2+b^2$，枚举b即可。

x与n一奇一偶时也可用类似的方法得出结果。

x，y不互质的时候，枚举x与y的最大公约数然后转换为互质的情形来做。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#define ele long long
using namespace std;
ele n;
inline ele abs(ele x){
	return (x<0)?(-x):(x);
}
ele gcd(ele a,ele b){
	if (!b) return a;
	return gcd(b,a%b);
}
ele calc(ele n){
	ele ans=0;
	for (ele i=1; i*i*3<n; ++i){
		ele j=n-i*i*3,k=(ele)(sqrt(j)+0.5);
		if (k*k==j){
			ele x=abs(i*i*3-k*k),y=i*k*2;
			if (gcd(x,y)==1 && (x&1)) ++ans;
		}
	}
	n<<=1;
	for (ele i=1; i*i*3<n; ++i){
		ele j=n-i*i*3,k=(ele)(sqrt(j)+0.5);
		if (k*k==j){
			ele x=abs(i*i*3-k*k)>>1,y=i*k;
			if (gcd(x,y)==1 && (y&1)) ++ans;
		}
	}
	return ans;
}
int main(){
	ele T;
	scanf("%lld",&T);
	while (T--){
		scanf("%lld",&n);
		ele ans=0;
		for (ele i=1; i*i<=n; ++i)
			if (n%i==0){
				ans+=calc(i);
				if (i*i!=n) ans+=calc(n/i);
			}
		printf("%lld\n",(ans<<2)+2);
	}
	return 0;
}
