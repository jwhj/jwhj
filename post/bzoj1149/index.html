bzoj1149题解

Jul 12, 2017 10:02 · 138 words · 1 minute read 题解 bzoj dp

一开始没看到题目里面的“或至多相差一层”，那就很尴尬了……

首先，如果最深深度和最浅深度相差2或以上，那肯定是不能通过交换满足条件的。

满足条件之后，任何一条线上吊着的”子风铃“也都是满足条件的。让f[i]代表以i为根的子树的最小交换次数，如果其左右子树均包含较深深度和较浅深度，那么也肯定是不行的。否则，设i左右子树为j，k，则f[i]=f[j]+f[k]+(是否要交换)。处理出左右子树包含的深度，是否要交换是很好判断的。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
const ele INF=1e9;
ele n,a[maxn],b[maxn],dmn,dmx,st[maxn];
bool f[maxn][2];
void dfs1(ele i,ele d){
	if (!~i){
		dmn=min(dmn,d);
		dmx=max(dmx,d);
		return;
	}
	dfs1(a[i],d+1); dfs1(b[i],d+1);
}
ele dfs2(ele i,ele d){
	if (!~a[i] && !~b[i]){
		st[i]=(d+1==dmn)?1:2;
		return 0;
	}
	ele ls,rs,ans;
	if (!~a[i]){
		ls=(d+1==dmn)?1:2;
		ans=dfs2(b[i],d+1);
		rs=st[b[i]];
	}
	else if (!~b[i]){
		ans=dfs2(a[i],d+1);
		ls=st[a[i]];
		rs=(d+1==dmn)?1:2;
	}
	else{
		ele t1=dfs2(a[i],d+1),t2=dfs2(b[i],d+1);
		if (!~t1 || !~t2) return -1;
		ans=t1+t2;
		ls=st[a[i]];
		rs=st[b[i]];
	}
	if (!~ans) return -1;
	if (ls==3 && rs==3) return -1;
	st[i]=ls | rs;
	if (ls==1) return ans+(rs>>1);
	else if (ls==2) return ans;
	else if (rs==1) return ans;
	else return ans+(ls&1);
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i){
		scanf("%d%d",a+i,b+i);
		if (~a[i]) --a[i];
		if (~b[i]) --b[i];
	}
	dmn=INF; dmx=-1;
	dfs1(0,0);
	if (dmx-dmn>1){
		puts("-1");
	}
	else printf("%d\n",dfs2(0,0));
	return 0;
}
