代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
#define maxk 18
struct ANS{
	ele x,l,r;
};
struct node{
	ANS a;
	ele tag;
	node *l,*r;
}np[maxn<<2];
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<1];
ele n,q,ecnt,ncnt,scnt,c[maxn],r[maxn],dep[maxn],size[maxn],seq[maxn],pos[maxn],fa[maxn][maxk];
edge *h[maxn],*hc[maxn];
node *root;
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
inline ANS operator+(ANS a,ANS b){
	if (b.x==-1) return a;
	ANS c=(ANS){a.x+b.x,a.l,b.r};
	if (a.r==b.l)
		--c.x;
	return c;
}
inline void pushdown(node *x){
	if (x->tag!=-1){
		x->a.x=1;
		x->a.l=x->a.r=x->tag;
		if (x->l)
			x->l->tag=x->r->tag=x->tag;
		x->tag=-1;
	}
}
node* build(ele l,ele r){
	node *p=&np[ncnt++];
	p->tag=-1;
	if (l==r){
		p->a=(ANS){1,c[seq[l-1]],c[seq[l-1]]};
		p->l=p->r=NULL;
	}
	else{
		ele mid=(l+r)>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		p->a=p->l->a+p->r->a;
	}
	return p;
}
void upd(node *x,ele a,ele b,ele l,ele r,ele c){
	pushdown(x);
	if (l<=a && b<=r)
		x->tag=c,pushdown(x);
	else{
		ele mid=(a+b)>>1;
		if (l<=mid)
			upd(x->l,a,mid,l,r,c);
		if (mid<r)
			upd(x->r,mid+1,b,l,r,c);
		pushdown(x->l); pushdown(x->r);
		x->a=x->l->a+x->r->a;
	}
}
ANS query(node *x,ele a,ele b,ele l,ele r){
	pushdown(x);
	if (l<=a && b<=r)
		return x->a;
	else{
		ele mid=(a+b)>>1;
		if (r<=mid)
			return query(x->l,a,mid,l,r);
		else if (l>mid)
			return query(x->r,mid+1,b,l,r);
		else return query(x->l,a,mid,l,r)+query(x->r,mid+1,b,l,r);
	}
}
void dfs1(ele p,ele i){
	fa[i][0]=p;
	size[i]=1;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p){
			dep[j->v]=dep[i]+1;
			dfs1(i,j->v);
			size[i]+=size[j->v];
		}
	hc[i]=NULL;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p)
			if (hc[i]==NULL || size[j->v]>size[hc[i]->v])
				hc[i]=j;
}
void dfs2(ele p,ele i){
	pos[i]=scnt;
	seq[scnt++]=i;
	if (hc[i])
		r[hc[i]->v]=r[i],dfs2(i,hc[i]->v);
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && j!=hc[i])
			r[j->v]=j->v,dfs2(i,j->v);
}
inline ele lca(ele u,ele v){
	if (dep[u]>dep[v])
		swap(u,v);
	ele k=dep[v]-dep[u];
	for (int i=0; k; ++i,k>>=1)
		if (k&1)
			v=fa[v][i];
	if (u==v) return u;
	for (int i=maxk-1; i>=0; --i)
		if (fa[u][i]!=fa[v][i])
			u=fa[u][i],v=fa[v][i];
	return fa[u][0];
}
inline void upd(ele x,ele d,ele p,ele c){
	while (x!=-1 && dep[x]>=d){
		ele y=r[x];
		upd(root,1,n,max(p,pos[y])+1,pos[x]+1,c);
		x=fa[y][0];
	}
}
inline ANS query(ele x,ele d,ele p){
	ANS ans=(ANS){-1,0,0};
	while (x!=-1 && dep[x]>=d){
		ele y=r[x];
		ans=query(root,1,n,max(p,pos[y])+1,pos[x]+1)+ans;
		x=fa[y][0];
	}
	return ans;
}
int main(){
	scanf("%d%d",&n,&q);
	for (int i=0; i<n; ++i)
		scanf("%d",c+i);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v);
		--u,--v;
		addedge(u,v); addedge(v,u);
	}
	dep[0]=0;
	dfs1(-1,0);
	for (int i=1; i<maxk; ++i)
		for (int j=0; j<maxn; ++j)
			if (fa[j][i-1]==-1)
				fa[j][i]=-1;
			else fa[j][i]=fa[fa[j][i-1]][i-1];
	scnt=0;
	r[0]=0;
	dfs2(-1,0);
	ncnt=0;
	root=build(1,n);
	while (q--){
		char s[5];
		ele u,v,w,c;
		scanf("%s%d%d",s,&u,&v);
		--u,--v;
		w=lca(u,v);
		if (s[0]=='C'){
			scanf("%d",&c);
			upd(u,dep[w],pos[w],c);
			upd(v,dep[w],pos[w],c);
		}
		else{
			ANS a=query(u,dep[w],pos[w]);
			ANS b=query(v,dep[w],pos[w]);
			swap(a.l,a.r);
			a=a+b;
			printf("%d\n",a.x);
		}
	}
	return 0;
}
