bzoj2844题解

Jun 7, 2017 10:02 · 118 words · 1 minute read 题解 bzoj 线性基

如果重复出现的数只算一个，用线性基就可以很容易地解决。可是现在重复出现的数需要重复计算。

观察样例，发现每个数的出现次数是一样的。下面证明这一点。

假设集合的大小为n，线性基的大小为m。容易发现，对于不在线性基里的那n-m个数的任意一种取法，都有唯一的方案，异或上线性基中的若干个数，得到某个特定的数x。注意到从n-m个数中取若干个数共$2^{n-m}$种方案，所以可以得到每个数的出现子树不仅是一样的，而且刚好是$2^{n-m}$。

那么再加上一个快速幂就可以解决了。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 100010
#define W 32
#define MOD 10086
ele n,a[maxn],b[W],c[W],d[W];
inline ele pw(ele a,ele x){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp%MOD)
		if (x&1) ans=ans*tmp%MOD;
	return ans;
}
int main(){
	scanf("%d",&n);
	memset(b,0,sizeof(b));
	for (int i=0; i<n; ++i){
		ele x;
		scanf("%d",&x);
		for (int j=W-1; ~j; --j)
			if (x>>j&1){
				if (b[j]) x^=b[j];
				else{
					b[j]=x;
					for (int k=j-1; ~k; --k)
						if (b[j]>>k&1) b[j]^=b[k];
					for (int k=j+1; k<W; ++k)
						if (b[k]>>j&1) b[k]^=b[j];
					break;
				}
			}
	}
	ele m=0;
	for (int i=0; i<W; ++i)
		if (b[i]) d[m]=i,c[m++]=b[i];
	ele Q;
	scanf("%d",&Q);
	ele ans=0;
	for (int i=m-1; ~i; --i)
		if (Q>>d[i]&1){
			Q^=c[i];
			ans+=(1LL<<i)%MOD;
			ans%=MOD;
		}
	(ans*=pw(2,n-m))%=MOD;
	printf("%d\n",(ans+1)%MOD);
	return 0;
}
