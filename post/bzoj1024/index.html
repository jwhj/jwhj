bzoj1024题解

Dec 25, 2016 10:02 · 56 words · 1 minute read 题解 bzoj

最近刷水严重啊……

枚举切下去的位置，因为切出来大小一样，所以左右两块的大小必须是$\frac{xy}{n}$的整数倍。由于n的范围很小，直接暴搜就可以了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define real double
using namespace std;
ele n;
real x,y,p;
real dfs(real x,real y,ele n){
	if (n==1) return max(x,y)/min(x,y);
	real ans=1e9;
	for (int i=1; i<n; ++i)
		ans=min(ans,max(dfs(x,y/n*i,i),dfs(x,y/n*(n-i),n-i)));
	for (int i=1; i<n; ++i)
		ans=min(ans,max(dfs(x/n*i,y,i),dfs(x/n*(n-i),y,n-i)));
	return ans;
}
int main(){
	scanf("%lf%lf%d",&x,&y,&n);
	p=x*y/n;
	printf("%.6lf\n",dfs(x,y,n));
	return 0;
}
