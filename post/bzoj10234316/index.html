bzoj1023&4316题解

May 24, 2017 10:02 · 397 words · 2 minutes read 题解 bzoj 仙人掌 圆方树 DP 单调栈/队列

不会圆方树的话可以先看一下这篇题解。

这两题都是圆方树上的dp。

首先考虑在树上怎么dp。在圆方树的圆点上，情况是一样的，对于方点，再特殊考虑一下即可。

bzoj1023在dp的时候还要用单调队列优化一下。

代码（bzoj1023）：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#define ele int
using namespace std;
#define maxn 666666
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxn<<2],*ecnt;
ele n,m,ans,sqcnt,tcnt,top,dfn[maxn],low[maxn],stk[maxn],d[maxn],D[maxn],f[maxn];
edge *h[maxn],*h1[maxn];
vector<ele> vec[maxn];
inline void addedge1(edge **h,ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
inline void addedge(edge **h,ele u,ele v,ele w){
	addedge1(h,u,v,w); addedge1(h,v,u,w);
}
void tarjan(ele p,ele i){
	low[i]=dfn[i]=tcnt++;
	stk[top++]=i;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!~dfn[j->v]){
			d[j->v]=d[i]+1;
			tarjan(i,j->v);
			low[i]=min(low[i],low[j->v]);
			if (dfn[i]==low[j->v]){
				D[sqcnt]=d[stk[top-1]]-d[i]+1;
				vec[sqcnt].clear();
				addedge(h1,i,sqcnt,0);
				ele k;
				do{
					k=stk[--top];
					addedge(h1,sqcnt,k,min(d[k]-d[i],D[sqcnt]-d[k]+d[i]));
					vec[sqcnt].push_back(k);
				}while (k!=j->v);
				++sqcnt;
			}
			else if (dfn[i]<low[j->v]) addedge(h1,i,j->v,j->w),--top;
		}
		else if (j->v!=p) low[i]=min(low[i],dfn[j->v]);
}
void dfs(ele p,ele i){
	f[i]=0;
	for (edge *j=h1[i]; j; j=j->nxt)
		if (j->v!=p){
			dfs(i,j->v);
			f[i]=max(f[i],f[j->v]+j->w);
		}
	if (i>=n){
		static pair<ele,ele> Q[maxn];
		ele qh=0,qt=0,tmp=-1;
		for (int j=0; j<vec[i].size(); ++j){
			ele k=vec[i][j];
			while (qh<qt && (j-Q[qh].second)*2>D[i]){
				tmp=max(tmp,f[Q[qh].first]+Q[qh].second);
				++qh;
			}
			if (qh<qt) ans=max(ans,f[k]+f[Q[qh].first]+j-Q[qh].second);
			if (~tmp) ans=max(ans,D[i]+f[k]-j+tmp);
			while (qh<qt && f[Q[qt-1].first]-Q[qt-1].second<f[k]-j) --qt;
			Q[qt++]=make_pair(k,j);
		}
	}
	else{
		ele mx=0;
		for (edge *j=h1[i]; j; j=j->nxt)
			if (j->v!=p){
				ans=max(ans,mx+f[j->v]+j->w);
				mx=max(mx,f[j->v]+j->w);
			}
	}
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=ep; memset(h,0,sizeof(h)); memset(h1,0,sizeof(h1));
	while (m--){
		ele k;
		static ele ts[maxn];
		scanf("%d",&k);
		for (int i=0; i<k; ++i) scanf("%d",ts+i);
		for (int i=0; i<k-1; ++i) addedge(h,ts[i]-1,ts[i+1]-1,1);
	}
	tcnt=0; top=0; sqcnt=n; d[0]=0;
	memset(dfn,-1,sizeof(dfn));
	tarjan(-1,0);
	ans=0;
	dfs(-1,0);
	printf("%d\n",ans);
	return 0;
}

代码（bzoj4316）：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#define ele int
using namespace std;
#define maxn 60010
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<2],*ecnt;
ele n,m,sqcnt,top,tcnt,stk[maxn],dfn[maxn],low[maxn],f[maxn],g[maxn];
edge *h[maxn],*h1[maxn];
vector<ele> vec[maxn];
inline void addedge1(edge **h,ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
inline void addedge(edge **h,ele u,ele v){
	addedge1(h,u,v); addedge1(h,v,u);
}
void tarjan(ele p,ele i){
	low[i]=dfn[i]=tcnt++;
	stk[top++]=i;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!~dfn[j->v]){
			tarjan(i,j->v);
			low[i]=min(low[i],low[j->v]);
			if (dfn[i]==low[j->v]){
				addedge(h1,i,sqcnt);
				vec[sqcnt].clear();
				ele k;
				do{
					k=stk[--top];
					addedge(h1,sqcnt,k);
					vec[sqcnt].push_back(k);
				}while (k!=j->v);
				++sqcnt;
			}
			else if (dfn[i]<low[j->v]) addedge(h1,i,j->v),--top;
		}
		else if (j->v!=p) low[i]=min(low[i],dfn[j->v]);
}
void dfs(ele p,ele i){
	for (edge *j=h1[i]; j; j=j->nxt)
		if (j->v!=p) dfs(i,j->v);
	if (i<n){
		f[i]=1; g[i]=0;
		for (edge *j=h1[i]; j; j=j->nxt)
			if (j->v!=p){
				f[i]+=g[j->v];
				g[i]+=max(f[j->v],g[j->v]);
			}
	}
	else{
		static ele t1[maxn],t2[maxn];
		t1[0]=t2[0]=g[vec[i][0]];
		for (int j=1; j<vec[i].size(); ++j){
			ele k=vec[i][j];
			t1[j]=t2[j-1]+f[k];
			t2[j]=max(t1[j-1],t2[j-1])+g[k];
		}
		g[i]=t2[vec[i].size()-1];
		t1[0]=f[vec[i][0]]; t2[0]=g[vec[i][0]];
		for (int j=1; j<vec[i].size(); ++j){
			ele k=vec[i][j];
			t1[j]=t2[j-1]+f[k];
			t2[j]=max(t1[j-1],t2[j-1])+g[k];
		}
		f[i]=max(t1[vec[i].size()-1],t2[vec[i].size()-1]);
	}
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=ep; memset(h,0,sizeof(h)); memset(h1,0,sizeof(h1));
	for (int i=0; i<m; ++i){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		addedge(h,u,v);
	}
	sqcnt=n; top=0; tcnt=0;
	memset(dfn,-1,sizeof(dfn));
	for (int i=0; i<n; ++i)
		if (!~dfn[i]) tarjan(-1,i);
	memset(f,-1,sizeof(f));
	ele ans=0;
	for (int i=0; i<n; ++i)
		if (!~f[i]) dfs(-1,i),ans+=max(f[i],g[i]);
	printf("%d\n",ans);
	return 0;
}
