bzoj1069题解

Aug 15, 2017 10:02 · 162 words · 1 minute read 题解 bzoj 计算几何 旋转卡壳

先考虑一个简单一点的问题：找一个面积最大的三角形。一个很自然的想法是，枚举其中一边，并且找出离这条边最远的点。但是这个想法的朴素实现的复杂度是$O(n^3)$的。这个时候就可以使用类似旋转卡壳的思想。显然最优解在凸包上，固定边的其中一个端点，另一个端点沿凸包逆时针移动的时候，离这条边最远的点也一定沿凸包逆时针移动，并且距离变化是单峰的。这样就能将算法优化到$O(n^2)$。

找面积最大的四边形也是一个道理，枚举对角线，并且在对角线两边各找出一个最远的点。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 2010
#define eps 1e-6
#define vec pt
struct pt{
	double x,y;
	inline bool operator<(pt b)const{
		if (fabs(x-b.x)>eps) return x<b.x;
		if (fabs(y-b.y)>eps) return y<b.y;
		return false;
	}
};
ele n,cnt;
pt p[maxn],a[maxn];
inline vec operator+(vec a,vec b){
	return (vec){a.x+b.x,a.y+b.y};
}
inline vec operator-(vec a,vec b){
	return (vec){a.x-b.x,a.y-b.y};
}
inline double dot(vec a,vec b){ return a.x*b.x+a.y*b.y; }
inline double cross(vec a,vec b){ return a.x*b.y-a.y*b.x; }
inline double area(pt a,pt b,pt c){
	return fabs(cross(b-a,c-a));
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%lf%lf",&p[i].x,&p[i].y);
	sort(p,p+n);
	cnt=0;
	for (int i=0; i<n; ++i){
		while (cnt>1 && cross(a[cnt-1]-a[cnt-2],p[i]-a[cnt-1])<eps) --cnt;
		a[cnt++]=p[i];
	}
	ele tmp=cnt;
	for (int i=n-1; ~i; --i){
		while (cnt>tmp+1 && cross(a[cnt-1]-a[cnt-2],p[i]-a[cnt-1])<eps) --cnt;
		a[cnt++]=p[i];
	}
	for (int i=tmp; i<cnt; ++i) a[i]=a[i+1];
	--cnt;
	double ans=0;
	for (int i=0; i<cnt-1; ++i){
		ele u=i,v=i+1;
		for (int j=i+1; j<cnt-1; ++j){
			while (area(a[i],a[j],a[u])+eps<area(a[i],a[j],a[u+1])) (u+=1)%=(cnt-1);
			while (area(a[i],a[j],a[v])+eps<area(a[i],a[j],a[v+1])) (v+=1)%=(cnt-1);
			ans=max(ans,area(a[i],a[j],a[u])+area(a[i],a[j],a[v]));
		}
	}
	printf("%.3lf\n",ans/2);
	return 0;
}
