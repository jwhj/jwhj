bzoj2741题解

Dec 5, 2016 10:02 · 216 words · 2 minutes read 题解 bzoj 分块 trie

首先求出每个前缀的异或值，那么询问就可以转化为，在区间中找两个值，使它们异或起来最大。

考虑在线莫队，如果不会的话看一下这篇题解。

把区间中的数的二进制用一个trie维护起来，对于一个新的数x，可以$O(\log x)$地找到x与区间中某个数异或的最大值。

令f1[i]表示[0,i]中的数组成的trie，用可持久化的方法把他们维护起来，就可以用在线莫队来做了。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 12010
#define maxs 120
#define maxk 31
struct node{
    ele cnt;
    node *c[2];
}np[5000010];
ele n,m,S,ncnt,a[maxn],a1[maxn],f[maxs][maxn];
node *f1[maxn];
inline node* newnode(){
    node *p=&np[ncnt++];
    p->cnt=0;
    p->c[0]=p->c[1]=NULL;
    return p;
}
inline ele rev(ele x){
    ele ans=0;
    for (int i=0; i<maxk; ++i,x>>=1)
        ans<<=1,ans|=(x&1);
    return ans;
}
node* ins(node *x,ele k,ele d){
    if (d<0) return NULL;
    ele y=k&1;
    node *p=newnode();
    if (x)
        *p=*x;
    else p->cnt=1;
    p->c[y]=ins(p->c[y],k>>1,d-1);
    if (p->c[0])
        p->cnt+=p->c[0]->cnt;
    if (p->c[1])
        p->cnt+=p->c[1]->cnt;
    return p;
}
inline ele query(node *l,node *r,ele k){
    ele ans=0;
    for (int i=0; i<maxk; ++i,k>>=1){
        ele y=(k&1)^1;
        ans<<=1;
        ele rc=(r->c[y])?(r->c[y]->cnt):(0);
        ele lc=(l && l->c[y])?(l->c[y]->cnt):(0);
        if (rc-lc)
            ans|=1;
        else y^=1;
        if (l)
            l=l->c[y];
        r=r->c[y];
    }
    return ans;
}
int main(){
    scanf("%d%d",&n,&m);
    for (int i=0; i<n; ++i)
        scanf("%d",a1+i);
    a[0]=0;
    for (int i=1; i<=n; ++i)
        a[i]=a[i-1]^a1[i-1];
    ++n;
    S=sqrt(n);
    ncnt=0;
    f1[0]=ins(NULL,0,maxk);
    for (int i=1; i<n; ++i)
        f1[i]=ins(f1[i-1],rev(a[i]),maxk);
    for (int i=0; i*S<n; ++i){
        ele j=i*S;
        f[i][j]=0;
        for (int k=j+1; k<n; ++k)
            if (i==0)
                f[i][k]=max(f[i][k-1],query(NULL,f1[k-1],rev(a[k])));
            else f[i][k]=max(f[i][k-1],query(f1[i*S-1],f1[k-1],rev(a[k])));
    }
    ele ans=0;
    while (m--){
        long long u,v;
        scanf("%lld%lld",&u,&v);
        u=(u+ans)%(n-1)+1; v=(v+ans)%(n-1)+1;
        ele l=min(u,v),r=max(u,v);
        --l;
        if (l/S==r/S){
            ans=0;
            for (int j=r-1; j>=l; --j)
                ans=max(ans,query(f1[j],f1[r],rev(a[j])));
        }
        else{
            ele i1=l/S;
            ans=f[i1+1][r];
            for (int j=i1*S+S-1; j>=l; --j)
                ans=max(ans,query(f1[j],f1[r],rev(a[j])));
        }
        printf("%d\n",ans);
    }
    return 0;
}
