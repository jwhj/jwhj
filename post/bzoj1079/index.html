bzoj1079题解

Aug 19, 2017 10:02 · 73 words · 1 minute read 题解 bzoj 记忆化搜索

注意到ci<=5以及对于某一个状态，剩余次数相同的颜色地位是差不多的，可以考虑把状态表示为(a,b,c,d,e,k)，即剩下1次的有a种……剩下5次的有e种，当前最后一位的颜色还剩k次，然后记忆化搜索就可以了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxk 17
#define maxc 6
#define MOD 1000000007
ele k,c[maxk],cnt[maxc],f[maxk][maxk][maxk][maxk][maxk][maxc];
ele solve(ele a,ele b,ele c,ele d,ele e,ele k){
	if (~f[a][b][c][d][e][k]) return f[a][b][c][d][e][k];
	ele ans=0;
	if (a-(k==1)) (ans+=(a-(k==1))*solve(a-1,b,c,d,e,0))%=MOD;
	if (b-(k==2)) (ans+=(b-(k==2))*solve(a+1,b-1,c,d,e,1))%=MOD;
	if (c-(k==3)) (ans+=(c-(k==3))*solve(a,b+1,c-1,d,e,2))%=MOD;
	if (d-(k==4)) (ans+=(d-(k==4))*solve(a,b,c+1,d-1,e,3))%=MOD;
	if (e-(k==5)) (ans+=(e-(k==5))*solve(a,b,c,d+1,e-1,4))%=MOD;
	return f[a][b][c][d][e][k]=ans;
}
int main(){
	scanf("%d",&k);
	for (int i=0; i<k; ++i) scanf("%d",c+i),++cnt[c[i]];
	memset(f,-1,sizeof(f));
	f[0][0][0][0][0][0]=1;
	printf("%d\n",solve(cnt[1],cnt[2],cnt[3],cnt[4],cnt[5],0));
	return 0;
}
