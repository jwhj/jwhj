bzoj1177题解

Jul 30, 2017 10:02 · 195 words · 1 minute read 题解 bzoj dp

先考虑最简单的情况：取一个k×k的矩形。这个……不用考虑了吧……

再考虑一个复杂一些的情况：取两个k×k的矩形。易知，两个矩形投影到行上或投影到列上，至少有一种情况，两个矩形的投影是不重合的。那么只需要枚举一下行或列上的分界线就可以分别dp出来了。

最后考虑原题的情况：取三个k×k的矩形。可以证明，投影到行上或投影到列上，至少有一种情况，其中一个矩形的投影与其它矩形不重合。枚举这个分界线，然后一边可以看成两个矩形的情况，一边可以看成一个矩形的情况。

要写一大堆dp挺麻烦的。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 1510
ele n,m,k,ans,a[maxn][maxn],b[maxn][maxn],f[4][maxn],g[4][maxn][maxn];
inline ele sum(ele i,ele j){
	return b[i][j]-b[i-k][j]-b[i][j-k]+b[i-k][j-k];
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	memset(b,0,sizeof(b));
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j) scanf("%d",&a[i][j]),b[i+1][j+1]=a[i][j];
	for (int i=1; i<=n; ++i){
		for (int j=1; j<=m; ++j) b[i][j]+=b[i][j-1];
		for (int j=0; j<=m; ++j) b[i][j]+=b[i-1][j];
	}
	ans=0;
	memset(f,0,sizeof(f));
	for (int p=1; p<=3; ++p)
		for (int i=1; i<=n; ++i){
			f[p][i]=f[p][i-1];
			if (i>=k)
				for (int j=k; j<=m; ++j)
					f[p][i]=max(f[p][i],f[p-1][i-k]+sum(i,j));
		}
	ans=max(ans,f[3][n]);
	memset(f,0,sizeof(f));
	for (int p=1; p<=3; ++p)
		for (int j=1; j<=m; ++j){
			f[p][j]=f[p][j-1];
			if (j>=k)
				for (int i=k; i<=n; ++i)
					f[p][j]=max(f[p][j],f[p-1][j-k]+sum(i,j));
		}
	ans=max(ans,f[3][m]);
	for (int i=1; i<=n; ++i)
		for (int j=1; j<=m; ++j)
			if (i<k || j<k) g[0][i][j]=0;
			else g[0][i][j]=max(max(g[0][i-1][j],g[0][i][j-1]),sum(i,j));
	for (int i=1; i<=n; ++i)
		for (int j=m; j; --j)
			if (i<k || m-j+1<k) g[1][i][j]=0;
			else g[1][i][j]=max(max(g[1][i-1][j],g[1][i][j+1]),sum(i,j+k-1));
	for (int i=n; i; --i)
		for (int j=1; j<=m; ++j)
			if (n-i+1<k || j<k) g[2][i][j]=0;
			else g[2][i][j]=max(max(g[2][i+1][j],g[2][i][j-1]),sum(i+k-1,j));
	for (int i=n; i; --i)
		for (int j=m; j; --j)
			if (n-i+1<k || m-j+1<k) g[3][i][j]=0;
			else g[3][i][j]=max(max(g[3][i+1][j],g[3][i][j+1]),sum(i+k-1,j+k-1));
	for (int i=k; i<=n-k; ++i)
		for (int j=k; j<=m-k; ++j){
			ans=max(ans,g[0][i][j]+g[1][i][j+1]+g[2][i+1][m]);
			ans=max(ans,g[2][i+1][j]+g[3][i+1][j+1]+g[0][i][m]);
			ans=max(ans,g[0][i][j]+g[2][i+1][j]+g[1][n][j+1]);
			ans=max(ans,g[0][n][j]+g[1][i][j+1]+g[3][i+1][j+1]);
		}
	printf("%d\n",ans);
	return 0;
}
