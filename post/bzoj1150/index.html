bzoj1150题解

May 6, 2016 10:02 · 108 words · 1 minute read 题解 bzoj 贪心 乱搞 优先队列

考虑有1，2，3，4四栋楼，与其这样：

1-----4
  2-3

不如这样:

1-2 3-4

所以可以得到，最优解只能把相邻的楼配对。于是可以把n栋楼看成n-1条边，从中取k条，而且任意两条不能紧挨着。

数据小的话，dp肯定是能做的。

可以想到一个贪心策略：每次取最小边。但这是错的。

于是就有一种奇怪的乱搞：把所有边排成一行（它们貌似本来就是一行），每次取最小边，假设它的权为b，左右的权为a和c。删除左右的两边，把这条边的权改为a+c-b。

这是什么意思呢？可以发现，如果以后再次选了这条权为a+c-b边，就相当于放弃原来的b去选择a和c。

所以这么做的意义在于允许之前的某次贪心选择是错误的，可以在后面的某次选择中改正。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#define ele int
using namespace std;
#define maxn 100010
#define INF 1000000001
struct st{
	ele p,i;
	inline bool operator<(st b)const{
		if (p>b.p) return true;
		if (p<b.p) return false;
		return i>b.i;
	}
};
ele n,k,l[maxn],r[maxn],a[maxn];
bool flag[maxn];
priority_queue<st> q;
int main(){
	ele x,y;
	scanf("%d%d%d",&n,&k,&x);
	for (int i=1; i<n; ++i){
		scanf("%d",&y);
		a[i]=y-x;
		q.push((st){y-x,i});
		x=y;
	}
	a[0]=a[n]=INF;
	q.push((st){INF,0}); q.push((st){INF,n});
	for (int i=0; i<=n; ++i) l[i]=i-1,r[i]=i+1;
	l[0]=0; r[n]=n;
	memset(flag,0,sizeof(flag));
	ele tmp=0,ans=0;
	while (tmp!=k){
		st p=q.top(); q.pop();
		if (flag[p.i]) continue;
		ans+=p.p;
		flag[l[p.i]]=flag[r[p.i]]=true;
		a[p.i]=a[l[p.i]]+a[r[p.i]]-a[p.i];
		q.push((st){a[p.i],p.i});
		l[p.i]=l[l[p.i]]; r[l[p.i]]=p.i;
		r[p.i]=r[r[p.i]]; l[r[p.i]]=p.i;
		++tmp;
	}
	printf("%d\n",ans);
	return 0;
}
