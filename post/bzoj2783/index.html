##bzoj150题留念。

dfs整棵树，维护一个队列，每次某个点入栈，就把它加到队尾，并调整队首，使得队中元素之和恰好<=s。这个点出栈的时候就把队首恢复原位。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 100010
struct edge{
	ele v;
	edge *nxt;
}ep[maxn];
ele n,s,ecnt,ans,w[maxn],q[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele i,ele t1,ele t2,ele sum){
	q[t2++]=i;
	sum+=w[i];
	while (sum>s && t1<t2){
		sum-=w[q[t1]];
		++t1;
	}
	if (sum==s)
		++ans;
	for (edge *j=h[i]; j; j=j->nxt)
		dfs(j->v,t1,t2,sum);
}
int main(){
	scanf("%d%d",&n,&s);
	for (int i=0; i<n; ++i)
		scanf("%d",w+i);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v);
		--u,--v;
		addedge(u,v);
	}
	ans=0;
	dfs(0,0,0,0);
	printf("%d\n",ans);
	return 0;
}
