bzoj3110题解

Jan 15, 2017 10:02 · 196 words · 1 minute read 题解 bzoj 整体二分

题目写得比较奇怪……应该理解成每个位置可以堆很多个数。

用整体二分就可以做了。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 100010
struct ASK{
	ele l,r,k,t,id;
};
struct BIT{
	inline void alloc(ele n){
		memset(a,0,sizeof(a));
		this->n=n;
	}
	inline void add(ele i,ele k){
		for (int j=i; j<=n; j+=j&(-j))
			a[j]+=k;
	}
	inline ele q(ele i){
		ele ans=0;
		for (int j=i; j>0; j-=j&(-j))
			ans+=a[j];
		return ans;
	}
	ele n,a[maxn];
};
struct SEG{
	inline void alloc(ele n){
		a.alloc(n+1);
		b.alloc(n+1);
	}
	inline void add(ele l,ele r,ele k){
		a.add(l,k);
		a.add(r+1,-k);
		b.add(l,k*l);
		b.add(r+1,-k*(r+1));
	}
	inline ele q(ele i){
		return (i+1)*a.q(i)-b.q(i);
	}
	inline ele q(ele l,ele r){
		return q(r)-q(l-1);
	}
	BIT a,b;
};
ele n,m,res[maxn];
ASK q[maxn],q1[maxn];
SEG s;
inline void solve(ele l,ele r,ele a,ele b,ASK *q,ASK *q1){
	if (r-l<=1){
		for (int i=a; i<=b; ++i)
			if (q[i].t==2)
				res[q[i].id]=l;
	}
	else{
		ele u=a,v=b,mid=(l+r)/2,tmp;
		for (int i=a; i<=b; ++i)
			if (q[i].t==1)
				if (q[i].k<mid){
					q1[u++]=q[i];
					s.add(q[i].l,q[i].r,1);
				}
				else q1[v--]=q[i];
			else
				if (q[i].k<=(tmp=s.q(q[i].l,q[i].r)))
					q1[u++]=q[i];
				else q1[v]=q[i],q1[v--].k-=tmp;
		for (int i=a; i<u; ++i){
			q[i]=q1[i];
			if (q[i].t==1)
				s.add(q[i].l,q[i].r,-1);
		}
		for (int i=u; i<=b; ++i)
			q[i]=q1[u+b-i];
		solve(l,mid,a,v,q,q1);
		solve(mid,r,u,b,q,q1);
	}
}
int main(){
	scanf("%lld%lld",&n,&m);
	for (int i=0; i<m; ++i){
		scanf("%lld%lld%lld%lld",&q[i].t,&q[i].l,&q[i].r,&q[i].k);
		if (q[i].t==1)
			q[i].k=n+1-q[i].k;
		q[i].id=i;
	}
	s.alloc(n);
	memset(res,-1,sizeof(res));
	solve(1,n*2+2,0,m-1,q,q1);
	for (int i=0; i<m; ++i)
		if (res[i]!=-1)
			printf("%lld\n",n+1-res[i]);
	return 0;
}
