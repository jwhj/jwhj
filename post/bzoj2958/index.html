bzoj2958题解

Feb 23, 2017 10:02 · 75 words · 1 minute read 题解 bzoj DP

令f[i][j][k]表示0~i这段，状态为j，最后的一个字符为k的方案数。其中j=2代表已经出现了一段长大于等于k的B、一段长大于等于k的W；j=1代表只出现了一段长大于等于k的B；j=0代表都没有出现。k=0代表B，k=1代表W。

转移起来比较麻烦，看代码吧……

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 1000010
#define MOD 1000000007
ele n,k,f[maxn][3][2],g[maxn][2];
char s[maxn];
int main(){
	scanf("%lld%lld%s",&n,&k,s);
	memset(f,0,sizeof(f));
	f[0][0][0]=(s[0]!='W')?(1):(0);
	f[0][0][1]=(s[0]!='B')?(1):(0);
	f[0][1][0]=(k==1)?(f[0][0][0]):(0);
	f[0][1][1]=0;
	f[0][2][0]=f[0][2][1]=0;
	g[0][0]=(s[0]!='W')?(1):(0);
	g[0][1]=(s[0]!='B')?(1):(0);
	for (int i=1; i<n; ++i){
		g[i][0]=(s[i]!='W')?(g[i-1][0]+1):(0);
		g[i][1]=(s[i]!='B')?(g[i-1][1]+1):(0);
		if (s[i]!='W'){
			ele t=(i>=k && g[i][0]>=k)?(f[i-k][0][1]):(0);
			if (i==k-1 && g[i][0]>=k) t=1;
			f[i][0][0]=(f[i-1][0][0]+f[i-1][0][1]-t)%MOD;
			f[i][1][0]=(f[i-1][1][0]+f[i-1][1][1]+t)%MOD;
			f[i][2][0]=(f[i-1][2][0]+f[i-1][2][1])%MOD;
		}
		if (s[i]!='B'){
			ele t=(i>=k && g[i][1]>=k)?(f[i-k][1][0]):(0);
			f[i][0][1]=(f[i-1][0][0]+f[i-1][0][1])%MOD;
			f[i][1][1]=(f[i-1][1][0]+f[i-1][1][1]-t)%MOD;
			f[i][2][1]=(f[i-1][2][0]+f[i-1][2][1]+t)%MOD;
		}
	}
	ele ans=(f[n-1][2][0]+f[n-1][2][1])%MOD;
	(ans+=MOD)%=MOD;
	printf("%lld\n",ans);
	return 0;
}
