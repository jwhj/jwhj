bzoj2326题解

Jul 5, 2016 10:02 · 157 words · 1 minute read 题解 bzoj 矩阵

记$f_i=Concatenate(1..i)$。

如果i是k位数，就有$fi=10^kf{i-1}+i$，利用这个，就可以在$O(n)$的时间内求出答案了。

然而这么做还是慢。注意到 $$ \begin{bmatrix} fi\
i\
1 \end{bmatrix}= \begin{bmatrix} 10^k & 1 & 1\
0 & 1 & 1\
0 & 0 & 1 \end{bmatrix} \begin{bmatrix} f{i-1}\
i-1\
1 \end{bmatrix} $$

对于k相同的i，没必要一个一个地把f值算出来，可以用矩阵快速幂来做。

小心$10^{18}$再乘10溢出。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
struct MAT
{
	ele n,m,a[5][5];
};
ele n,m;
MAT ans,a;
inline MAT operator*(MAT a,MAT b)
{
	MAT ans;
	ans.n=a.n; ans.m=b.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<b.m; ++j)
		{
			ans.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				ans.a[i][j]+=a.a[i][k]*b.a[k][j]%m,
				ans.a[i][j]%=m;
		}
	return ans;
}
inline MAT pow(MAT a,ele x)
{
	MAT ans=a,tmp=a;
	memset(ans.a,0,sizeof(ans.a));
	for (int i=0; i<a.n; ++i) ans.a[i][i]=1;
	for (; x; x>>=1,tmp=tmp*tmp)
		if (x&1) ans=ans*tmp;
	return ans;
}
int main()
{
	scanf("%lld%lld",&n,&m);
	ele c=1;
	ans=(MAT){3,1,{{0},{-1},{1}}};
	while (c<=n)
	{
		a=(MAT){3,3,{{c*10%m,1,1},{0,1,1},{0,0,1}}};
		if (c==1)
			if (n>=10) a=pow(a,10);
			else a=pow(a,n+1);
		else
			if (c*10<=n &&
				!(c==n && n==1000000000000000000LL)) a=pow(a,c*9);
			else a=pow(a,n-c+1);
		ans=a*ans;
		if (c==n && n==1000000000000000000LL) break;
		c*=10;
	}
	if (ans.a[0][0]<0) ans.a[0][0]+=m;
	printf("%lld\n",ans.a[0][0]);
	return 0;
}
