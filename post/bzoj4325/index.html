bzoj4325题解

Nov 8, 2016 10:02 · 149 words · 1 minute read 题解 bzoj

以前一直认为是码农题，今天打了一下感觉还是比较好打的，码量绝对比前几天的一些题少……

显然出牌顺序是无关紧要的，所以先用dfs枚举怎么出顺子。对于剩下的牌，牌的点数大小已经不重要了，所以可以用(a,b,c,d)来表示状态，指的是有a张单牌，b种两张的牌，c种三张的牌，d种4张的牌。

然后记忆化搜索一下就可以得到答案了。

不过这道题被认为是码农题也不是没道理的，有好些细节要考虑一下，比如说两张鬼不能当成对子放在三带二里面，出王炸不一定是最优的。有些地方写起来比较麻烦，比如出三带一的时候，如果没有单牌，一样有可能需要从多张一样的牌里抽一张出来用。因此状态转移的时候要考虑非常多种情况，可以看着我的代码感受一下……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxc 15
#define INF 1000000000
ele n,a[maxc],b[5],f[maxc][maxc][maxc][maxc];
ele g(ele a,ele b,ele c,ele d){
	if (a<0 || b<0 || c<0 || d<0) return INF;
	if (f[a][b][c][d]!=-1) return f[a][b][c][d];
	f[a][b][c][d]=a+b+c+d;
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a-1,b,c-1,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a+1,b-1,c-1,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b+1,c-2,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b,c,d-1));

	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b-1,c-1,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a+1,b,c-2,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b+1,c-1,d-1));

	f[a][b][c][d]=min(f[a][b][c][d],1+g(a-2,b,c,d-1));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b-1,c,d-1));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a-1,b+1,c-1,d-1));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a-1,b,c+1,d-2));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a+1,b,c-1,d-1));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b-1,c+1,d-2));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b+1,c,d-2));

	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b-2,c,d-1));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a+1,b-1,c-1,d));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a,b,c,d-2));
	f[a][b][c][d]=min(f[a][b][c][d],1+g(a+1,b+1,c-1,d-2));
	return f[a][b][c][d];
}
ele dfs(){
	memset(b,0,sizeof(b));
	for (int i=0; i<maxc; ++i)
		++b[a[i]];
	ele ans=g(b[1],b[2],b[3],b[4]);
	if (a[0] && a[1])
		ans=min(ans,1+g(b[1]-2,b[2],b[3],b[4]));
	for (int t=1; t<=3; ++t)
		for (int i=3; i<maxc-1; ++i)
			if (a[i]>=t){
				ele j;
				for (j=i; j<maxc && a[j]>=t; ++j){
					a[j]-=t;
					if ((j-i+1)*t>=5)
						ans=min(ans,1+dfs());
				}
				for (--j; j>=i; --j)
					a[j]+=t;
			}
	return ans;
}
int main(){
	ele T;
	scanf("%d%d",&T,&n);
	memset(f,-1,sizeof(f));
	f[0][0][0][0]=0;
	while (T--){
		memset(a,0,sizeof(a));
		for (int i=0; i<n; ++i){
			ele u,v;
			scanf("%d%d",&u,&v);
			if (u==0 && v==2)
				++a[14];
			else
				++a[u];
		}
		swap(a[1],a[14]);
		printf("%d\n",dfs());
	}
	return 0;
}
