CF718C题解

Sep 24, 2016 10:02 · 478 words · 3 minutes read 题解 codeforces 线段树 矩阵 数学

令$b_i=\begin{bmatrix}f(a_i)\\f(a_i-1)\end{bmatrix}$，用线段树来维护序列${b_i}$。

注意到$a_i$加上x等价于$b_i$乘上$\begin{bmatrix}1&1\\1& 0\end{bmatrix}^x$，所以线段树上打一个标记就能支持修改操作了。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 100010
#define maxmn 2
#define MOD 1000000007
struct matrix{
	ele n,m,a[maxmn][maxmn];
	inline void I(){
		memset(a,0,sizeof(a));
		for (int i=0; i<n; ++i) a[i][i]=1;
	}
};
inline matrix operator+(matrix a,matrix b){
	matrix ans;
	ans.n=a.n; ans.m=a.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<a.m; ++j) ans.a[i][j]=(a.a[i][j]+b.a[i][j])%MOD;
	return ans;
}
inline matrix operator*(matrix a,matrix b){
	matrix ans;
	ans.n=a.n; ans.m=b.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<b.m; ++j){
			ans.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				ans.a[i][j]+=a.a[i][k]*b.a[k][j],
				ans.a[i][j]%=MOD;
		}
	return ans;
}
inline matrix pow(matrix a,ele x){
	matrix ans,tmp=a;
	ans.n=ans.m=a.n;
	ans.I();
	for (; x>0; x>>=1,tmp=tmp*tmp)
		if (x&1) ans=ans*tmp;
	return ans;
}
struct node{
	matrix sum,mul;
	node *l,*r;
}pool[maxn<<2];
ele ncnt,a[maxn];
inline void maintain(node *x){
	x->sum=x->l->sum+x->r->sum;
	x->sum=x->mul*x->sum;
}
node* build(ele l,ele r){
	node *p=&pool[ncnt++];
	p->mul.n=p->mul.m=2;
	p->mul.I();
	if (l==r){
		p->sum=(matrix){2,1,{{1},{0}}};
		matrix C=(matrix){2,2,{{1,1},{1,0}}};
		p->sum=pow(C,a[l]-1)*p->sum;
		p->l=p->r=NULL;
	}
	else{
		ele mid=l+r>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		maintain(p);
	}
	return p;
}
void upd(node *x,ele a,ele b,ele l,ele r,matrix k){
	if (l<=a && b<=r) x->mul=x->mul*k,x->sum=k*x->sum;
	else{
		ele mid=(a+b)>>1;
		if (l<=mid) upd(x->l,a,mid,l,r,k);
		if (mid<r) upd(x->r,mid+1,b,l,r,k);
		maintain(x);
	}
}
matrix query(node *x,ele a,ele b,ele l,ele r){
	if (l<=a && b<=r) return x->sum;
	ele mid=(a+b)>>1;
	matrix ans=(matrix){2,1,{{0},{0}}};
	if (l<=mid) ans=ans+query(x->l,a,mid,l,r);
	if (mid<r) ans=ans+query(x->r,mid+1,b,l,r);
	ans=x->mul*ans;
	return ans;
}
ele n,m;
node *root;
int main(){
	scanf("%lld%lld",&n,&m);
	for (int i=0; i<n; ++i) scanf("%lld",a+i);
	ncnt=0;
	root=build(0,n-1);
	while (m--){
		ele cmd,l,r,x;
		scanf("%lld%lld%lld",&cmd,&l,&r);
		--l,--r;
		if (cmd==1){
			scanf("%lld",&x);
			matrix C=(matrix){2,2,{{1,1},{1,0}}};
			upd(root,0,n-1,l,r,pow(C,x));
		}
		else printf("%lld\n",query(root,0,n-1,l,r).a[0][0]);
	}
	return 0;
}

其实不用线段树，用tagged zkw也是可以的：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 100010
#define maxmn 2
#define MOD 1000000007
struct matrix{
	ele n,m,a[maxmn][maxmn];
	inline void I(){
		memset(a,0,sizeof(a));
		for (int i=0; i<n; ++i) a[i][i]=1;
	}
};
inline matrix operator+(matrix a,matrix b){
	matrix ans;
	ans.n=a.n; ans.m=a.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<a.m; ++j) ans.a[i][j]=(a.a[i][j]+b.a[i][j])%MOD;
	return ans;
}
inline matrix operator*(matrix a,matrix b){
	matrix ans;
	ans.n=a.n; ans.m=b.m;
	for (int i=0; i<a.n; ++i)
		for (int j=0; j<b.m; ++j){
			ans.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				ans.a[i][j]+=a.a[i][k]*b.a[k][j],
				ans.a[i][j]%=MOD;
		}
	return ans;
}
inline matrix pow(matrix a,ele x){
	matrix ans,tmp=a;
	ans.n=ans.m=a.n;
	ans.I();
	for (; x>0; x>>=1,tmp=tmp*tmp)
		if (x&1) ans=ans*tmp;
	return ans;
}
const ele M=1<<17;
const ele size=M<<1;
matrix tmp[maxn];
inline void maintain(matrix *d,matrix *tag,ele i){
	if ((i<<1|1)<size) d[i]=d[i<<1]+d[i<<1|1]; else d[i]=tmp[i-M-1];
	d[i]=tag[i]*d[i];
}
inline void upd(matrix *d,matrix *tag,ele l,ele r,matrix k){
	for (l=l+M-1,r=r+M+1; l^r^1; l>>=1,r>>=1){
		maintain(d,tag,l); maintain(d,tag,r);
		if (~l&1) tag[l^1]=k*tag[l^1],maintain(d,tag,l^1);
		if (r&1) tag[r^1]=k*tag[r^1],maintain(d,tag,r^1);
	}
	maintain(d,tag,l); maintain(d,tag,r);
	while (l>1) maintain(d,tag,l>>=1);
}
inline matrix query(matrix *d,matrix *tag,ele l,ele r){
	matrix lans,rans;
	lans=rans=(matrix){2,1,{{0},{0}}};
	for (l=l+M-1,r=r+M+1; l^r^1; l>>=1,r>>=1){
		lans=tag[l]*lans; rans=tag[r]*rans;
		if (~l&1) lans=lans+d[l^1];
		if (r&1) rans=rans+d[r^1];
	}
	lans=tag[l]*lans; rans=tag[r]*rans;
	lans=lans+rans;
	while (l>1) lans=tag[l>>=1]*lans;
	return lans;
}
ele n,m,a[maxn];
matrix d[size],tag[size];
int main(){
	scanf("%lld%lld",&n,&m);
	matrix C=(matrix){2,2,{{1,1},{1,0}}};
	for (int i=0; i<n; ++i){
		scanf("%lld",a+i);
		d[i+M+1]=(matrix){2,1,{{1},{0}}};
		tmp[i]=d[i+M+1]=pow(C,a[i]-1)*d[i+M+1];
	}
	for (int i=1; i<=size; ++i) tag[i].n=tag[i].m=2,tag[i].I();
	for (int i=M-1; i>0; --i) d[i].n=2,d[i].m=1,maintain(d,tag,i);
	while (m--){
		ele cmd,l,r,x;
		scanf("%lld%lld%lld",&cmd,&l,&r);
		if (cmd==1){
			scanf("%lld",&x);
			upd(d,tag,l,r,pow(C,x));
		}
		else printf("%lld\n",query(d,tag,l,r).a[0][0]);
	}
	return 0;
}
