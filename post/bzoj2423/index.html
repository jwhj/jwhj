bzoj2423题解

Sep 9, 2016 10:02 · 85 words · 1 minute read 题解 bzoj DP

DP硬上，细节看代码。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define MOD 100000000
#define maxn 5010
ele n,m,f[2][maxn],g[2][maxn];
char s1[maxn],s2[maxn];
int main()
{
	scanf("%s%s",s1,s2);
	n=strlen(s1); m=strlen(s2);
	--n,--m;
	memset(f[0],0,sizeof(f[0]));
	for (int i=0; i<=m; ++i) g[0][i]=1;
	ele p=0;
	for (int i=1; i<=n; ++i)
	{
		p^=1;
		f[p][0]=0; g[p][0]=1;
		for (int j=1; j<=m; ++j)
			if (s1[i-1]==s2[j-1])
			{
				f[p][j]=f[p^1][j-1]+1;
				g[p][j]=g[p^1][j-1];
				if (f[p][j]==f[p^1][j]) g[p][j]+=g[p^1][j],g[p][j]%=MOD;
				if (f[p][j]==f[p][j-1]) g[p][j]+=g[p][j-1],g[p][j]%=MOD;
			}
			else
			{
				f[p][j]=max(f[p^1][j],f[p][j-1]);
				g[p][j]=0;
				if (f[p][j]==f[p^1][j]) g[p][j]+=g[p^1][j],g[p][j]%=MOD;
				if (f[p][j]==f[p][j-1]) g[p][j]+=g[p][j-1],g[p][j]%=MOD;
				if (f[p][j]==f[p^1][j-1]) g[p][j]-=g[p^1][j-1],g[p][j]=(g[p][j]+MOD)%MOD;
			}
	}
	printf("%d\n%d\n",f[p][m],g[p][m]);
	return 0;
}
