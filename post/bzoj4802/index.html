bzoj4802题解

Apr 24, 2017 10:02 · 164 words · 1 minute read 题解 bzoj 数学 miller_rabin pollard_rho

这么大的数$O(\sqrt{n})$的算法肯定过不了，花了点时间研究了一下pollard_rho算法，感觉玄学无比。

这个算法基于这样一个事实，反复从$[1,n-1]$中取出两个数$a,b$，$|a-b|$与$n$不互质的概率是很大的。这之中的原理跟生日攻击类似。

这个算法还要更玄学一些，每次随机生成一个数t，定义函数$f(x)=x^2+t$。一开始$a=t,b=f(t)$，接着每次让$a=f(a),b=f(f(b))$。这么做有可能出现循环，如果出现，就随机再找一个t。用floyd判圈法可以快速判断循环，注意到$b$的前进速度是$a$的两倍，什么时候$b$赶上$a$了，就循环了。

分解质因数的时候，先用miller_rabin算法判断是不是质数，如果不是，就用pollard_rho来分解。

代码：

#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <set>
#define ele long long
using namespace std;
inline ele mul(ele a,ele b,ele p){
	ele ans=0,tmp=a;
	for (; b; b>>=1,(tmp<<=1)%=p)
		if (b&1) (ans+=tmp)%=p;
	return ans;
}
inline ele ABS(ele x){ return (x<0)?-x:x; }
ele gcd(ele a,ele b){
	return (!b)?a:gcd(b,a%b);
}
inline ele pw(ele a,ele x,ele p){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=mul(tmp,tmp,p))
		if (x&1) ans=mul(ans,tmp,p);
	return ans;
}
inline bool miller_rabin(ele n){
	if (n==1) return false;
	ele T=4;
	while (T--){
		ele a=rand()%(n-1)+1;
		if (pw(a,n-1,n)!=1) return false;
		ele p=n-1;
		while (p%2==0){
			p>>=1;
			ele t=pw(a,p,n);
			if (t==n-1) break;;
			if (t!=1) return false;
		}
	}
	return true;
}
inline ele pollard_rho(ele n){
	ele t=rand()%100,p=1,a=t,b=(mul(a,a,n)+t)%n;
	while (p==1 && a!=b){
		p=gcd(ABS(a-b),n);
		a=(mul(a,a,n)+t)%n;
		b=(mul(b,b,n)+t)%n;
		b=(mul(b,b,n)+t)%n;
	}
	return p;
}
ele n,ans;
set<ele> S;
void calc(ele n){
	if (n==1) return;
	if (miller_rabin(n)){
		if (!S.count(n)){
			S.insert(n);
			ans/=n; ans*=(n-1);
		}
		return;
	}
	ele p=1;
	while (p==1 || p==n) p=pollard_rho(n);
	calc(p); calc(n/p);
}
int main(){
	scanf("%lld",&n);
	ans=n;
	calc(n);
	printf("%lld\n",ans);
	return 0;
}
