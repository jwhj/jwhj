bzoj1009题解

Dec 18, 2016 10:02 · 151 words · 1 minute read 题解 bzoj 矩阵 DP kmp

这里有一篇写得比较好的题解。

可以考虑出一道题，有多个不吉利的串，那样应该就要用AC自动机来做了。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxm 25
struct matrix{
	inline void I(){
		memset(a,0,sizeof(a));
		for (int i=0; i<n; ++i)
			a[i][i]=1;
	}
	ele n,m,a[maxm][maxm];
};
ele n,m,MOD,p[maxm];
char s[maxm];
matrix f,C;
inline matrix operator*(matrix a,matrix b){
	matrix c;
	c.n=a.n; c.m=b.m;
	for (int i=0; i<c.n; ++i)
		for (int j=0; j<c.m; ++j){
			c.a[i][j]=0;
			for (int k=0; k<a.m; ++k)
				(c.a[i][j]+=a.a[i][k]*b.a[k][j])%=MOD;
		}
	return c;
}
inline matrix pow(matrix a,ele x){
	matrix ans,tmp=a;
	ans.n=ans.m=a.n;
	ans.I();
	for (; x; x>>=1,tmp=tmp*tmp)
		if (x&1)
			ans=ans*tmp;
	return ans;
}
int main(){
	scanf("%d%d%d%s",&n,&m,&MOD,s);
	p[0]=0; p[1]=0;
	for (int i=2; i<=m; ++i){
		int j=p[i-1];
		while (j && s[i-1]!=s[j])
			j=p[j];
		if (s[i-1]==s[j])
			++j;
		p[i]=j;
	}
	f.n=m; f.m=1;
	memset(f.a,0,sizeof(f.a));
	f.a[0][0]=1;
	C.n=C.m=m;
	memset(C.a,0,sizeof(C.a));
	for (int i=0; i<m; ++i)
		for (int j=0; j<10; ++j){
			int k=i;
			while (k && s[k]!=j+'0')
				k=p[k];
			if (s[k]==j+'0')
				++k;
			(C.a[k][i]+=1)%=MOD;
		}
	C=pow(C,n);
	f=C*f;
	ele ans=0;
	for (int i=0; i<m; ++i)
		(ans+=f.a[i][0])%=MOD;
	printf("%d\n",ans);
	return 0;
}
