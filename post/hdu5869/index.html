hdu5869题解

Jan 27, 2017 10:02 · 218 words · 2 minutes read 题解 hdu 树状数组 二分

首先，固定右端点的话，不同的gcd至多有$\log\mathrm{max}\{a_i\}$个。这是因为，不断左移左端点的时候，只要gcd变化了，一定是原来的因数，所以至少会除以2。

预处理出一个类似st表的东西，就可以在$O(\log n)$的时间里求出一个区间的gcd。这样通过二分就可以求出固定右端点之后所有的gcd。

这道题可以使用类似于SPOJ DQUERY的一个思路：把询问按右端点排序，把序列中的数一个个加入，如果出现重复的数，保留最靠右的一个。加入第k个数之后回答所有右端点为k的询问。这道题也是一样，出现重复的gcd则保留最靠右的一个。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
#define maxa 1000010
#define maxk 18
struct ASK{
	ele l,r,id;
	inline bool operator<(ASK b)const{
		return r<b.r;
	}
};
struct BIT{
	inline void alloc(ele n){
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele k){
		for (int j=i; j<=n; j+=j&(-j))
			a[j]+=k;
	}
	inline ele q(ele i){
		ele ans=0;
		for (int j=i; j; j-=j&(-j))
			ans+=a[j];
		return ans;
	}
	inline ele q(ele l,ele r){
		if (l>r) return 0;
		return q(r)-q(l-1);
	}
	ele n,a[maxn];
};
ele n,Q,a[maxn],f[maxn][maxk],p[maxa],res[maxn];
ASK q[maxn];
BIT b;
ele gcd(ele a,ele b){
	if (!b) return a;
	return gcd(b,a%b);
}
inline ele qgcd(ele l,ele r){
	ele L=r-l+1,ans=0;
	for (int i=0; L; ++i,L>>=1)
		if (L&1)
			ans=gcd(ans,f[l][i]),l+=1<<i;
	return ans;
}
inline ele nxt(ele i,ele j){
	ele l=0,r=i,tmp=qgcd(i,j);
	while (r-l>1){
		ele mid=(l+r)>>1;
		if (qgcd(mid,j)<tmp)
			l=mid;
		else r=mid;
	}
	if (qgcd(l,j)==tmp)
		return -1;
	else return l;
}
int main(){
	while (~scanf("%d%d",&n,&Q)){
		for (int i=0; i<n; ++i)
			scanf("%d",a+i),f[i][0]=a[i];
		for (int i=1; i<maxk; ++i)
			for (int j=0; j<n; ++j)
				f[j][i]=gcd(f[j][i-1],f[j+(1<<i-1)][i-1]);
		for (int i=0; i<Q; ++i)
			scanf("%d%d",&q[i].l,&q[i].r),--q[i].l,--q[i].r,q[i].id=i;
		sort(q,q+Q);
		memset(p,-1,sizeof(p));
		ele now=0;
		b.alloc(n);
		for (int i=0; i<Q; ++i){
			for (int j=now; j<=q[i].r; ++j)
				for (int k=j; k>=0; k=nxt(k,j)){
					ele tmp=qgcd(k,j);
					if (p[tmp]!=-1)
						b.add(p[tmp]+1,-1);
					p[tmp]=k;
					b.add(k+1,1);
				}
			now=q[i].r+1;
			res[q[i].id]=b.q(q[i].l+1,q[i].r+1);
		}
		for (int i=0; i<Q; ++i)
			printf("%d\n",res[i]);
	}
	return 0;
}
