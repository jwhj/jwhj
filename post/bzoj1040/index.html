bzoj1040题解

Aug 1, 2017 10:02 · 180 words · 1 minute read 题解 bzoj dp tarjan

虽然题目中给出的厌恶的关系是有向的，但事实上可以将其看成是无向的。

对于每一个联通块，若里面有m个点，由题意则有m条边，那么这个联通块一定是一个只含一个环的仙人掌，用tarjan把这个环找出来然后dp就可以了。要注意处理2元环的情况。

太久没搞OI现在tarjan都能打得这么丑了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 1000010
struct edge{
	ele v;
	edge *nxt,*rev;
}ep[maxn<<1],*ecnt;
ele n,top,scnt,tcnt,a[maxn],stk[maxn],f[maxn],g[maxn],seq[maxn],dfn[maxn],low[maxn];
bool flag[maxn],instack[maxn],vis[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u]; p->rev=ep+((ecnt-ep-1)^1);
	h[u]=p;
}
void tarjan(ele p,ele i){
	stk[top++]=i;
	instack[i]=true;
	vis[i]=true;
	dfn[i]=low[i]=tcnt++;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!~dfn[j->v]){
			tarjan(i,j->v);
			low[i]=min(low[i],low[j->v]);
			if (dfn[i]==low[j->v]){
				do{
					flag[stk[top-1]]=true;
					seq[scnt++]=stk[--top];
				}while (stk[top]!=j->v);
				flag[i]=true;
				seq[scnt++]=i;
			}
			else if (dfn[i]<low[j->v]) --top;
		}
		else if (j->v!=p) low[i]=min(low[i],dfn[j->v]);
}
void dfs(ele p,ele i){
	f[i]=a[i]; g[i]=0;
	vis[i]=false;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!flag[j->v] && vis[j->v]){
			dfs(i,j->v);
			f[i]+=g[j->v];
			g[i]+=max(f[j->v],g[j->v]);
		}
}
int main(){
	scanf("%lld",&n);
	ecnt=ep;
	memset(h,0,sizeof(h));
	for (int i=0; i<n; ++i){
		ele p;
		scanf("%lld%lld",a+i,&p);
		addedge(i,p-1); addedge(p-1,i);
	}
	memset(flag,0,sizeof(flag));
	memset(instack,0,sizeof(instack));
	memset(f,0,sizeof(f));
	memset(dfn,-1,sizeof(dfn));
	ele ans=0;
	tcnt=0;
	for (int i=0; i<n; ++i)
		if (!f[i]){
			scnt=0;
			top=0;
			memset(vis,0,sizeof(vis));
			tarjan(0,i);
			if (!scnt){
				scnt=1;
				seq[0]=i;
			}
			for (int j=0; j<scnt; ++j) dfs(-1,seq[j]);
			static ele h[maxn][2];
			h[0][0]=g[seq[0]]; h[0][1]=0;
			for (int j=1; j<scnt; ++j){
				h[j][0]=max(h[j-1][0],h[j-1][1])+g[seq[j]];
				h[j][1]=h[j-1][0]+f[seq[j]];
			}
			ele tmp=max(h[scnt-1][0],h[scnt-1][1]);
			h[0][0]=0; h[0][1]=f[seq[0]];
			for (int j=1; j<scnt; ++j){
				h[j][0]=max(h[j-1][0],h[j-1][1])+g[seq[j]];
				h[j][1]=h[j-1][0]+f[seq[j]];
			}
			tmp=max(tmp,h[scnt-1][scnt==1]);
			ans+=tmp;
		}
	printf("%lld\n",ans);
	return 0;
}
