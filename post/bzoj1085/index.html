bzoj1085题解

Aug 13, 2017 10:02 · 149 words · 1 minute read 题解 bzoj A*

IDA*的模板题，把跟目标状态不同的格子数量作为估价函数。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
const ele di[]={-1,-1,1,1,-2,-2,2,2};
const ele dj[]={-2,2,-2,2,-1,1,-1,1};
const ele n=5;
char s[n];
ele x,y,a[n][n],b[n][n];
inline ele est(){
	ele ans=0;
	for (int i=0; i<n; ++i)
		for (int j=0; j<n; ++j) ans+=(a[i][j]!=b[i][j]);
	return ans;
}
bool dfs(ele d,ele md){
	if (d==md) return !est();
	for (int r=0; r<8; ++r){
		ele ni=x+di[r],nj=y+dj[r];
		if (0<=ni && ni<n && 0<=nj && nj<n){
			swap(a[ni][nj],a[x][y]); swap(x,ni); swap(y,nj);
			if (est()+d<=md) if (dfs(d+1,md)) return true;
			swap(a[ni][nj],a[x][y]); swap(x,ni); swap(y,nj);
		}
	}
	return false;
}
int main(){
	for (int i=0; i<n; ++i){
		for (int j=0; j<i; ++j) b[i][j]=1;
		for (int j=i; j<n; ++j) b[i][j]=-1;
	}
	b[2][2]=0; b[3][3]=1; b[4][4]=1;
	ele T;
	scanf("%d",&T);
	while (T--){
		for (int i=0; i<n; ++i){
			scanf("%s",s);
			for (int j=0; j<n; ++j)
				if (s[j]=='1') a[i][j]=-1;
				else if (s[j]=='0') a[i][j]=1;
				else a[i][j]=0,x=i,y=j;
		}
		ele ans;
		for (ans=0; ans<=15; ++ans)
			if (dfs(0,ans)) break;
		printf("%d\n",(ans<=15)?ans:-1);
	}
	return 0;
}
