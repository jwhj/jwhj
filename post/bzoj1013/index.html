bzoj1013题解

Nov 14, 2016 10:02 · 107 words · 1 minute read 题解 bzoj 高斯消元

把方程列出来消掉二次项再高斯消元就可以了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define real double
using namespace std;
#define maxn 20
ele n;
real p[maxn][maxn],a[maxn][maxn],b[maxn],x[maxn];
int main(){
	scanf("%d",&n);
	for (int i=0; i<n+1; ++i)
		for (int j=0; j<n; ++j)
			scanf("%lf",&p[i][j]);
	for (int i=0; i<n; ++i){
		for (int j=0; j<n; ++j)
			a[i][j]=2*(p[i][j]-p[i+1][j]);
		b[i]=0;
		for (int j=0; j<n; ++j)
			b[i]+=p[i][j]*p[i][j]-p[i+1][j]*p[i+1][j];
	}
	for (int i=0; i<n; ++i)
		for (int j=i+1; j<n; ++j){
			real tmp=a[i][i]/a[j][i];
			for (int k=i; k<n; ++k){
				a[i][k]-=a[j][k]*tmp;
				swap(a[i][k],a[j][k]);
			}
			b[i]-=b[j]*tmp;
			swap(b[i],b[j]);
		}
	x[n-1]=b[n-1]/a[n-1][n-1];
	for (int i=n-2; i>=0; --i){
		real tmp=0;
		for (int j=n-1; j>i; --j)
			tmp+=a[i][j]*x[j];
		x[i]=(b[i]-tmp)/a[i][i];
	}
	for (int i=0; i<n-1; ++i)
		printf("%.3lf ",x[i]);
	printf("%.3lf\n",x[n-1]);
	return 0;
}
