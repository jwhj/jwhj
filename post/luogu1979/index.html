luogu1979题解

Nov 13, 2016 10:02 · 286 words · 2 minutes read 题解 luogu spfa

贴个代码题解晚点再写吧……头疼得连堆优化的dijkstra都不想打了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
#define maxn 40
#define maxnode 6400
#define INF 1000000000
const ele di[]={-1,1,0,0};
const ele dj[]={0,0,-1,1};
struct qn{
	ele x,y,d;
};
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxnode<<2];
ele n,m,q,ecnt;
bool map[maxn][maxn];
edge *g[maxnode];
inline bool valid(ele i,ele j){
	return 0<=i && i<n && 0<=j && j<m && map[i][j];
}
inline bool valid(ele i,ele j,ele d){
	ele ni=i+di[d],nj=j+dj[d];
	return valid(i,j) && valid(ni,nj);
}
inline ele bfs(ele x1,ele y1,ele x2,ele y2){
	if (x1==x2 && y1==y2) return 0;
	static qn Q[maxn*maxn];
	static bool vis[maxn][maxn];
	Q[0]={x1,y1,0};
	memset(vis,0,sizeof(vis));
	ele h=-1,t=0;
	while (h<t){
		++h;
		for (int w=0; w<4; ++w){
			ele ni=Q[h].x+di[w],nj=Q[h].y+dj[w];
			if (valid(ni,nj) && !vis[ni][nj]){
				vis[ni][nj]=true;
				++t;
				Q[t]={ni,nj,Q[h].d+1};
				if (ni==x2 && nj==y2)
					return Q[t].d;
			}
		}
	}
	return INF;
}
inline ele idx(ele i,ele j,ele d){
	return (i*m+j<<2)+d;
}
inline void addedge(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->w=w; p->nxt=g[u];
	g[u]=p;
}
int main(){
	scanf("%d%d%d",&n,&m,&q);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			scanf("%d",&map[i][j]);
	ecnt=0;
	memset(g,0,sizeof(g));
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j)
			for (int d=0; d<4; ++d)
				if (valid(i,j,d)){
					ele ni=i+di[d],nj=j+dj[d];
					addedge(idx(i,j,d),idx(ni,nj,d^1),1);
					for (int w=0; w<4; ++w)
						if (d!=w && valid(i,j,w)){
							ele mi=i+di[w],mj=j+dj[w];
							map[i][j]=false;
							ele tmp=bfs(ni,nj,mi,mj);
							map[i][j]=true;
							if (tmp<INF)
								addedge(idx(i,j,d),idx(i,j,w),tmp);
						}
				}
	while (q--){
		ele ex,ey,sx,sy,tx,ty;
		scanf("%d%d%d%d%d%d",&ex,&ey,&sx,&sy,&tx,&ty);
		--ex,--ey,--sx,--sy,--tx,--ty;
		if (sx==tx && sy==ty){
			puts("0");
			continue;
		}
		ele ans=INF;
		for (int d=0; d<4; ++d)
			if (valid(sx,sy,d)){
				ele ni=sx+di[d],nj=sy+dj[d];
				map[sx][sy]=false;
				ele t1=bfs(ex,ey,ni,nj);
				map[sx][sy]=true;
				static ele dis[maxnode];
				for (int i=0; i<=idx(n-1,m-1,3); ++i)
					dis[i]=INF;
				ele s=idx(sx,sy,d);
				dis[s]=0;
				static queue<ele> Q;
				Q.push(s);
				while (!Q.empty()){
					ele k=Q.front(); Q.pop();
					for (edge *j=g[k]; j; j=j->nxt)
						if (dis[k]+j->w<dis[j->v]){
							dis[j->v]=dis[k]+j->w;
							Q.push(j->v);
						}
				}
				for (int w=0; w<4; ++w)
					if (valid(tx,ty,w))
						ans=min(ans,t1+dis[idx(tx,ty,w)]);
			}
		if (ans<INF)
			printf("%d\n",ans);
		else puts("-1");
	}
	return 0;
}
