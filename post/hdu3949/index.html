hdu3949题解

May 27, 2017 10:02 · 117 words · 1 minute read 题解 hdu 线性基

线性基的介绍可以看这里。

多组数据忘了清零……我真是越来越sb了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 10010
#define W 64
ele n,Q,m,a[maxn],b[W],c[W];
int main(){
	ele T;
	scanf("%lld",&T);
	for (int tc=0; tc<T; ++tc){
		printf("Case #%d:\n",tc+1);
		scanf("%lld",&n);
		for (int i=0; i<n; ++i) scanf("%lld",a+i);
		memset(b,0,sizeof(b));
		for (int i=0; i<n; ++i)
			for (int j=W-1; ~j; --j)
				if (a[i]>>j&1){
					if (b[j]) a[i]^=b[j];
					else{
						b[j]=a[i];
						for (int k=j-1; ~k; --k)
							if (b[j]>>k&1) b[j]^=b[k];
						for (int k=j+1; k<W; ++k)
							if (b[k]>>j&1) b[k]^=b[j];
						break;
					}
				}
		m=0;
		for (int i=0; i<W; ++i)
			if (b[i]) c[m++]=b[i];
		scanf("%lld",&Q);
		while (Q--){
			ele k;
			scanf("%lld",&k);
			k-=(m<n);
			if (k>=(1LL<<m)){ puts("-1"); continue; }
			ele ans=0;
			for (int i=0; i<m; ++i){
				ans^=c[i]*(k&1);
				k>>=1;
			}
			printf("%lld\n",ans);
		}
	}
	return 0;
}
