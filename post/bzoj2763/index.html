bzoj2763题解

Nov 3, 2016 10:02 · 145 words · 1 minute read 题解 bzoj 图论 分层图 dijkstra 优先队列

考虑把图拆成k+1层，第i层表示已经免费i次。

对于每条原来的边，每层内照常，层间连一条权值为0的边，表示使用了一次免费机会。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 10010
#define maxm 50010
#define maxk 12
#define INF 1000000000
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxm*4*maxk];
struct hn{
	ele i,d;
	inline bool operator<(hn b)const{
		return d>b.d;
	}
};
ele n,m,k,s,t,ecnt,d[maxn*maxk];
bool vis[maxn*maxk];
priority_queue<hn> q;
edge *h[maxn*maxk];
inline void addedge(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
inline void dij(ele n,ele s){
	memset(vis,0,sizeof(vis));
	for (int i=0; i<n; ++i)
		d[i]=INF;
	d[s]=0;
	q.push({s,0});
	while (!q.empty()){
		hn k=q.top(); q.pop();
		if (vis[k.i]) continue;
		vis[k.i]=true;
		for (edge *j=h[k.i]; j; j=j->nxt)
			if (d[j->v]>k.d+j->w){
				d[j->v]=k.d+j->w;
				q.push({j->v,d[j->v]});
			}
	}
}
int main(){
	scanf("%d%d%d%d%d",&n,&m,&k,&s,&t);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<m; ++i){
		ele u,v,w;
		scanf("%d%d%d",&u,&v,&w);
		for (int j=0; j<k; ++j)
			addedge(j*n+u,(j+1)*n+v,0),
			addedge(j*n+v,(j+1)*n+u,0);
		for (int j=0; j<=k; ++j)
			addedge(j*n+u,j*n+v,w),
			addedge(j*n+v,j*n+u,w);
	}
	dij(n*(k+1),s);
	ele ans=INF;
	for (int i=0; i<=k; ++i)
		ans=min(ans,d[i*n+t]);
	printf("%d\n",ans);
	return 0;
}
