bzoj1082题解

Sep 11, 2017 10:02 · 154 words · 1 minute read 题解 bzoj 贪心 乱搞 二分

怎么评价这道题好呢……

首先一个显然的想法是，如果某个最优方案得到k块木板，那么一定有一种最优方案的到的是最小的k块木板。于是我们可以二分答案k再想办法判断可行性。

判断可行性，我一开始的想法是网络流，结果建图建了半天发现是错的，感觉并不可做。后来看了一眼题解发现是爆搜+剪枝？？？

我搜索的思路是，对于每块提供的木材，枚举它被分成哪些需要的木材，然后用以下的一些思路来优化：

  • 提供的木材从小到大枚举，需要的木材从大到小枚举
  • 每块提供的木材被用到不能再分出一块需要的木材时，才枚举下一块
  • 统计当前浪费的木材长度，如果超过提供木材总量减去所需木材长度，直接剪枝
  • 直接忽略大于提供木材最大长度的所需木材

最大一组数据开了O2在本机还是跑了1.74s，在bzoj上勉强A了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 1010
#define maxm 55
ele n,m,p,wst,mxw,a[maxn],b[maxm],c[maxm];
bool flag[maxn];
bool dfs(ele x,ele i){
	if (wst>mxw) return false;
	if (!p) return true;
	ele tmp=c[i];
	bool f=false,f1=true;
	for (--c[i]; ~c[i] && !f; --c[i])
		if (!flag[c[i]] && b[i]>=a[c[i]]){
			f1=false;
			flag[c[i]]=true;
			b[i]-=a[c[i]];
			--p;
			if (dfs(x,i)) f=true;
			flag[c[i]]=false;
			b[i]+=a[c[i]];
			++p;
		}
	if (f1){
		wst+=b[i];
		if (dfs(x,i+1)) f=true;
		wst-=b[i];
	}
	c[i]=tmp;
	return f;
}
inline bool test(ele x){
	p=x; memset(flag,0,sizeof(bool)*x);
	for (int i=0; i<m; ++i) c[i]=x;
	mxw=wst=0;
	for (int i=0; i<m; ++i) mxw+=b[i];
	for (int i=0; i<x; ++i) mxw-=a[i];
	if (mxw<0) return false;
	return dfs(x,0);
}
int main(){
	scanf("%d",&m);
	ele mx=0;
	for (int i=0; i<m; ++i) scanf("%d",b+i),mx=max(mx,b[i]);
	sort(b,b+m);
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%d",a+i);
	sort(a,a+n);
	while (n && a[n-1]>mx) --n;
	ele L=0,R=n+1;
	while (R-L>1){
		ele mid=(L+R)>>1;
		if (test(mid)) L=mid; else R=mid;
	}
	printf("%d\n",L);
	return 0;
}
