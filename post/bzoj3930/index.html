bzoj3930题解

May 16, 2016 10:02 · 117 words · 1 minute read 题解 bzoj 数学 mobius

问题可以转化为，在$[\lceil\frac{l}{k}\rceil,\lfloor\frac{h}{k}\rfloor]$间取n个整数，他们互质。

容易推得$ans=\sum(\lfloor\frac{h}{dk}\rfloor-\lceil\frac{l}{dk}\rceil+1)^n\mu(d)$。

然而这么做是$O(n)$的。

先膜popoqq的做法。

我的做法是这样的：

当取出来的整数不全相等时，上面那条式子要改成$\sum[(\lfloor\frac{h}{dk}\rfloor-\lceil\frac{l}{dk}\rceil+1)^n-(\lfloor\frac{h}{dk}\rfloor-\lceil\frac{l}{dk}\rceil+1)]\mu(d)$。

因为$(x,y)\le|x-y|$，所以这里d的值不超过$10^5$，直接做就可以了。

如果取出来的数全都一样，那只能全是1。如果$\lceil\frac{l}{dk}\rceil\le 1$，那么就可以全部取1，最后的答案还要加上1。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define MOD 1000000007
#define maxn 100010
ele n,k,l,h,pcnt,plist[maxn],mu[maxn];
bool flag[maxn];
inline ele pow(ele a,ele x){
	ele ans=1,tmp=a;
	for (; x>0; x>>=1,tmp=tmp*tmp%MOD)
		if (x&1) ans=ans*tmp%MOD;
	return ans;
}
int main(){
	scanf("%lld%lld%lld%lld",&n,&k,&l,&h);
	if (l%k) l=l/k+1; else l/=k;
	h/=k;
	ele tmp=h-l+10;
	memset(flag,0,sizeof(flag));
	mu[1]=1;
	pcnt=0;
	for (int i=2; i<tmp; ++i){
		if (!flag[i]) plist[pcnt++]=i,mu[i]=-1;
		for (int j=0; j<pcnt && i*plist[j]<tmp; ++j){
			flag[i*plist[j]]=true;
			if (i%plist[j])
				mu[i*plist[j]]=-mu[i];
			else{
				mu[i*plist[j]]=0;
				break;
			}
		}
	}
	ele ans=0;
	for (int d=1; d<=h && d<=h-l; ++d){
		ele L=l/d,R=h/d;
		if (l%d) ++L;
		ans+=(pow(R-L+1,n)-(R-L+1))*mu[d];
		ans%=MOD;
	}
	if (l==1) ++ans;
	ans+=MOD;
	ans%=MOD;
	printf("%lld\n",ans);
	return 0;
}
