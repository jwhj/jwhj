bzoj1452题解

Nov 15, 2016 10:02 · 128 words · 1 minute read 题解 bzoj 树状数组

为每种权值开一个二位树状数组记录就可以了。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 310
#define maxc 110
struct BIT{
	inline void alloc(ele n,ele m){
		this->n=n; this->m=m;
		memset(a,0,sizeof(a));
	}
	inline void add(ele x,ele y,ele k){
		for (int i=x; i<=n; i+=i&(-i))
			for (int j=y; j<=m; j+=j&(-j))
				a[i][j]+=k;
	}
	inline ele query(ele x,ele y){
		ele ans=0;
		for (int i=x; i>0; i-=i&(-i))
			for (int j=y; j>0; j-=j&(-j))
				ans+=a[i][j];
		return ans;
	}
	inline ele query(ele x1,ele y1,ele x2,ele y2){
		return query(x2,y2)-query(x1-1,y2)-query(x2,y1-1)+query(x1-1,y1-1);
	}
	ele n,m,a[maxn][maxn];
};
ele n,m,q,a[maxn][maxn];
BIT b[maxc];
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0; i<maxc; ++i)
		b[i].alloc(n,m);
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j){
			scanf("%d",&a[i][j]);
			b[a[i][j]].add(i+1,j+1,1);
		}
	scanf("%d",&q);
	while (q--){
		ele op,x1,x2,y1,y2,c;
		scanf("%d",&op);
		if (op==1){
			scanf("%d%d%d",&x1,&y1,&c);
			b[a[x1-1][y1-1]].add(x1,y1,-1);
			a[x1-1][y1-1]=c;
			b[c].add(x1,y1,1);
		}
		else{
			scanf("%d%d%d%d%d",&x1,&x2,&y1,&y2,&c);
			printf("%d\n",b[c].query(x1,y1,x2,y2));
		}
	}
	return 0;
}
