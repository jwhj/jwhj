bzoj1095题解

Nov 6, 2016 10:02 · 385 words · 2 minutes read 题解 bzoj 树分治 优先队列

先考虑静态的情况，显然点剖一下就做完了。

动态也是一个道理，但是要用优先队列来维护信息，每个点维护两个堆：

  • 以该点为重心的分治结构中所有关灯的点到上一层分治结构重心的距离
  • 以该点为重心的分治结构中所有子树的堆顶，如果该点关灯，还有一个0

用一个全局的堆维护每个点的第二个堆的最大值+次大值，那么全局堆的堆顶就是答案。

修改自己想吧，知道要维护什么堆应该不难想……

空间复杂度分析类似树套树，可以得到是$O(n\log n)$的。

发现一个实现删除任意已知元素的优先队列的实现方法，详情看代码。

写了句char s[10],u;调了几个小时……惨啊

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
#define maxn 100010
#define maxk 17
struct heap{
	priority_queue<ele> a,b;
	inline void push(ele x){
		a.push(x);
	}
	inline void del(ele x){
		b.push(x);
	}
	inline ele top(){
		while (!a.empty() && !b.empty() && a.top()==b.top())
			a.pop(),b.pop();
		return a.top();
	}
	inline ele sec(){
		ele tmp=top(); del(tmp);
		ele ans=top(); push(tmp);
		return ans;
	}
	inline ele size(){
		return a.size()-b.size();
	}
};
struct data{
	heap h1,h2;
};
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<1];
ele n,m,cnt,ecnt,size[maxn],w[maxn],pv[maxn],dep[maxn],fa[maxn][maxk];
bool flag[maxn],c[maxn];
edge *h[maxn];
data a[maxn];
heap gh;
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void dfs0(ele p,ele i){
	fa[i][0]=p;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p)
			dep[j->v]=dep[i]+1,dfs0(i,j->v);
}
ele dfs1(ele p,ele i,ele s){
	size[i]=1;
	w[i]=0;
	ele k=-1;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v]){
			ele tmp=dfs1(i,j->v,s);
			size[i]+=size[j->v];
			w[i]=max(w[i],size[j->v]);
			if (k==-1 || w[tmp]<w[k])
				k=tmp;
		}
	w[i]=max(w[i],s-size[i]);
	if (k==-1 || w[i]<w[k])
		k=i;
	return k;
}
ele get_size(ele p,ele i){
	ele ans=1;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v])
			ans+=get_size(i,j->v);
	return ans;
}
void dfs2(ele p,ele i,ele d,heap&H){
	H.push(d);
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p && !flag[j->v])
			dfs2(i,j->v,d+1,H);
}
inline void maintain(heap&H,ele k){
	if (H.size()>=2){
		ele tmp=H.top()+H.sec();
		if (k>0) gh.push(tmp); else gh.del(tmp);
	}
}
ele build(ele p,ele i){
	ele s=get_size(-1,i);
	ele k=dfs1(-1,i,s);
	dfs2(-1,i,1,a[k].h1);
	flag[k]=true;
	pv[k]=p;
	for (edge *j=h[k]; j; j=j->nxt)
		if (!flag[j->v]){
			ele q=build(k,j->v);
			a[k].h2.push(a[q].h1.top());
		}
	a[k].h2.push(0);
	maintain(a[k].h2,1);
	return k;
}
inline ele lca(ele u,ele v){
	if (dep[u]>dep[v]) swap(u,v);
	ele k=dep[v]-dep[u];
	for (int i=0; k; ++i,k>>=1)
		if (k&1)
			v=fa[v][i];
	if (u==v) return u;
	for (int i=maxk-1; i>=0; --i)
		if (fa[u][i]!=fa[v][i])
			u=fa[u][i],v=fa[v][i];
	return fa[u][0];
}
inline ele dis(ele u,ele v){
	ele w=lca(u,v);
	return dep[u]+dep[v]-2*dep[w];
}
inline void upd(ele x){
	ele i=x;
	maintain(a[x].h2,-1);
	if (c[x])
		a[x].h2.push(0);
	else a[x].h2.del(0);
	maintain(a[x].h2,1);
	while (pv[i]!=-1){
		maintain(a[pv[i]].h2,-1);
		if (a[i].h1.size())
			a[pv[i]].h2.del(a[i].h1.top());
		if (c[x])
			a[i].h1.push(dis(pv[i],x));
		else a[i].h1.del(dis(pv[i],x));
		if (a[i].h1.size())
			a[pv[i]].h2.push(a[i].h1.top());
		maintain(a[pv[i]].h2,1);
		i=pv[i];
	}
	if (c[x]) ++cnt; else --cnt;
	c[x]=!c[x];
}
int main(){
	scanf("%d",&n);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v);
		--u,--v;
		addedge(u,v); addedge(v,u);
	}
	dep[0]=0;
	dfs0(-1,0);
	for (int i=1; i<maxk; ++i)
		for (int j=0; j<n; ++j)
			if (fa[j][i-1]==-1)
				fa[j][i]=-1;
			else fa[j][i]=fa[fa[j][i-1]][i-1];
	memset(c,0,sizeof(c));
	build(-1,0);
	scanf("%d",&m);
	cnt=n;
	while (m--){
		char s[10];
		ele u;
		scanf("%s",s);
		if (s[0]=='C'){
			scanf("%d",&u);
			--u;
			upd(u);
		}
		else
			if (cnt<=1)
				printf("%d\n",cnt-1);
			else printf("%d\n",gh.top());
	}
	return 0;
}
