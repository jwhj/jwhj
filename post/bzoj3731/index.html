bzoj3731题解

Jan 23, 2017 10:02 · 262 words · 2 minutes read 题解 bzoj 块状树

不考虑删边的话就是bzoj3720,在这里有题解。

如果要删边的话，其实就是要把删除的点从所在的块里面分离出来，因为块里的元素不多，直接暴力分离就可以了。

细节还是比较多的。考虑到邻接表删边不方便，需要保存一个数组，记录每个点到父节点的边有没有被删除；用一个数组保存每个块的根上面那个块的根。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#define ele int
using namespace std;
#define maxn 200010
#define B 500
struct edge{
    ele v;
    edge *nxt;
}ep[maxn*3];
ele n,q,ecnt,w[maxn],fa[maxn],fb[maxn],r[maxn],s[maxn];
bool flag[maxn];
edge *h[maxn],*h1[maxn];
vector<ele> lst[maxn];
inline void addedge(edge *h[],ele u,ele v){
    edge *p=&ep[ecnt++];
    p->v=v; p->nxt=h[u];
    h[u]=p;
}
inline void insert(ele i){
    ele j=r[i];
    lst[j].push_back(w[i]);
    for (int p=lst[j].size()-1; p && lst[j][p-1]>lst[j][p]; --p)
        swap(lst[j][p-1],lst[j][p]);
}
inline void erase(ele i){
    ele j=r[i],p=lower_bound(lst[j].begin(),lst[j].end(),w[i])-lst[j].begin();
    for (; p<lst[j].size()-1; ++p)
        lst[j][p]=lst[j][p+1];
    lst[j].erase(lst[j].end()-1);
}
void dfs1(ele p,ele i){
    fa[i]=p;
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p){
            if (s[r[i]]==B){
                r[j->v]=j->v;
                addedge(h1,r[i],j->v);
                fb[j->v]=r[i];
            }
            else r[j->v]=r[i];
            ++s[r[j->v]];
            dfs1(i,j->v);
        }
    insert(i);
}
ele dfs2(ele i,ele x){
    ele ans=lst[i].end()-upper_bound(lst[i].begin(),lst[i].end(),x);
    for (edge *j=h1[i]; j; j=j->nxt)
        if (fb[j->v]==i)
            ans+=dfs2(j->v,x);
    return ans;
}
ele dfs3(ele p,ele i,ele x){
    if (i==r[i]) return dfs2(i,x);
    ele ans=(w[i]>x)?(1):(0);
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p && !flag[j->v])
            ans+=dfs3(i,j->v,x);
    return ans;
}
void dfs4(ele p,ele i,ele u,ele v){
    if (i==r[i]){
        fb[i]=u;
        if (u!=i) addedge(h1,u,i);
        return;
    }
    --s[r[v]];
    erase(i);
    r[i]=u;
    insert(i);
    ++s[r[u]];
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p && !flag[j->v])
            dfs4(i,j->v,u,v);
}
int main(){
    scanf("%d",&n);
    ecnt=0;
    memset(h,0,sizeof(h)); memset(h1,0,sizeof(h1));
    for (int i=0; i<n-1; ++i){
        ele u,v;
        scanf("%d%d",&u,&v);
        --u,--v;
        addedge(h,u,v); addedge(h,v,u);
    }
    for (int i=0; i<n; ++i)
        scanf("%d",w+i);
    r[0]=0; s[0]=1;
    dfs1(-1,0);
    memset(flag,0,sizeof(flag));
    scanf("%d",&q);
    ele ans=0;
    while (q--){
        ele op,u,x;
        scanf("%d%d",&op,&u);
        u^=ans;
        --u;
        if (op!=3)
            scanf("%d",&x),x^=ans;
        if (op==0){
            ans=dfs3(fa[u],u,x);
            printf("%d\n",ans);
        }
        else if (op==1){
            erase(u);
            w[u]=x;
            insert(u);
        }
        else if (op==2){
            addedge(h,n,u); addedge(h,u,n);
            fa[n]=u;
            if (s[r[u]]==B){
                r[n]=n;
                addedge(h1,r[u],n);
                fb[n]=r[u];
            }
            else r[n]=r[u];
            ++s[r[n]];
            w[n]=x;
            insert(n);
            ++n;
        }
        else if (op==3){
            flag[u]=true;
            dfs4(fa[u],u,u,r[u]);
        }
    }
    return 0;
}
