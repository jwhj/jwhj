bzoj2653题解

Aug 26, 2016 10:02 · 272 words · 2 minutes read 题解 bzoj 主席树 线段树 统计 二分

军训完第一道OI题。

感觉方法挺难想的……不过看了题解马上就懂了。

对于一个数x，置数列中所有比x小的数为-1，所有等于x的数为0，所有大于x的数为1。取$l \in [a,b]$,$r \in [c,d]$，区间[l,r]的和最大为k。如果k>=0，那么答案>=x,否则答案小于x。

于是就可以二分了。

但是有一个问题：如何把数列中的数改为-1,0和1？

考虑使用主席树，一开始所有数为1，把原数列中的数从小到大排序，每次加入一个数，就把前面的那个数改为-1或0。当原数列中最后一个x被插入时，数列中所有比x小的数为-1，所有等于x的数为0，所有大于x的数为1。

最大区间和的维护是一个简单的线段树问题。

代码（写得好丑啊）：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 20010
#define maxq 25010
struct ANS
{
	ele sum,lmax,rmax,max;
};
struct node
{
	ANS a;
	node *l,*r;
}pool[maxn<<6];
struct data
{
	ele x,id;
	inline bool operator<(data b)const
	{
		return x<b.x;
	}
};
ele n,q,ncnt,len,a[maxn],arr[maxn],pos[maxn];
node *t[maxn];
data d[maxn];
inline ANS operator+(ANS a,ANS b)
{
	ANS c;
	c.sum=a.sum+b.sum;
	c.lmax=max(a.lmax,a.sum+b.lmax);
	c.rmax=max(b.rmax,a.rmax+b.sum);
	c.max=max(a.rmax+b.lmax,max(a.max,b.max));
	return c;
}
inline void maintain(node *x)
{
	if (x->l) x->a=x->l->a+x->r->a;
}
node* build(ele l,ele r)
{
	node *p=&pool[ncnt++];
	if (l==r)
	{
		if (l==d[0].id) p->a.sum=0; else p->a.sum=1;
		p->a.lmax=p->a.rmax=p->a.max=p->a.sum;
		p->l=p->r=NULL;
	}
	else
	{
		ele mid=(l+r)>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		maintain(p);
	}
	return p;
}
node* upd(node *x,ele a,ele b,ele i,ele k)
{
	node *p=&pool[ncnt++];
	*p=*x;
	if (a==b)
		p->a.sum=p->a.lmax=p->a.rmax=p->a.max=k;
	else
	{
		ele mid=(a+b)>>1;
		if (i<=mid) p->l=upd(x->l,a,mid,i,k);
		else p->r=upd(x->r,mid+1,b,i,k);
		maintain(p);
	}
	return p;
}
ANS query(node *x,ele a,ele b,ele l,ele r)
{
	ANS ans;
	if (l>r)
		ans.sum=ans.lmax=ans.rmax=ans.max=0;
	else if (l<=a && b<=r)
		ans=x->a;
	else
	{
		ele mid=(a+b)>>1;
		if (r<=mid)
			ans=query(x->l,a,mid,l,r);
		else if (l>mid)
			ans=query(x->r,mid+1,b,l,r);
		else
		{
			ANS lans=query(x->l,a,mid,l,r);
			ANS rans=query(x->r,mid+1,b,l,r);
			ans=lans+rans;
		}
	}
	return ans;
}
inline ele idx(ele x)
{
	return lower_bound(arr,arr+len,x)-arr;
}
int main()
{
	scanf("%d",&n);
	for (int i=0; i<n; ++i)
		scanf("%d",a+i),arr[i]=a[i],d[i].id=i;
	sort(arr,arr+n);
	len=unique(arr,arr+n)-arr;
	for (int i=0; i<n; ++i) d[i].x=a[i]=idx(a[i]);
	sort(d,d+n);
	ncnt=0;
	t[0]=build(0,n-1);
	pos[0]=0;
	for (int i=1; i<n; ++i)
	{
		pos[d[i].x]=i;
		ele p;
		if (d[i].x==d[i-1].x) p=0; else p=-1;
		t[i]=upd(t[i-1],0,n-1,d[i-1].id,p);
	}
	scanf("%d",&q);
	ele lastans=0;
	while (q--)
	{
		ele a,b,c,d;
		scanf("%d%d%d%d",&a,&b,&c,&d);
		ele q[4]={(a+lastans)%n,(b+lastans)%n,(c+lastans)%n,(d+lastans)%n};
		sort(q,q+4);
		a=q[0],b=q[1],c=q[2],d=q[3];
		ele l=0,r=len;
		while (r-l>1)
		{
			ele mid=(l+r)>>1;
			ANS tmp1=query(t[pos[mid]],0,n-1,a,b);
			ANS tmp2=query(t[pos[mid]],0,n-1,b+1,c-1);
			ANS tmp3=query(t[pos[mid]],0,n-1,c,d);
			ele tmp=tmp1.rmax+tmp2.sum+tmp3.lmax;
			if (tmp>=0) l=mid; else r=mid;
		}
		printf("%d\n",lastans=arr[l]);
	}
	return 0;
}
