bzoj3563题解

Oct 21, 2016 10:02 · 186 words · 1 minute read 题解 bzoj 并查集

注意到，k也要异或之前连通的个数，但总共读进来多少条边，也就是k实际的取值是知道的，因此对于前q-1个询问，均可以把k与实际边数异或一下得到答案。

至于最后一个询问，用并查集求一下就可以了。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 100010
#define maxm 500010
#define maxq 50010
#define maxk 25
struct ASK{
	ele n,a[maxk];
};
struct edge{
	ele u,v;
};
ele getf(ele *u,ele x){
	return (x==u[x])?(x):(u[x]=getf(u,u[x]));
}
inline void uni(ele *u,ele x,ele y){
	ele a=getf(u,x),b=getf(u,y);
	if (a!=b)
		u[a]=b;
}
inline void getl(char *s){
	char c;
	ele i=0;
	while ((c=getchar())!='\n' && c!='\r')
		s[i++]=c;
	s[i++]=0;
}
ele n,m,q,u[maxn];
bool flag[maxm];
char s[1010];
edge e[maxm];
ASK a[maxq];
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0; i<m; ++i)
		scanf("%d%d",&e[i].u,&e[i].v),--e[i].u,--e[i].v;
	scanf("%d",&q);
	for (int i=0; i<q; ++i){
		getl(s);
		while (!s[0]) getl(s);
		ele len=strlen(s)+1,k=0;
		a[i].a[0]=0;
		for (int j=0; j<len; ++j)
			if (s[j]>='0' && s[j]<='9')
				a[i].a[k]*=10,a[i].a[k]+=s[j]-'0';
			else if (s[j-1]>='0' && s[j-1]<='9') ++k,a[i].a[k]=0;
		a[i].n=k;
	}
	ele tmp=0;
	for (int i=1; i<q; ++i){
		ele tmp1=a[i].a[0]^(a[i].n-1);
		if (tmp1-tmp)
			puts("Connected");
		else puts("Disconnected");
		tmp=tmp1;
		for (int j=1; j<a[i].n; ++j)
			a[i].a[j]^=tmp;
	}
	memset(flag,0,sizeof(flag));
	for (int i=1; i<a[q-1].n; ++i)
		flag[a[q-1].a[i]-1]=true;
	for (int i=0; i<n; ++i)
		u[i]=i;
	for (int i=0; i<m; ++i)
		if (!flag[i]) uni(u,e[i].u,e[i].v);
	memset(flag,0,sizeof(flag));
	for (int i=0; i<n; ++i)
		flag[getf(u,i)]=true;
	tmp=0;
	for (int i=0; i<n; ++i)
		if (flag[i])
			++tmp;
	if (tmp>1)
		puts("Disconnected");
	else puts("Connected");
	return 0;
}
