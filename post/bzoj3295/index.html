bzoj3295题解

Feb 1, 2017 10:02 · 217 words · 2 minutes read 题解 bzoj 分块 块状链表

网上题解似乎都是树套树&cdq分治。我感觉可以用块状链表水过去，于是就写了一下，没开long long结果WA了一次,然后就A了。

块的大小要设为$\sqrt{n\log n}$。这道题只有删除，所以可以不考虑块的合并/块状链表重构，把每$\sqrt{n}$次操作后重构的代码删了以后程序快了很多……

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
#define ll long long
using namespace std;
#define maxn 100010
#define maxs 1300
struct blk{
	ele n,a[maxs],b[maxs];
};
struct BIT{
	inline void init(ele n){
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele k){
		for (int j=i; j<=n; j+=j&(-j))
			a[j]+=k;
	}
	inline ele q(ele i){
		ele ans=0;
		for (int j=i; j>0; j-=j&(-j))
			ans+=a[j];
		return ans;
	}
	ele n,a[maxn];
};
ele n,m,bcnt,a[maxn],pos[maxn];
blk b[maxs];
BIT bit;
inline ele build(ele n,ele *a,blk *b){
	ele j=0,S=sqrt(n*log(n)/log(2));
	for (int i=0; i<n; i+=S,++j){
		b[j].n=min(n-i,S);
		for (int k=0; k<b[j].n; ++k){
			b[j].a[k]=b[j].b[k]=a[i+k];
			pos[a[i+k]]=j;
		}
		sort(b[j].b,b[j].b+b[j].n);
	}
	return j;
}
inline ele expt(ele bcnt,blk *b,ele *a){
	ele j=0;
	for (int i=0; i<bcnt; ++i)
		for (int k=0; k<b[i].n; ++k)
			a[j++]=b[i].a[k];
	return j;
}
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i)
		scanf("%d",a+i),--a[i];
	bcnt=build(n,a,b);
	ll ans=0;
	bit.init(n);
	for (int i=n-1; i>=0; --i){
		ans+=bit.q(a[i]);
		bit.add(a[i]+1,1);
	}
	for (int p=0; p<m; ++p){
		printf("%lld\n",ans);
		ele x;
		scanf("%d",&x);
		--x;
		ele i1=pos[x],j1=0;
		while (b[i1].a[j1]!=x)
			++j1;
		for (int i=0; i<i1; ++i){
			ele p=upper_bound(b[i].b,b[i].b+b[i].n,x)-b[i].b;
			ans-=b[i].n-p;
		}
		for (int j=0; j<j1; ++j)
			if (b[i1].a[j]>x)
				--ans;
		for (int i=i1+1; i<bcnt; ++i){
			ele p=lower_bound(b[i].b,b[i].b+b[i].n,x)-b[i].b;
			ans-=p;
		}
		for (int j=j1+1; j<b[i1].n; ++j)
			if (b[i1].a[j]<x)
				--ans;
		for (int j=j1; j<b[i1].n-1; ++j)
			b[i1].a[j]=b[i1].a[j+1];
		ele p1=lower_bound(b[i1].b,b[i1].b+b[i1].n,x)-b[i1].b;
		for (int j=p1; j<b[i1].n-1; ++j)
			b[i1].b[j]=b[i1].b[j+1];
		--b[i1].n;
	}
	return 0;
}
