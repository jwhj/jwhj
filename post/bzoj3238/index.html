bzoj3238题解

May 26, 2016 10:02 · 157 words · 1 minute read 题解 bzoj 字符串 后缀数据结构

网上题解大多数使用后缀数组来做，可是我的后缀数组写出来是$O(n\log^2 n)$的，肯定过不了。

观察以下题目要求的式子，$\sum_{1\le i\lt j\le n}len(T_i)+len(T_j)=\frac{n(n-1)(n+1)}{2}$。

对于后缀树上每一个节点，假设它的长度为L，那么在它的两个不同子树中选出两个节点，这两个节点对应的后缀的lcp为L。所以要求剩下的那一项，可以在后缀树上dp一下，转移方程还是比较好想的。

可是后缀树要怎么构造呢？后缀树标准的构造算法太复杂了，我太弱不会写。不过后缀自动机有一个很好的性质：把所有节点和后缀链拿出来，得到的树就是反串的后缀树。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define sigma 27
#define maxn 500010
struct node
{
    ele len,s,ss;
    node *f,*l[sigma],*nxt;
    inline void init()
    {
        s=ss=0;
        f=NULL;
        memset(l,0,sizeof(l));
    }
}pool[maxn<<1];
ele ncnt;
node *h[maxn];
struct SAM
{
    inline void onnewnode(node *p)
    {
        p->nxt=h[p->len];
        h[p->len]=p;
    }
    inline void init()
    {
        memset(h,0,sizeof(h));
        ncnt=0;
        root=&pool[ncnt++];
        root->init();
        root->len=0;
        onnewnode(root);
        end=root;
    }
    inline void add(ele c)
    {
        node *p=&pool[ncnt++],*q;
        p->init(); p->len=end->len+1; onnewnode(p);
        for (q=end; q && !q->l[c]; q=q->f) q->l[c]=p;
        if (!q) p->f=root;
        else if (q->len+1==q->l[c]->len) p->f=q->l[c];
        else
        {
            node *r=&pool[ncnt++],*buf=q->l[c];
            *r=*buf;
            r->len=q->len+1;
            onnewnode(r);
            buf->f=p->f=r;
            for (; q && q->l[c]==buf; q=q->f) q->l[c]=r;
        }
        end=p;
    }
    node *root,*end;
};
ele n;
char s[maxn];
SAM sam;
int main()
{
    scanf("%s",s);
    n=strlen(s);
    sam.init();
    sam.add(26);
    for (int i=n-1; i>=0; --i) sam.add(s[i]-'a');
    ele ans=n*(n-1)*(n+1)/2;
    for (int i=sam.end->len; i>0; --i)
        for (node *j=h[i]; j; j=j->nxt)
        {
            if (!j->s) j->s=j->ss=1;
            else ans-=j->len*(j->s*j->s-j->ss);
            j->f->s+=j->s,j->f->ss+=j->s*j->s;
        }
    printf("%lld\n",ans);
    return 0;
}
