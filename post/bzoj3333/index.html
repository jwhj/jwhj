bzoj3333题解

Jul 3, 2016 10:02 · 222 words · 2 minutes read 题解 bzoj 线段树 统计

注意到这样一个性质：

每次选定一个位置P后，减少的逆序对数量就是p~n中满足要求的人原来构成的逆序对——而且当操作完成之后，这些人将永远不会产生逆序对了。

于是可以先求出以i开头的逆序对fi以及开始时的逆序对总数，对于一个询问p，选出p~n中所有使hj<=hp的j，在答案中减去fj，最后把j这个人删掉。

为了找出所有使hj<=hp的j，维护一个线段树，每次查询p~n中hj最小的位置j，答案中减去fj之后，把hj置为无穷，相当于把这个人删掉。重复此操作直到身高不超过hp的人都被删掉为止。

细节还是有点多的。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define INF 1000000001
#define maxn 500010
const ele size=1<<20;
const ele M=size>>1;
struct ZKW
{
	inline void maintain(ele i)
	{
		a[i]=min(a[i<<1],a[i<<1|1]);
		if (a[i<<1]<a[i<<1|1]) id[i]=id[i<<1]; else id[i]=id[i<<1|1];
	}
	inline void alloc(ele n,ele *b)
	{
		memset(a,0,sizeof(a));
		for (int i=0; i<n; ++i) a[i+1+M]=b[i],id[i+1+M]=i;
		for (int i=M-1; i>0; --i) maintain(i);
	}
	inline void upd(ele i,ele x)
	{
		i+=M+1;
		a[i]=x;
		while (i>1) maintain(i>>=1);
	}
	inline ele minidx(ele l,ele r)
	{
		ele _min=INF,ans;
		++l,++r;
		for (l=l-1+M,r=r+1+M; l^r^1; l>>=1,r>>=1)
		{
			if (~l&1)
				if (a[l^1]<_min) _min=a[l^1],ans=id[l^1];
			if (r&1)
				if (a[r^1]<_min) _min=a[r^1],ans=id[r^1];
		}
		return ans;
	}
	ele a[size],id[size];
};
struct BIT
{
	inline void alloc(ele n)
	{
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele x)
	{
		while (i<=n)
			a[i]+=x,i+=i&(-i);
	}
	inline ele prefix(ele i)
	{
		ele ans=0;
		while (i>0)
			ans+=a[i],i-=i&(-i);
		return ans;
	}
	inline ele sum(ele l,ele r)
	{
		if (l>r) return 0;
		return prefix(r)-prefix(l-1);
	}
	ele n,a[maxn];
};
ele n,m,len,h[maxn],arr[maxn],f[maxn];
ZKW z;
BIT b;
int main()
{
	scanf("%lld%lld",&n,&m);
	for (int i=0; i<n; ++i) scanf("%lld",h+i),arr[i]=h[i];
	sort(arr,arr+n);
	len=unique(arr,arr+n)-arr;
	for (int i=0; i<n; ++i)
		h[i]=lower_bound(arr,arr+len,h[i])-arr+1;
	h[n]=INF-1;
	b.alloc(len);
	ele ans=0;
	for (int i=n-1; i>=0; --i)
		b.add(h[i],1),f[i]=b.sum(0,h[i]-1),ans+=f[i];
	printf("%lld\n",ans);
	z.alloc(n+1,h);
	while (m--)
	{
		ele p;
		scanf("%lld",&p);
		--p;
		if (z.a[p+1+M]<INF)
			while (1)
			{
				ele i=z.minidx(p,n);
				if (h[i]>h[p] || i==n) break;
				ans-=f[i];
				z.upd(i,INF);
			}
		printf("%lld\n",ans);
	}
	return 0;
}
