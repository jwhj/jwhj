#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 810
#define maxk 10
const ele INF=1e9;
ele n,m,q,_log[maxn],a[maxn][maxn],b[maxn][maxn];
struct st{
	ele a[maxn][maxn],f[maxn][maxn][maxk];
	inline void init(){
		for (int i=0; i<n; ++i)
			for (int j=0; j<m; ++j) f[i][j][0]=a[i][j];
		for (int k=1; k<maxk; ++k)
			for (int i=0; i<n; ++i)
				for (int j=0; j<m; ++j){
					f[i][j][k]=f[i][j][k-1];
					ele L=1<<(k-1);
					if (i+L<n) f[i][j][k]=min(f[i][j][k],f[i+L][j][k-1]);
					if (j+L<m) f[i][j][k]=min(f[i][j][k],f[i][j+L][k-1]);
					if (i+L<n && j+L<m)
						f[i][j][k]=min(f[i][j][k],f[i+L][j+L][k-1]);
				}
	}
	inline ele qry(ele i,ele j,ele L){
		ele k=_log[L],L1=1<<k;
		ele ans=INF;
		ans=min(ans,f[i][j][k]);
		ans=min(ans,f[i+L-L1][j][k]);
		ans=min(ans,f[i][j+L-L1][k]);
		ans=min(ans,f[i+L-L1][j+L-L1][k]);
		return ans;
	}
	inline ele qry(ele x1,ele y1,ele x2,ele y2){
		ele Lx=x2-x1+1,Ly=y2-y1+1;
		ele ans=INF;
		if (Lx<Ly){
			for (ele y=y1; y<=y2; y+=Lx)
				ans=min(ans,qry(x1,min(y,y2-Lx+1),Lx));
		}
		else{
			for (ele x=x1; x<=x2; x+=Ly)
				ans=min(ans,qry(min(x,x2-Ly+1),y1,Ly));
		}
		return ans;
	}
}qmx,qmn;
int main(){
//	freopen("phalanx.in","r",stdin);
//	freopen("phalanx.out","w",stdout);
	scanf("%lld%lld",&n,&m);
	_log[1]=0;
	for (int i=2; i<=n || i<=m; ++i) _log[i]=_log[i>>1]+1;
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j){
			scanf("%lld",&a[i][j]);
			qmx.a[i][j]=-a[i][j];
			qmn.a[i][j]=a[i][j];
		}
	qmx.init(); qmn.init();
	memset(b,0,sizeof(b));
	for (int i=1; i<=n; ++i){
		for (int j=1; j<=m; ++j) b[i][j]=b[i][j-1]+a[i-1][j-1];
		for (int j=1; j<=m; ++j) b[i][j]+=b[i-1][j];
	}
	scanf("%lld",&q);
	while (q--){
		char s[10];
		ele x1,y1,x2,y2;
		scanf("%s%lld%lld%lld%lld",s,&x1,&y1,&x2,&y2);
		if (s[1]=='U') printf("%lld\n",b[x2+1][y2+1]-b[x1][y2+1]-b[x2+1][y1]+b[x1][y1]);
		else if (s[1]=='A') printf("%lld\n",-qmx.qry(x1,y1,x2,y2));
		else printf("%lld\n",qmn.qry(x1,y1,x2,y2));
	}
	return 0;
}

代码（T2）：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele long long
using namespace std;
#define maxn 200010
struct edge{
	ele v;
	edge *nxt;
}ep[maxn],*ecnt;
ele n,m,p,fac[maxn],ifac[maxn],ind[maxn],f[maxn],size[maxn],tmp[maxn];
edge *h[maxn];
queue<ele> Q;
void exgcd(ele a,ele b,ele&x,ele&y){
	if (!b){ x=1; y=0; return; }
	exgcd(b,a%b,y,x);
	y-=a/b*x;
}
inline ele inv(ele k){
	ele x,y;
	exgcd(k,p,x,y);
	x%=p; x+=p; x%=p;
	return x;
}
inline void addedge(ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele i){
	f[i]=1;
	size[i]=1;
	for (edge *j=h[i]; j; j=j->nxt){
		dfs(j->v);
		size[i]+=size[j->v];
		(f[i]*=f[j->v])%=p;
		(f[i]*=ifac[size[j->v]])%=p;
	}
	(f[i]*=fac[size[i]-1])%=p;
}
int main(){
//	freopen("photo.in","r",stdin);
//	freopen("photo.out","w",stdout);
	ele T;
	scanf("%lld",&T);
	while (T--){
		scanf("%lld%lld%lld",&n,&m,&p);
		fac[0]=ifac[0]=1;
		for (int i=1; i<=n; ++i){
			fac[i]=fac[i-1]*i%p;
			ifac[i]=inv(fac[i]);
		}
		ecnt=ep; memset(h,0,sizeof(h));
		memset(ind,0,sizeof(ind));
		for (int r=0; r<m; ++r){
			ele x,y;
			scanf("%lld%lld",&x,&y); --x,--y;
			addedge(y,x); ++ind[x];
		}
		memcpy(tmp,ind,sizeof(ind));
		for (int i=0; i<n; ++i)
			if (!ind[i]) Q.push(i);
		ele cnt=0;
		while (!Q.empty()){
			ele k=Q.front(); Q.pop();
			++cnt;
			for (edge *j=h[k]; j; j=j->nxt){
				--ind[j->v];
				if (!ind[j->v]) Q.push(j->v);
			}
		}
		if (cnt<n) puts("0");
		else{
			memcpy(ind,tmp,sizeof(tmp));
			ele ans=fac[n];
			for (int i=0; i<n; ++i)
				if (!ind[i]){
					dfs(i);
					(ans*=f[i])%=p;
					(ans*=ifac[size[i]])%=p;
				}
			printf("%lld\n",ans);
		}
	}
	return 0;
}

代码（T3）：

#pragma GCC optimize(2)
#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 200010
const ele INF=1e18;
struct node{
	ele s,u,v,st;
	node *l,*r;
}np[maxn<<2],*ncnt;
inline void pushdown(node *x,ele a,ele b){
	if (x && ~x->st){
		x->s=x->st*(b-a+1);
		if (x->l) x->l->st=x->st,x->r->st=x->st;
		if (x->st) x->u=a,x->v=b; else x->u=INF,x->v=-1;
		x->st=-1;
	}
}
inline void maintain(node *x,ele a,ele b){
	pushdown(x,a,b);
	if (a==b){
		if (x->s) x->u=x->v=a; else x->u=INF,x->v=-1;
	}
	else{
		ele mid=(a+b)>>1;
		pushdown(x->l,a,mid); pushdown(x->r,mid+1,b);
		x->s=x->l->s+x->r->s;
		x->u=min(x->l->u,x->r->u);
		x->v=max(x->l->v,x->r->v);
	}
}
node* build(ele l,ele r){
	node *p=ncnt++;
	p->st=-1;
	if (l==r){
		p->l=p->r=NULL;
		p->s=1;
		p->u=p->v=l;
	}
	else{
		ele mid=(l+r)>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
		maintain(p,l,r);
	}
	return p;
}
void upd(node *x,ele a,ele b,ele l,ele r,ele k){
	pushdown(x,a,b);
	if (l<=a && b<=r) x->st=k;
	else{
		ele mid=(a+b)>>1;
		if (l<=mid) upd(x->l,a,mid,l,r,k);
		if (mid<r) upd(x->r,mid+1,b,l,r,k);
		pushdown(x->l,a,mid); pushdown(x->r,mid+1,b);
	}
	maintain(x,a,b);
}
ele qry(node *x,ele a,ele b,ele l,ele r){
	pushdown(x,a,b);
	if (l<=a && b<=r) return x->s;
	else{
		ele mid=(a+b)>>1,ans=0;
		if (l<=mid) ans+=qry(x->l,a,mid,l,r);
		if (mid<r) ans+=qry(x->r,mid+1,b,l,r);
		return ans;
	}
}
ele qry1(node *x,ele a,ele b,ele i){
	pushdown(x,a,b);
	if (a==b) return x->v;
	else{
		ele mid=(a+b)>>1;
		if (i<=mid) return qry1(x->l,a,mid,i);
		pushdown(x->l,a,mid);
		return max(x->l->v,qry1(x->r,mid+1,b,i));
	}
}
ele qry2(node *x,ele a,ele b,ele i){
	pushdown(x,a,b);
	if (a==b) return x->u;
	else{
		ele mid=(a+b)>>1;
		if (i>mid) return qry2(x->r,mid+1,b,i);
		pushdown(x->r,mid+1,b);
		return min(qry2(x->l,a,mid,i),x->r->u);
	}
}
struct BIT{
	ele n,a[maxn];
	inline void init(ele n){
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele k){
		++i;
		for (int j=i; j<=n; j+=j&(-j)) a[j]+=k;
	}
	inline ele qry(ele i){
		++i;
		ele ans=0;
		for (int j=i; j; j-=j&(-j)) ans+=a[j];
		return ans;
	}
	inline ele qry(ele l,ele r){
		if (l>r) return 0;
		return qry(r)-qry(l-1);
	}
};
struct st{
	ele n;
	BIT a,b,c;
	inline void init(ele n){
		this->n=n;
		a.init(n); b.init(n); c.init(n);
	}
	inline void add(ele i,ele k){
		a.add(i,k);
		b.add(i,i*k);
		c.add(i,(n-i)*k);
	}
	inline ele qry(ele i){
		if (i==n) return INF;
		return c.qry(i-1)-(n-i)*a.qry(i-1)+b.qry(i+1,n-1)-i*a.qry(i+1,n-1);
	}
}s;
ele n,q;
node *root;
inline ele calc(ele l,ele r){
	while (r-l>1){
		ele mid=(l+r)>>1;
		if (s.qry(mid)>s.qry(mid+1)) l=mid;
		else r=mid;
	}
	return r;
}
int main(){
//	freopen("position.in","r",stdin);
//	freopen("position.out","w",stdout);
	scanf("%lld%lld",&n,&q);
	ncnt=np;
	s.init(n); root=build(0,n-1);
	for (int i=0; i<n; ++i){
		ele x;
		scanf("%lld",&x);
		s.add(i,x);
	}
	while (q--){
		ele cmd,u,v;
		scanf("%lld%lld%lld",&cmd,&u,&v);
		if (cmd==1) s.add(u-1,v);
		else if (cmd==2) s.add(u-1,-v);
		else if (cmd==3) upd(root,0,n-1,u-1,v-1,1);
		else if (cmd==4) upd(root,0,n-1,u-1,v-1,0);
		ele i=calc(-1,n-1);
		ele j=qry1(root,0,n-1,i);
		ele k=qry2(root,0,n-1,i);
		if (!~j && k==INF) puts("-1");
		else if (!~j) printf("%lld\n",k+1);
		else if (k==INF) printf("%lld\n",j+1);
		else{
			if (s.qry(j)>s.qry(k)) swap(j,k);
			printf("%lld\n",j+1);
		}
	}
	return 0;
}
