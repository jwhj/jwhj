bzoj1857题解

Jan 14, 2017 10:02 · 425 words · 2 minutes read 题解 bzoj 三分 分块搜索法

三分套三分就可以了。

至于外层三分为什么是对的，我不知道……晚点看心情试着证一下吧。

UPD：可以用分块搜索法，这样不需要证明是单峰函数。

UPD：内层肯定是单峰函数所以可以内层三分外层分块搜索，这样不需要证明外层是不是单峰，而且效率也提高了。

代码（分块搜索套三分）：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define eps 1e-6
#define K 27
#define M 3
double ax,ay,bx,by,cx,cy,dx,dy,ex,ey,p,q,r;
inline void func(double ax,double ay,double bx,double by,double&cx,double&cy,double lambda){
	double x=(bx-ax)*lambda,y=(by-ay)*lambda;
	cx=ax+x; cy=ay+y;
}
inline double dis(double a,double b,double c,double d){
	return sqrt((a-c)*(a-c)+(b-d)*(b-d));
}
inline double f2(double lambda){
	double fx,fy;
	func(cx,cy,dx,dy,fx,fy,lambda);
	return dis(ex,ey,fx,fy)/r+dis(fx,fy,dx,dy)/q;
}
inline double s2(double l,double r){
	if (r-l<eps) return f2(l);
	double L=(r-l)/3;
	double m1=l+L,m2=r-L;
	if (f2(m1)<=f2(m2)) return s2(l,m2); else return s2(m1,r);
}
inline double f1(double lambda){
	func(ax,ay,bx,by,ex,ey,lambda);
	return dis(ax,ay,ex,ey)/p+s2(0,1);
}
inline double s1(double l,double r){
	if (r-l<eps) return f1(l);
	pair<double,int> f[K];
	double L=(r-l)/K;
	for (int i=0; i<K; ++i) f[i]=make_pair(f1(l+i*L),i);
	sort(f,f+K);
	double ans=1e9;
	for (int i=0; i<M; ++i)
		ans=min(ans,s1(l+f[i].second*L,l+f[i].second*L+L));
	return ans;
}
int main(){
	scanf("%lf%lf%lf%lf%lf%lf%lf%lf%lf%lf%lf",&ax,&ay,&bx,&by,&cx,&cy,&dx,&dy,&p,&q,&r);
	printf("%.2lf\n",s1(0,1));
	return 0;
}

代码（三分套三分）：

#include <cstdio>
#include <cstring>
#include <cmath>
#define ele int
#define real double
using namespace std;
#define eps 1e-6
ele ax,ay,bx,by,cx,cy,dx,dy,p,q,r;
real ex,ey;
inline void func(real ax,real ay,real bx,real by,real lambda,real&cx,real&cy){
	cx=ax+lambda*(bx-ax);
	cy=ay+lambda*(by-ay);
}
inline real sqr(real x){
	return x*x;
}
inline real dis(real ax,real ay,real bx,real by){
	return sqrt(sqr(ax-bx)+sqr(ay-by));
}
inline real f2(real lambda){
	real x,y;
	func(cx,cy,dx,dy,lambda,x,y);
	return dis(ex,ey,x,y)/r+dis(x,y,dx,dy)/q;
}
inline real s2(real l,real r,real fl,real fr){
	if (r-l<eps) return fl;
	real L=(r-l)/3;
	real m1=l+L,m2=r-L;
	real fm1=f2(m1),fm2=f2(m2);
	if (fm1<=fm2)
		return s2(l,m2,fl,fm2);
	else return s2(m1,r,fm1,fr);
}
inline real f1(real lambda){
	func(ax,ay,bx,by,lambda,ex,ey);
	return dis(ax,ay,ex,ey)/p+s2(0,1,f2(0),f2(1));
}
inline real s1(real l,real r,real fl,real fr){
	if (r-l<eps) return fl;
	real L=(r-l)/3;
	real m1=l+L,m2=r-L;
	real fm1=f1(m1),fm2=f1(m2);
	if (fm1<=fm2)
		return s1(l,m2,fl,fm2);
	else return s1(m1,r,fm1,fr);
}
int main(){
	scanf("%d%d%d%d%d%d%d%d%d%d%d",&ax,&ay,&bx,&by,&cx,&cy,&dx,&dy,&p,&q,&r);
	printf("%.2lf\n",s1(0,1,f1(0),f1(1)));
	return 0;
}

代码（分块搜索套分块搜索）：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define M 1
#define K 27
#define eps 1e-5
ele ax,ay,bx,by,cx,cy,dx,dy,p,q,r;
double ex,ey;
inline void func(double ax,double ay,double bx,double by,double lambda,double&cx,double&cy){
    cx=ax+lambda*(bx-ax);
    cy=ay+lambda*(by-ay);
}
inline double sqr(double x){
    return x*x;
}
inline double dis(double ax,double ay,double bx,double by){
    return sqrt(sqr(ax-bx)+sqr(ay-by));
}
inline double f2(double lambda){
    double x,y;
    func(cx,cy,dx,dy,lambda,x,y);
    return dis(ex,ey,x,y)/r+dis(x,y,dx,dy)/q;
}
inline double s2(double l,double r){
    if (r-l<eps) return f2(l);
    double d=(r-l)/K;
    ele j=0;
    double ans=f2(l);
    for (int i=1; i<K; ++i){
    	double tmp=f2(l+d*i);
    	if (tmp<ans){
    		j=i;
    		ans=tmp;
    	}
    }
    for (int i=-M; i<=M; ++i)
    	if (0<=j+i && j+i<K){
    		double tmp=s2(l+d*(j+i),l+d*(j+i+1));
    		ans=min(ans,tmp);
    	}
    return ans;
}
inline double f1(double lambda){
    func(ax,ay,bx,by,lambda,ex,ey);
    return dis(ax,ay,ex,ey)/p+s2(0,1);
}
inline double s1(double l,double r){
    if (r-l<eps) return f1(l);
    double d=(r-l)/K;
    ele j=0;
    double ans=f1(l);
    for (int i=1; i<K; ++i){
    	double tmp=f1(l+d*i);
    	if (tmp<ans){
    		j=i;
    		ans=tmp;
    	}
    }
    for (int i=-M; i<=M; ++i)
    	if (0<=j+i && j+i<K){
    		double tmp=s1(l+d*(j+i),l+d*(j+i+1));
    		ans=min(ans,tmp);
    	}
    return ans;
}
int main(){
    scanf("%d%d%d%d%d%d%d%d%d%d%d",&ax,&ay,&bx,&by,&cx,&cy,&dx,&dy,&p,&q,&r);
    printf("%.2lf\n",s1(0,1));
    return 0;
}
