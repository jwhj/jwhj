bzoj3527题解

Apr 22, 2016 10:02 · 204 words · 1 minute read 题解 bzoj FFT

由题,$Ei=\sum{j<i}\frac{qj}{(i-j)^2}-\sum{j>i}\frac{q_j}{(i-j)^2}$。

假设$Ai=\sum{j<i}\frac{q_j}{(i-j)^2},Bi=\sum{j>i}\frac{q_j}{(i-j)^2}$，则$E_i=A_i-B_i$。

如果让$f(i)=q_i,g(i)=\frac{1}{i^2}$，那么有$Ai=\sum{j<i} f(i)g(i-j)$，这是一个典型的卷积，直接用FFT求就可以了。

如果把q倒过来，就可以用类似的方法求出B。

代码：

#include <cstdio>
#include <cstring>
#include <complex>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 400010
#define pn complex<double>
const double pi=3.1415926535897;
inline void DFT(ele n,pn *a,pn *b){
	for (int i=0; i<n; ++i){
		ele tmp=0,k=i;
		for (int j=1; j<n; j<<=1){
			tmp<<=1;
			tmp|=(k&1);
			k>>=1;
		}
		b[tmp]=a[i];
	}
	for (int i=1; i<n; i<<=1){
		pn o1=(pn){cos(pi/i),sin(pi/i)};
		for (int j=0; j<n; j+=(i<<1)){
			pn o=1;
			for (int k=j; k<j+i; ++k){
				pn u=b[k],t=o*b[k+i];
				b[k]=u+t;
				b[k+i]=u-t;
				o*=o1;
			}
		}
	}
}
inline void IDFT(ele n,pn *a,pn *b){
	for (int i=0; i<n; ++i){
		ele tmp=0,k=i;
		for (int j=1; j<n; j<<=1){
			tmp<<=1;
			tmp|=(k&1);
			k>>=1;
		}
		b[tmp]=a[i];
	}
	for (int i=1; i<n; i<<=1){
		pn o1=(pn){cos(-pi/i),sin(-pi/i)};
		for (int j=0; j<n; j+=(i<<1)){
			pn o=1;
			for (int k=j; k<j+i; ++k){
				pn u=b[k],t=o*b[k+i];
				b[k]=u+t;
				b[k+i]=u-t;
				o*=o1;
			}
		}
	}
	for (int i=0; i<n; ++i) b[i]/=n;
}
ele n;
double q[maxn];
pn a1[maxn],a2[maxn],b1[maxn],b2[maxn],A[maxn],B[maxn];
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%lf",q+i);
	ele tmp=1;
	while (tmp<n) tmp<<=1;
	tmp<<=1;
	for (int i=0; i<tmp; ++i) a1[i]=q[i];
	for (int i=1; i<n; ++i) a2[i]=1.0/i/i;
	DFT(tmp,a1,b1); DFT(tmp,a2,b2);
	for (int i=0; i<tmp; ++i) b1[i]*=b2[i];
	IDFT(tmp,b1,A);
	for (int i=0; i<n; ++i) a1[i]=q[n-i-1];
	DFT(tmp,a1,b1);
	for (int i=0; i<tmp; ++i) b1[i]*=b2[i];
	IDFT(tmp,b1,B);
	for (int i=0; i<n; ++i)
		printf("%.6lf\n",A[i].real()-B[n-i-1].real());
	return 0;
}
