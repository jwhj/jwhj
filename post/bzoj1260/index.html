bzoj1260题解

May 8, 2016 10:02 · 66 words · 1 minute read 题解 bzoj DP

让f[i][j]代表刷[i,j]这段区间至少要几下。

转移方程看代码吧。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 60
#define INF 1000000000
ele n,f[maxn][maxn];
char s[maxn];
int main(){
	scanf("%s",s);
	n=strlen(s);
	memset(f,0,sizeof(f));
	for (int i=0; i<n; ++i) f[i][i]=1;
	for (int p=1; p<n; ++p)
		for (int i=0; i+p<n; ++i){
			int j=i+p;
			if (s[i]==s[j])
				f[i][j]=min(f[i][j-1],f[i+1][j]),
				f[i][j]=min(f[i][j],f[i+1][j-1]+1);
			else{
				f[i][j]=INF;
				for (int k=i; k<=j; ++k)
					f[i][j]=min(f[i][j],f[i][k]+f[k+1][j]);
			}
		}
	printf("%d\n",f[0][n-1]);
	return 0;
}
