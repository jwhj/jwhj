bzoj2435题解

Jun 8, 2016 10:02 · 122 words · 1 minute read 题解 bzoj

理论上来讲dfs一下就好了，然而我写的dfs一直RE，换成bfs就过了，大概是爆栈了吧。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#define ele long long
using namespace std;
#define maxn 1000010
struct edge
{
	ele v,w;
	edge *nxt;
}pool[maxn<<1];
ele n,ecnt,scnt,f[maxn],f1[maxn],size[maxn],seq[maxn];
edge *h[maxn];
queue<ele> q;
inline void addedge(ele u,ele v,ele w)
{
	edge *p=&pool[ecnt++];
	p->v=v; p->w=w;
	p->nxt=h[u];
	h[u]=p;
}
inline ele abs(ele x)
{
	return (x>0)?(x):(-x);
}
int main()
{
	scanf("%lld",&n);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i)
	{
		ele x,y,z;
		scanf("%lld%lld%lld",&x,&y,&z);
		--x,--y;
		addedge(x,y,z); addedge(y,x,z);
	}
	for (edge *j=h[0]; j; j=j->nxt)
	{
		q.push(j->v);
		f[j->v]=0;
		f1[j->v]=j->w;
	}
	scnt=0;
	while (!q.empty())
	{
		ele p=seq[scnt++]=q.front();
		q.pop();
		size[p]=1;
		for (edge *j=h[p]; j; j=j->nxt)
			if (j->v!=f[p])
			{
				q.push(j->v);
				f[j->v]=p;
				f1[j->v]=j->w;
			}
	}
	for (int i=scnt-1; i>=0; --i)
		size[f[seq[i]]]+=size[seq[i]];
	ele ans=0;
	for (int i=1; i<n; ++i)
		ans+=f1[i]*abs(n-size[i]*2);
	printf("%lld\n",ans);
	return 0;
}
