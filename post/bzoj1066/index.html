bzoj1066题解

Mar 1, 2017 10:02 · 219 words · 2 minutes read 题解 bzoj 网络流

用网络流来做。

每个石柱拆成两个点，设为a，b，这两个点之间连一条容量为石柱高的边。

源点向每个有蜥蜴的石柱的a点连一条容量为1的边。

每个可以跳出去的石柱的b点向汇点连一条容量为INF的边。

石柱x能跳到石柱y，则在x的b点与y的a点间连一条容量为INF的边。

这个网络的最大流即为能跳出去的蜥蜴数量的最大值。

一开始RE+WA了几回，错误大概是：

  • 忘了拆点之后点的数量要乘2,数组开得不够大
  • 把i*c+j写成i*r+j
  • 判断能不能跳出的时候不等式带不带等号弄错

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 810
#define INF 1000000000
struct edge{
	ele v,r;
	edge *nxt,*rev;
}ep[10000000];
ele r,c,d,cnt,ecnt,dep[maxn];
char s[maxn];
edge *g[maxn];
inline void addedge1(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->r=w; p->nxt=g[u]; p->rev=ep+((ecnt-1)^1);
	g[u]=p;
}
inline void addedge(ele u,ele v,ele w){
	addedge1(u,v,w); addedge1(v,u,0);
}
inline ele bfs(ele s,ele t){
	static ele Q[maxn];
	ele qh=0,qt=1;
	Q[0]=s;
	memset(dep,0,sizeof(dep));
	dep[s]=1;
	while (qh<qt){
		for (edge *j=g[Q[qh]]; j; j=j->nxt)
			if (!dep[j->v] && j->r){
				dep[j->v]=dep[Q[qh]]+1;
				Q[qt++]=j->v;
			}
		++qh;
	}
	return dep[t];
}
ele dfs(ele i,ele t,ele p){
	if (i==t) return p;
	for (edge *j=g[i]; j; j=j->nxt)
		if (dep[i]+1==dep[j->v] && j->r){
			ele tmp=dfs(j->v,t,min(p,j->r));
			if (tmp){
				j->r-=tmp;
				j->rev->r+=tmp;
				return tmp;
			}
		}
	return 0;
}
inline ele dinic(ele s,ele t){
	ele ans=0;
	while (bfs(s,t)) ans+=dfs(s,t,INF);
	return ans;
}
int main(){
	scanf("%d%d%d",&r,&c,&d);
	ecnt=0;
	for (int i=0; i<r; ++i){
		scanf("%s",s);
		for (int j=0; j<c; ++j)
			addedge(2+(i*c+j)*2,2+(i*c+j)*2+1,s[j]-'0');
	}
	cnt=0;
	for (int i=0; i<r; ++i){
		scanf("%s",s);
		for (int j=0; j<c; ++j)
			if (s[j]=='L')
				addedge(0,2+(i*c+j)*2,1),++cnt;
	}
	for (int i=0; i<r; ++i)
		for (int j=0; j<c; ++j){
			for (int x=-d; x<=d; ++x)
				for (int y=-d; y<=d; ++y)
					if (i+x>=0 && i+x<r && j+y>=0 && j+y<c && x*x+y*y<=d*d)
						addedge(2+(i*c+j)*2+1,2+((i+x)*c+j+y)*2,INF);
			if (i<d || r-i<=d || j<d || c-j<=d) addedge(2+(i*c+j)*2+1,1,INF);
		}
	printf("%d\n",cnt-dinic(0,1));
	return 0;
}
