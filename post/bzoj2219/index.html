bzoj2219题解

Feb 15, 2017 10:02 · 241 words · 2 minutes read 题解 bzoj 数学 BSGS 原根

如果2k+1存在原根，直接用原根来做就可以了。

不过，大多数情况下2k+1都是不存在原根的。

将2k+1分解质因数，设$2k+1=p_1^{a_1} p_2^{a_2}\ldots p_s^{a_s}$。我们对于每个$\mod p_i^{a_i}$分别用原根做一遍，然后乘起来就是答案，这个可以用中国剩余定理证明。

细节比较多，可以看一下这个题解。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <map>
#define ele long long
using namespace std;
inline void exgcd(ele a,ele b,ele&x,ele&y){
	if (!b){ x=1; y=0; return; }
	exgcd(b,a%b,y,x);
	y-=a/b*x;
}
inline ele inv(ele k,ele MOD){
	ele x,y;
	exgcd(k,MOD,x,y);
	return (x%MOD+MOD)%MOD;
}
inline ele bsgs(ele a,ele b,ele MOD){
	ele m=(ele)(sqrt(MOD)+1),t1=1;
	static map<ele,ele> h;
	h.clear();
	for (int i=0; i<m; ++i,t1=t1*a%MOD) h[t1]=i;
	ele t2=1;
	for (int i=0; i<m; ++i,t2=t2*t1%MOD){
		ele t3=b*inv(t2,MOD)%MOD;
		if (h.find(t3)!=h.end()) return m*i+h[t3];
	}
}
inline ele pow(ele a,ele x){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp)
		if (x&1) ans=ans*tmp;
	return ans;
}
inline ele pow(ele a,ele x,ele MOD){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp%MOD)
		if (x&1) ans=ans*tmp%MOD;
	return ans;
}
inline ele calcg(ele MOD,ele phi){
	static ele f[100000];
	ele cnt=0;
	for (int i=2; i*i<=phi; ++i)
		if (phi%i==0)
			f[cnt++]=i,f[cnt++]=phi/i;
	for (int g=2; g<MOD; ++g){
		bool flag=true;
		for (int i=0; i<cnt; ++i)
			if (pow(g,f[i],MOD)==1){
				flag=false;
				break;
			}
		if (flag) return g;
	}
}
ele gcd(ele a,ele b){
	return (b)?(gcd(b,a%b)):(a);
}
inline ele func(ele a,ele b,ele MOD){
	ele tmp=gcd(a,MOD);
	return (b%tmp)?(0):(tmp);
}
int main(){
	int T;
	scanf("%d",&T);
	while (T--){
		ele a,b,k;
		scanf("%lld%lld%lld",&a,&b,&k);
		k=k<<1|1;
		ele ans=1,tmp=k;
		for (int i=2; ans && i*i<=tmp; ++i){
			ele t1=1,t2=0;
			while (k%i==0) k/=i,t1*=i,++t2;
			if (!t2) continue;
			if (b%t1==0)
				ans*=pow(i,t2-(t2-1)/a-1);
			else{
				ele t3=b,t4=0;
				while (t3%i==0 && t1%i==0)
					t3/=i,t1/=i,++t4;
				if (t4%a)
					ans=0;
				else{
					ele g=calcg(t1,t1/i*(i-1));
					ele b1=bsgs(g,t3,t1);
					ans*=func(a,b1,t1/i*(i-1))*pow(i,t4-t4/a);
				}
			}
		}
		if (k>1){
			ele g=calcg(k,k-1);
			ele b1=bsgs(g,b,k);
			ans*=func(a,b1,k-1);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
