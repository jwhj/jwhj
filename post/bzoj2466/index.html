bzoj2466题解

Dec 23, 2016 10:02 · 158 words · 1 minute read 题解 bzoj 数学 高斯消元

首先，按按钮的次序是无关紧要的。

一开始我在想树形dp，然而不知道怎么让它无后效性，结果就没想出来。

所以可以列一个异或方程组用高斯消元来做。异或方程组的高斯消元，在普通高斯消元的基础上想一想就可以想出来了，当然略微有一点不同。细节可以看代码。

对异或方程组进行高斯消元后，主对角线上可能会有若干个0,它们对应的未知数是自由元，需要枚举它们的取值然后找到一个最优方案。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 110
struct edge{
    ele v;
    edge *nxt;
}ep[maxn<<1];
ele n,ecnt,ans,a[maxn][maxn],b[maxn],x[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v){
    edge *p=&ep[ecnt++];
    p->v=v; p->nxt=h[u];
    h[u]=p;
}
inline void gauss(ele n,ele a[maxn][maxn],ele *b){
    for (int i=0; i<n-1; ++i)
        for (int j=i+1; j<n; ++j)
            if (a[i][i]==0){
                swap(b[i],b[j]);
                for (int k=0; k<n; ++k)
                    swap(a[i][k],a[j][k]);
            }
            else
                if (a[j][i]==1){
                    b[j]^=b[i];
                    for (int k=0; k<n; ++k)
                        a[j][k]^=a[i][k];
                }
}
void dfs(ele i,ele j){
    if (j>ans) return;
    if (i<0){
        ans=j;
        return;
    }
    ele k=0;
    for (int p=n-1; p>i; --p)
        k^=(x[p] && a[i][p]);
    if (a[i][i]==0){
        if (k!=b[i]) return;
        x[i]=0;
        dfs(i-1,j);
        x[i]=1;
        dfs(i-1,j+1);
    }
    else{
        x[i]=b[i]^k;
        dfs(i-1,j+x[i]);
    }
}
int main(){
    while (scanf("%d",&n),n){
        ecnt=0;
        memset(h,0,sizeof(h));
        for (int i=0; i<n-1; ++i){
            ele u,v;
            scanf("%d%d",&u,&v);
            --u,--v;
            addedge(u,v); addedge(v,u);
        }
        memset(a,0,sizeof(a));
        for (int i=0; i<n; ++i){
            a[i][i]=1;
            for (edge *j=h[i]; j; j=j->nxt)
                a[i][j->v]=1;
            b[i]=1;
        }
        gauss(n,a,b);
        ans=n;
        dfs(n-1,0);
        printf("%d\n",ans);
    }
    return 0;
}
