bzoj3720题解

Jan 21, 2017 10:02 · 211 words · 1 minute read 题解 bzoj 块状树

用块状树来做，每次加入一个点的时候，如果父节点所在的块已经满了，就新开一个块，否则把这个节点加到父节点所在的块里面。

每个块维护块中的所有权值排好序的列表。

代码：

#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 60010
#define B 250
struct edge{
    ele v;
    edge *nxt;
}ep[maxn*3];
ele n,q,ecnt,w[maxn],fa[maxn],r[maxn],s[maxn];
edge *h[maxn],*h1[maxn];
vector<ele> lst[maxn];
inline void addedge(edge *h[],ele u,ele v){
    edge *p=&ep[ecnt++];
    p->v=v; p->nxt=h[u];
    h[u]=p;
}
inline void insert(ele i){
    ele j=r[i];
    lst[j].push_back(w[i]);
    for (int p=lst[j].size()-1; p && lst[j][p-1]>lst[j][p]; --p)
        swap(lst[j][p-1],lst[j][p]);
}
inline void erase(ele i){
    ele j=r[i],p=lower_bound(lst[j].begin(),lst[j].end(),w[i])-lst[j].begin();
    for (; p+1<lst[j].size(); ++p)
        swap(lst[j][p],lst[j][p+1]);
    lst[j].erase(lst[j].end()-1);
}
void dfs(ele p,ele i){
    fa[i]=p;
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p){
            if (s[r[i]]==B){
                r[j->v]=j->v;
                addedge(h1,r[i],j->v);
            }
            else r[j->v]=r[i];
            ++s[r[j->v]];
            dfs(i,j->v);
        }
    insert(i);
}
ele dfs1(ele i,ele x){
    ele ans=lst[i].end()-upper_bound(lst[i].begin(),lst[i].end(),x);
    for (edge *j=h1[i]; j; j=j->nxt)
        ans+=dfs1(j->v,x);
    return ans;
}
ele dfs2(ele p,ele i,ele x){
    if (i==r[i]) return dfs1(i,x);
    ele ans=(w[i]>x)?(1):(0);
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p)
            ans+=dfs2(i,j->v,x);
    return ans;
}
int main(){
    scanf("%d",&n);
    ecnt=0;
    memset(h,0,sizeof(h)); memset(h1,0,sizeof(h1));
    for (int i=0; i<n-1; ++i){
        ele u,v;
        scanf("%d%d",&u,&v);
        --u,--v;
        addedge(h,u,v); addedge(h,v,u);
    }
    for (int i=0; i<n; ++i)
        scanf("%d",w+i);
    memset(s,0,sizeof(s));
    s[0]=1;
    r[0]=0;
    dfs(-1,0);
    scanf("%d",&q);
    ele ans=0;
    while (q--){
        ele op,u,x;
        scanf("%d%d%d",&op,&u,&x);
        u^=ans; x^=ans;
        --u;
        if (op==0){
            ans=dfs2(fa[u],u,x);
            printf("%d\n",ans);
        }
        else if (op==1){
            erase(u);
            w[u]=x;
            insert(u);
        }
        else if (op==2){
            addedge(h,n,u); addedge(h,u,n);
            fa[n]=u;
            if (s[r[u]]==B){
                addedge(h1,r[u],n);
                r[n]=n;
            }
            else r[n]=r[u];
            ++s[r[n]];
            w[n]=x;
            insert(n);
            ++n;
        }
    }
    return 0;
}
