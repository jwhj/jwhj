bzoj1076题解

Aug 17, 2017 10:02 · 91 words · 1 minute read 题解 bzoj dp 状态压缩

题目里面的最优策略应该是使得期望得分最高。

看到数据范围里面的15就很容易想到状态压缩了。一个很自然的想法是，让f[i][s]表示已经出现了i个宝物，是否出现的状态为s时的期望得分，然后枚举最后一个出现的宝物进行dp。不过，如果让f[i][s]表示已经出现了i个宝物，是否出现的状态为s时，到最后一个宝物出现期望还能得多少分，dp起来会方便一些。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxk 110
#define maxs (1<<15)
ele k,n,p[maxk],prv[maxk],_log[maxs];
double f[maxk][maxs];
inline ele getv(ele x,ele i){
	return (x>>i)&1;
}
inline void setv(ele&x,ele i,ele k){
	x&=~(1<<i);
	x|=(k<<i);
}
int main(){
	_log[1]=0;
	for (int i=2; i<maxs; ++i) _log[i]=_log[i>>1]+1;
	scanf("%d%d",&k,&n);
	for (int i=0; i<n; ++i){
		scanf("%d",p+i);
		ele x;
		prv[i]=0;
		while (scanf("%d",&x),x) setv(prv[i],x-1,1);
	}
	memset(f,0,sizeof(f));
	for (int i=k-1; ~i; --i)
		for (int s=0; s<(1<<n); ++s){
			for (int j=0; j<n; ++j)
				if (prv[j]&~s) f[i][s]+=f[i+1][s];
				else f[i][s]+=max(f[i+1][s],f[i+1][s|(1<<j)]+p[j]);
			f[i][s]/=n;
		}
	printf("%.6lf\n",f[0][0]);
	return 0;
}
