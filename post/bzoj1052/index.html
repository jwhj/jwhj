bzoj1052题解

Sep 13, 2017 10:02 · 140 words · 1 minute read 题解 bzoj 二分 乱搞

一个很明显的思路就是二分答案，然后判定可行性。

判定可行性的时候，有一个很容易想到的贪心思路：用一个边平行于坐标轴的最小矩形覆盖所有点，然后在左上角塞一个正方形，对剩下的点反复进行这样的操作。很可惜这个思路不对，结果我就卡了很久，后来发现正确思路是，在矩形的四个角之一塞一个正方形，然后对剩下的点反复进行这个操作。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 20010
const ele INF=1e9;
ele n,m,x[maxn],y[maxn],a[maxn];
inline ele _abs(ele x){
	return (x<0)?-x:x;
}
bool dfs(ele L,ele i){
	if (i==3) return !m;
	ele mnx=INF,mxx=-INF,mny=INF,mxy=-INF;
	for (int j=0; j<m; ++j){
		mnx=min(mnx,x[a[j]]); mxx=max(mxx,x[a[j]]);
		mny=min(mny,y[a[j]]); mxy=max(mxy,y[a[j]]);
	}
	ele *t1=new ele[m];
	ele m1=m;
	memcpy(t1,a,sizeof(ele)*m);
	bool f=false;
	for (int s=0; s<2 && !f; ++s)
		for (int t=0; t<2 && !f; ++t){
			ele p=mnx+s*(mxx-mnx),q=mny+t*(mxy-mny);
			m=0;
			for (int j=0; j<m1; ++j)
				if (_abs(p-x[t1[j]])>L || _abs(q-y[t1[j]])>L) a[m++]=t1[j];
			if (dfs(L,i+1)) f=true;
		}
	m=m1;
	memcpy(a,t1,sizeof(ele)*m);
	return f;
}
inline bool test(ele L){
	m=n;
	for (int i=0; i<n; ++i) a[i]=i;
	return dfs(L,0);
}
int main(){
	scanf("%d",&n);
	ele mnx=INF,mxx=-INF,mny=INF,mxy=-INF;
	for (int i=0; i<n; ++i){
		scanf("%d%d",x+i,y+i);
		mnx=min(mnx,x[i]); mxx=max(mxx,x[i]);
		mny=min(mny,y[i]); mxy=max(mxy,y[i]);
	}
	ele L=-1,R=max(mxx-mnx,mxy-mny);
	while (R-L>1){
		ele mid=(L+R)>>1;
		if (test(mid)) R=mid; else L=mid;
	}
	printf("%d\n",R);
	return 0;
}
