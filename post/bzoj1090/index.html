bzoj1090题解

Aug 28, 2017 10:02 · 125 words · 1 minute read 题解 bzoj hash dp

用f[i][j]表示i到j的最短折叠长度，转移的时候考虑两种情况，一个是枚举k，分成f[i][k]+f[k+1][j]，另一个则要枚举循环节长度，用hash判断可行性。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define ull unsigned long long
using namespace std;
#define maxn 110
#define seed 10007
ele n,f[maxn][maxn];
ull xl[maxn],h[maxn];
char s[maxn];
inline ull hsh(ele i,ele L){
	return h[i]-h[i+L]*xl[L];
}
inline ele calc(ele x){
	if (x<10) return 1;
	if (x<100) return 2;
	return 3;
}
int main(){
	scanf("%s",s); n=strlen(s);
	xl[0]=1;
	for (int i=1; i<=n; ++i) xl[i]=xl[i-1]*seed;
	h[n-1]=s[n-1];
	for (int i=n-2; ~i; --i) h[i]=h[i+1]*seed+s[i];
	for (int i=0; i<n; ++i) f[i][i]=1;
	for (int L=2; L<=n; ++L)
		for (int i=0; i+L<=n; ++i){
			ele j=i+L-1;
			f[i][j]=L;
			for (int k=i; k<j; ++k) f[i][j]=min(f[i][j],f[i][k]+f[k+1][j]);
			for (int p=1; p<=L; ++p)
				if (L%p==0){
					bool flag=true;
					for (int k=i+p; k<=j; k+=p)
						if (hsh(i,p)!=hsh(k,p)){ flag=false; break; }
					if (flag) f[i][j]=min(f[i][j],f[i][i+p-1]+2+calc(L/p));
				}
		}
	printf("%d\n",f[0][n-1]);
	return 0;
}
