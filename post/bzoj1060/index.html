bzoj1060题解

Aug 28, 2017 10:02 · 94 words · 1 minute read 题解 bzoj dp

树形dp。

据说当时做数据的时候标程有问题，要除了ans以外全部用int，ans用long long才能过……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 500010
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxn<<1],*ecnt;
ele n,r,d[maxn];
long long ans;
edge *h[maxn];
inline void addedge(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele p,ele i){
	d[i]=0;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p){
			dfs(i,j->v);
			d[i]=max(d[i],j->w+d[j->v]);
		}
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p) ans+=d[i]-d[j->v]-j->w;
}
int main(){
	scanf("%d%d",&n,&r); --r;
	ecnt=ep; memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v,w;
		scanf("%d%d%d",&u,&v,&w); --u,--v;
		addedge(u,v,w); addedge(v,u,w);
	}
	ans=0;
	dfs(-1,r);
	printf("%lld\n",ans);
	return 0;
}
