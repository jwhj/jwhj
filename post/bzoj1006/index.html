bzoj1006题解

Aug 5, 2017 10:02 · 133 words · 1 minute read 题解 bzoj 图论 弦图

可以去看一下陈丹琦09年讲稿《弦图与区间图》。

一开始我很智障地打了这么一段代码：

for (int p=n-1; ~p; --p){
	...
	if (vis[k]) continue;
	...
}

mdzz……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define ele int
using namespace std;
#define maxn 10010
#define maxm 1000010
struct edge{
	ele v;
	edge *nxt;
}ep[maxm<<1],*ecnt;
ele n,m,seq[maxn],w[maxn],pos[maxn];
bool vis[maxn];
edge *h[maxn];
priority_queue<pair<ele,ele> > Q;
inline void addedge(ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=ep;
	memset(h,0,sizeof(h));
	while (m--){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		addedge(u,v); addedge(v,u);
	}
	memset(w,0,sizeof(w));
	memset(vis,0,sizeof(vis));
	for (int i=0; i<n; ++i) Q.push(make_pair(0,i));
	for (int p=n-1; ~p;){
		ele k=Q.top().second; Q.pop();
		if (vis[k]) continue;
		seq[p]=k; pos[k]=p; vis[k]=true;
		for (edge *j=h[k]; j; j=j->nxt){
			if (vis[j->v]) continue;
			++w[j->v];
			Q.push(make_pair(w[j->v],j->v));
		}
		--p;
	}
	ele ans=1;
	for (int i=n-1; ~i; --i){
		ele tmp=1;
		for (edge *j=h[i]; j; j=j->nxt)
			if (pos[j->v]>pos[i]) ++tmp;
		ans=max(ans,tmp);
	}
	printf("%d\n",ans);
	return 0;
}
