bzoj1028题解

May 27, 2017 10:02 · 157 words · 1 minute read 题解 bzoj 记忆化搜索

大水题……

先枚举等候牌的序号，然后判断是否是和牌。

状态(i,a,b,p)表示只考虑前i种牌，第i-1种牌被钦定用掉了a张，第i种牌被钦定用掉了b张，p为是否存在一个对子。然后记忆化搜索就可以了。

状态总数很多，但有很多状态实际上是达不到的，所以用set记忆化。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#include <set>
#define ele int
using namespace std;
#define maxn 410
struct st{
	ele i,a,b,p;
	inline bool operator<(st rhs)const{
		if (i!=rhs.i) return i<rhs.i;
		if (a!=rhs.a) return a<rhs.a;
		if (b!=rhs.b) return b<rhs.b;
		return p<rhs.p;
	}
};
ele n,m,acnt,cnt[maxn],ans[maxn];
set<st> S;
queue<st> Q;
inline void add(st k){
	if (!S.count(k)){
		S.insert(k);
		Q.push(k);
	}
}
inline bool dp(){
	S.clear();
	Q.push((st){n,0,0,1});
	bool ans=false;
	while (!Q.empty()){
		st k=Q.front(); Q.pop();
		if (k.i==1){
			if (k.a) continue;
			if (!k.p && (cnt[1]-k.b)%3!=0) continue;
			if (k.p && (cnt[1]-k.b)%3!=2) continue;
			ans=true;
			continue;
		}
		if (k.p && cnt[k.i]-k.b>=2) add((st){k.i,k.a,k.b+2,0});
		for (int i=0; i+k.b<=cnt[k.i] && i+k.a<=cnt[k.i-1] && i<=cnt[k.i-2]; ++i)
			if ((cnt[k.i]-i-k.b)%3==0) add((st){k.i-1,i,i+k.a,k.p});
	}
	return ans;
}
int main(){
	scanf("%d%d",&n,&m);
	memset(cnt,0,sizeof(cnt));
	for (int i=0; i<3*m+1; ++i){
		ele x;
		scanf("%d",&x);
		++cnt[x];
	}
	acnt=0;
	for (int i=1; i<=n; ++i){
		++cnt[i];
		if (dp()) ans[acnt++]=i;
		--cnt[i];
	}
	if (!acnt) puts("NO");
	else{
		for (int i=0; i<acnt-1; ++i) printf("%d ",ans[i]);
		printf("%d\n",ans[acnt-1]);
	}
	return 0;
}
