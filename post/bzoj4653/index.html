bzoj4653(noi2016 区间)题解

Aug 27, 2016 10:02 · 204 words · 1 minute read 题解 bzoj 线段树

对于若干个区间，如果其中有一个点被覆盖至少m次，那么一定能从这些区间中选出m个，满足题意。

于是我就开始想扫瞄线，想了好久没想出来……最后只好看题解。

先把区间按长度排序，用两个指针l，r表示当前选出的最短和最长的区间。

用一个线段树维护每个点被l，r间的这些区间覆盖了多少次。

每次将r右移，并尝试将l右移，使l指向的区间尽可能长。用r指向的区间长度-l指向的区间长度来更新答案。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 500010
#define INF 1000000001
struct node
{
	ele add,max;
	node *l,*r;
}pool[maxn<<3];
ele n,m,len,ncnt,arr[maxn<<1];
node *root;
struct itv
{
	ele l,r;
	inline bool operator<(itv b)const
	{
		return arr[r]-arr[l]<arr[b.r]-arr[b.l];
	}
}a[maxn];
inline void maintain(node *x)
{
	if (x->l) x->max=max(x->l->max,x->r->max); else x->max=0;
	x->max+=x->add;
}
node* build(ele l,ele r)
{
	node *p=&pool[ncnt++];
	p->add=p->max=0;
	if (l==r)
		p->l=p->r=NULL;
	else
	{
		ele mid=(l+r)>>1;
		p->l=build(l,mid);
		p->r=build(mid+1,r);
	}
	return p;
}
void add(node *x,ele a,ele b,ele l,ele r,ele k)
{
	if (l<=a && b<=r) x->add+=k;
	else
	{
		ele mid=(a+b)>>1;
		if (l<=mid) add(x->l,a,mid,l,r,k);
		if (mid<r) add(x->r,mid+1,b,l,r,k);
	}
	maintain(x);
}
ele query(node *x,ele a,ele b,ele l,ele r)
{
	if (l<=a && b<=r) return x->max;
	ele ans=0,mid=(a+b)>>1;
	if (l<=mid) ans=max(ans,query(x->l,a,mid,l,r));
	if (mid<r) ans=max(ans,query(x->r,mid+1,b,l,r));
	return ans+x->add;
}
inline ele idx(ele x)
{
	return lower_bound(arr,arr+len,x)-arr;
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; ++i)
	{
		scanf("%lld%lld",&a[i].l,&a[i].r);
		arr[i<<1]=a[i].l,arr[i<<1|1]=a[i].r;
	}
	sort(arr,arr+(n<<1));
	len=unique(arr,arr+(n<<1))-arr;
	for (int i=0; i<n; ++i)
		a[i].l=idx(a[i].l),a[i].r=idx(a[i].r);
	sort(a,a+n);
	ncnt=0;
	root=build(0,len-1);
	ele l=0,r,ans=INF;
	for (int r=0; r<n; ++r)
	{
		add(root,0,len-1,a[r].l,a[r].r,1);
		ele tmp=query(root,0,len-1,0,len-1);
		if (tmp<m) continue;
		while (tmp>=m)
		{
			add(root,0,len-1,a[l].l,a[l].r,-1);
			tmp=query(root,0,len-1,0,len-1);
			++l;
		}
		--l;
		add(root,0,len-1,a[l].l,a[l].r,1);
		ans=min(ans,arr[a[r].r]-arr[a[r].l]-arr[a[l].r]+arr[a[l].l]);
	}
	if (ans<INF) printf("%d\n",ans);
	else puts("-1");
	return 0;
}
