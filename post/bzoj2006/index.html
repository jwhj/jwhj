bzoj2006题解

Jan 15, 2017 10:02 · 143 words · 1 minute read 题解 bzoj A*

令四元组(i,l,r,t)表示一个决策：左端点选i，右端点在l～r中选，并且选t的时候取到最大值。

把(1,l1,r1,t1) (2,l2,r2,t2)…扔进一个优先队列里，每次取出最大的，假设是(i,l,r,t)，把(i,l,t-1,t’)和(i,t+1,r,t”)扔进优先队列里。

代码：

#include <cstdio>
#include <cstring>
#include <queue>
#define ele long long
using namespace std;
#define maxn 500010
#define maxk 20
ele n,k,l,r,a[maxn],log2[maxn],st[maxn][maxk];
struct hn{
    ele i,l,r,t;
    inline bool operator<(hn b)const{
        return a[i]-a[t+1]<a[b.i]-a[b.t+1];
    }
};
priority_queue<hn> q;
inline ele query(ele l,ele r){
    ele L=1<<log2[r-l+1];
    ele j=r-L+1;
    if (a[st[l][log2[L]]]<a[st[j][log2[L]]])
        return st[l][log2[L]];
    else return st[j][log2[L]];
}
int main(){
    scanf("%lld%lld%lld%lld",&n,&k,&l,&r);
    for (int i=0; i<n; ++i)
        scanf("%lld",a+i);
    a[n]=0;
    for (int i=n-2; i>=0; --i)
        a[i]+=a[i+1];
    log2[1]=0;
    for (int i=2; i<=n; i<<=1){
        log2[i]=log2[i>>1]+1;
        for (int j=i+1; j<(i<<1) && j<=n; ++j)
            log2[j]=log2[i];
    }
    for (int i=0; i<=n; ++i)
        st[i][0]=i;
    for (int j=1; j<maxk; ++j)
        for (int i=0; i<=n; ++i){
            if (i+(1<<j-1)>n || a[st[i][j-1]]<a[st[i+(1<<j-1)][j-1]])
                st[i][j]=st[i][j-1];
            else st[i][j]=st[i+(1<<j-1)][j-1];
        }
    for (int i=0; i+l-1<n; ++i){
        ele v=min(n-1,i+r-1);
        q.push({i,i+l-1,v,query(i+l,v+1)-1});
    }
    ele ans=0;
    for (int i=0; i<k; ++i){
        hn k=q.top(); q.pop();
        ans+=a[k.i]-a[k.t+1];
        if (k.t>k.l)
            q.push({k.i,k.l,k.t-1,query(k.l+1,k.t)-1});
        if (k.t<k.r)
            q.push({k.i,k.t+1,k.r,query(k.t+2,k.r+1)-1});
    }
    printf("%lld\n",ans);
    return 0;
}
