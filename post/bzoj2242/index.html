bzoj2242题解

Jun 10, 2016 10:02 · 256 words · 2 minutes read 题解 bzoj 数学 exgcd BSGS

做道数学题给明天高联初赛求RP。

第一问用快速幂，第二问用拓展欧几里得，第三问用BSGS。

注意一大堆特判。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#define ele long long
using namespace std;
#define R 100007
struct HASH
{
	const ele null;
	HASH():null(-1){}
	inline void clear()
	{
		for (int i=0; i<R; ++i)
			a[i]=null;
	}
	inline ele f2(ele x)
	{
		return R-x%R;
	}
	inline ele probe(ele x,ele i)
	{
		return (x%R+i*f2(x))%R;
	}
	inline ele find(ele x)
	{
		for (int i=0; i<R; ++i)
		{
			ele j=probe(x,i);
			if (a[j]==null) return null;
			if (a[j]==x) return j;
		}
		return null;
	}
	inline void ins(ele x,ele s)
	{
		for (int i=0; i<R; ++i)
		{
			ele j=probe(x,i);
			if (a[j]==null)
			{
				a[j]=x;
				b[j]=s;
				return;
			}
		}
	}
	ele a[R],b[R];
};
inline ele pow(ele a,ele x,ele p)
{
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp%p)
		if (x&1)
			ans=ans*tmp%p;
	return ans;
}
void exgcd(ele a,ele b,ele&x,ele&y)
{
	if (!b)
	{
		x=1; y=0;
		return;
	}
	exgcd(b,a%b,y,x);
	y-=x*(a/b);
}
inline ele rev(ele y,ele p)
{
	ele x1,y1;
	exgcd(y,p,x1,y1);
	return (x1%p+p)%p;
}
inline ele bsgs(ele y,ele z,ele p)
{
	if (z==1) return 0;
	if (!y) return -1;
	ele c=(ele)(sqrt(p)+0.5)+1;
	ele u=1;
	static HASH m;
	m.clear();
	ele j;
	for (int i=0; i<c; ++i,u=u*y%p)
		if ((j=m.find(u))==m.null)
			m.ins(u,i);
	ele v=1;
	for (int i=0; i<c; ++i)
	{
		ele tmp=z*rev(v,p)%p,j;
		if ((j=m.find(tmp))!=m.null)
		{
			ele ans=i*c+m.b[j];
			return ans;
		}
		v=v*u%p;
	}
	return -1;
}
int main()
{
	ele t,k;
	scanf("%lld%lld",&t,&k);
	while (t--)
	{
		ele y,z,p;
		scanf("%lld%lld%lld",&y,&z,&p);
		y%=p;
		if (k==1) printf("%lld\n",pow(y,z,p));
		else if (k==2)
			if (!y)
				puts("Orz, I cannot find x!");
			else printf("%lld\n",rev(y,p)*z%p);
		else if (k==3)
		{
			ele ans=bsgs(y,z,p);
			if (ans==-1) puts("Orz, I cannot find x!");
			else printf("%lld\n",ans);
		}
	}
	return 0;
}
