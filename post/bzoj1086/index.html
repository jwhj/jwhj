bzoj1086题解

Aug 21, 2017 10:02 · 108 words · 1 minute read 题解 bzoj 块状树

dfs整棵树，当dfs(u)的时候，为u维护一个等待序列，初始时为空。设v为u的一个孩子，递归dfs(v)结束后将v的等待序列加到u的末尾。如果u的等待序列长度不小于B，将等待序列中的所有点划入一个省，以u为省会（但不将v划入这个省），同时将u的等待序列清空。u的所有孩子都dfs完后，将u加入它的等待序列，回溯。

dfs(root)结束后，root的等待序列中还会剩下一些点，直接划入最后一个省就可以了。

等待序列的维护可以用一个栈，另外要注意维护一个“栈底”，以此将当前节点的等待序列与其祖先的兄弟结点的等待序列分开。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 1010
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<1],*ecnt;
ele n,B,bcnt,btm,top,blg[maxn],stk[maxn],a[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v){
	edge *p=ecnt++;
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele p,ele i){
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p){
			ele tmp=btm; btm=top;
			dfs(i,j->v);
			btm=tmp;
			if (top-btm>=B){
				a[bcnt]=i;
				while (top>btm) blg[stk[--top]]=bcnt;
				++bcnt;
			}
		}
	stk[top++]=i;
}
int main(){
	scanf("%d%d",&n,&B);
	ecnt=ep; memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		addedge(u,v); addedge(v,u);
	}
	btm=top=0;
	dfs(-1,0);
	while (top) blg[stk[--top]]=bcnt-1;
	printf("%d\n",bcnt);
	for (int i=0; i<n; ++i) printf((i==n-1)?"%d\n":"%d ",blg[i]+1);
	for (int i=0; i<bcnt; ++i) printf((i==bcnt-1)?"%d\n":"%d ",a[i]+1);
	return 0;
}
