bzoj1087题解

Jun 29, 2016 10:02 · 124 words · 1 minute read 题解 bzoj DP 状态压缩

##bzoj100题留念。

直接状压dp就可以了，不过第一次写状压dp压力还是挺大的。

假设相邻两行的状态为s,s2，可以用位运算来判断是否有跨行的攻击：

if (!(s&s2) && !((s>>1)&s2) && !((s<<1)&s2))

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 19
#define maxs 530
ele n,k,f[maxn][maxn*maxn][maxs];
ele solve(ele i,ele j,ele s)
{
	if (i<0)
		if (!j && !s) return 1;
		else return 0;
	if (f[i][j][s]!=-1) return f[i][j][s];
	ele k=0,s1=s;
	for (int p=0; p<n-1; ++p)
	{
		k+=s1&1;
		if ((s1&1) && ((s1>>1)&1)) return f[i][j][s]=0;
		s1>>=1;
	}
	k+=s1;
	if (j<k) return f[i][j][s]=0;
	if (!j) return f[i][j][s]=1;
	f[i][j][s]=0;
	for (int s2=0; s2<(1<<n); ++s2)
		if (!(s&s2) && !((s>>1)&s2) && !((s<<1)&s2))
			f[i][j][s]+=solve(i-1,j-k,s2);
	return f[i][j][s];
}
int main()
{
	scanf("%lld%lld",&n,&k);
	for (int i=0; i<n; ++i)
		for (int j=0; j<=k; ++j)
			for (int s=0; s<(1<<n); ++s)
				f[i][j][s]=-1;
	ele ans=0;
	for (int s=0; s<(1<<n); ++s)
		ans+=solve(n-1,k,s);
	printf("%lld\n",ans);
	return 0;
}
