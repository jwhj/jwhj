hdu6017题解

Feb 25, 2017 10:02 · 169 words · 1 minute read 题解 hdu DP st表

比赛的时候没调出来，我还是太弱……

首先，其实就是要移动序列中的2，使得移动次数不超过$\lfloor\frac{m}{2}\rfloor$，且后面接着至少两个3的2的数量尽可能多。

让f[i][j][k]表示将前i个2分散在前j-1位里，且下一个2在第j位，移动次数上限为k时，后面接着至少两个3的2的数量的最大值。

考虑i的位置，如果p[i]>=j-3，那么不应该将其移动到j-3的左边，这是因为从j-3继续往左移的话，移动次数增多，左边留有的空间变小，答案不会变得更优。

如果p[i]<j-3，那么不应该将其左移。如果i被移到j-1或j-2，单独计算一下，否则假设其右移t位，此时的个数为f[i-1][p[i]+t][k-t]+1。

如果按照上面的方法转移，时间复杂度是四次的。注意到在f[i-1][p[i]+t][k-t]中，p[i]+t和k-t的和是一个定值，所以可以对f[i-1][j][k]中的每一个j+k维护一个st表，转移的时候就能$O(1)$地查到[i-1][p[i]+t][k-t]的最大值。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 210
#define maxk 9
#define INF 1000000000
ele LOG[maxn];
struct ST{
	inline void init(ele n){
		for (int i=0; i<n; ++i) f[i][0]=a[i];
		for (int j=1; j<maxk; ++j)
			for (int i=0; i<n; ++i)
				f[i][j]=max(f[i][j-1],f[min(n-1,i+(1<<j-1))][j-1]);
	}
	inline ele qry(ele l,ele r){
		ele L=LOG[r-l+1];
		return max(f[l][L],f[r-(1<<L)+1][L]);
	}
	ele a[maxn],f[maxn][maxk];
};
ele n,m,n1,p[maxn],f[maxn][maxn][maxn];
char s[maxn];
ST st[maxn<<1];
inline ele ABS(ele x){
	return (x<0)?(-x):(x);
}
int main(){
	ele T;
	scanf("%d",&T);
	LOG[1]=0;
	for (int i=2; i<maxn; ++i) LOG[i]=LOG[i>>1]+1;
	while (T--){
		scanf("%d%d%s",&n,&m,s);
		m>>=1;
		n1=0;
		for (int i=0; i<n; ++i)
			if (s[i]=='2') p[++n1]=i;
		memset(f[0],0,sizeof(f[0]));
		for (int i=1; i<=n1; ++i){
			for (int p=0; p<=n+m; ++p){
				for (int k=max(0,p-n); k<=p && k<=m; ++k)
					st[p].a[k]=f[i-1][p-k][k];
				st[p].init(p+1);
			}
			for (int j=0; j<=n; ++j)
				for (int k=0; k<=m; ++k){
					f[i][j][k]=-INF;
					if (j-1>=0 && ABS(j-1-p[i])<=k)
						f[i][j][k]=max(f[i][j][k],f[i-1][j-1][k-ABS(j-1-p[i])]);
					if (j-2>=0 && ABS(j-2-p[i])<=k)
						f[i][j][k]=max(f[i][j][k],f[i-1][j-2][k-ABS(j-2-p[i])]);
					if (p[i]>=j-3){
						if (j-3>=0 && ABS(j-3-p[i])<=k)
							f[i][j][k]=max(f[i][j][k],f[i-1][j-3][k-ABS(j-3-p[i])]+1);
					}
					else{
						ele t=p[i]+k;
						ele L=0,R=min(k,j-3-p[i]);
						f[i][j][k]=max(f[i][j][k],st[t].qry(k-R,k-L)+1);
					}
				}
		}
		printf("%d\n",f[n1][n][m]);
	}
	return 0;
}
