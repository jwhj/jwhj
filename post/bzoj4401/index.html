首先，每块的大小确定的时候，分割的方式最多一种。网上的题解都没有证明这一点，我也不会证。

假设每块的大小为k。把树变成有根树，如果一个节点i作为某块的根，那么以它为根的子树的大小（记为size[i]）必然是k的倍数，因为它为根的子树必然包含整数个块。

假设我们已经知道了size[i]为k的倍数的i的个数，这个个数少于n/k，就不存在分割方法，否则存在。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 1000010
struct edge{
	ele v;
	edge *nxt;
}ep[maxn<<1];
ele n,ecnt,size[maxn],cnt[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v){
	edge *p=&ep[ecnt++];
	p->v=v; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele p,ele i){
	size[i]=1;
	for (edge *j=h[i]; j; j=j->nxt)
		if (j->v!=p){
			dfs(i,j->v);
			size[i]+=size[j->v];
		}
	++cnt[size[i]];
}
inline ele calc(ele k){
	ele tmp=0;
	for (int i=k; i<=n; i+=k)
		tmp+=cnt[i];
	if (tmp>=n/k)
		return 1;
	return 0;
}
int main(){
	scanf("%d",&n);
	ecnt=0;
	memset(h,0,sizeof(h));
	for (int i=0; i<n-1; ++i){
		ele u,v;
		scanf("%d%d",&u,&v);
		--u,--v;
		addedge(u,v); addedge(v,u);
	}
	memset(cnt,0,sizeof(cnt));
	dfs(-1,0);
	ele ans=0;
	for (int i=1; i*i<=n; ++i)
		if (n%i==0){
			ans+=calc(i);
			if (i*i!=n)
				ans+=calc(n/i);
		}
	printf("%d\n",ans);
	return 0;
}
