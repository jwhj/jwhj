bzoj1037题解

Aug 9, 2017 10:02 · 104 words · 1 minute read 题解 bzoj 记忆化搜索

去掉一个合法排列的最后一个人（假设是男孩），剩下的人也应该满足题目条件，即对于任意连续的一段，男孩与女孩的数目之 差不超过k。除了这之外，剩下的人还应该满足一个额外的条件，即对于任意一个后缀，男孩数减女孩数在-k到k-1之间。

所以我们可以这样表示状态：(i,j,s,t)代表i个男孩，j个女孩排成一排，满足题目条件，并且对于任意一个后缀，男孩数减女孩数在s到t之间的方案数。转移挺简单的，看代码吧。

直接这么dp还是不行的，因为会MLE。但是注意到很多状态实际上是用不到的，所以可以改成记忆化搜索并用散列表/map来保存状态。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <map>
#define ele int
using namespace std;
#define maxn 155
#define maxk 25
#define MOD 12345678
struct st{
	ele i,j,s,t;
	inline bool operator<(st b)const{
		if (i!=b.i) return i<b.i;
		if (j!=b.j) return j<b.j;
		if (s!=b.s) return s<b.s;
		return t<b.t;
	}
};
ele n,m,k;
map<st,ele> f;
ele solve(ele i,ele j,ele s,ele t){
	if (f.find((st){i,j,s,t})!=f.end()) return f[(st){i,j,s,t}];
	ele ans=0;
	if (i && s<=1 && 1<=t) (ans+=solve(i-1,j,max(s-1,-k),max(t-1,-k)))%=MOD;
	if (j && s<=-1 && -1<=t) (ans+=solve(i,j-1,min(s+1,k),min(t+1,k)))%=MOD;
	f[(st){i,j,s,t}]=ans;
	return ans;
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for (int i=-k; i<=0; ++i)
		for (int j=0; j<=k; ++j) f[(st){0,0,i,j}]=1;
	printf("%d\n",solve(n,m,-k,k));
	return 0;
}
