bzoj2671题解

Jan 27, 2017 10:02 · 105 words · 1 minute read 题解 bzoj 数学 mobius

懒癌发作不写题解了……膜popoqqq的题解。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#define ele int
#define ll long long
using namespace std;
#define maxb 50010
ele n,pcnt,plist[maxb],mu[maxb];
bool flag[maxb];
inline ele calc(ele k,ele j,ele i){
	return mu[k]*(i/k-j/k);
}
int main(){
	scanf("%d",&n);
	ele lim=(ele)(sqrt(n)+1);
	pcnt=0;
	mu[1]=1;
	memset(flag,0,sizeof(flag));
	for (int i=2; i<lim; ++i){
		if (!flag[i])
			plist[pcnt++]=i,mu[i]=-1;
		for (int j=0; j<pcnt && i*plist[j]<lim; ++j){
			flag[i*plist[j]]=true;
			if (i%plist[j])
				mu[i*plist[j]]=-mu[i];
			else{
				mu[i*plist[j]]=0;
				break;
			}
		}
	}
	ll ans=0;
	for (int b=1; (ll)b*b<n; ++b){
		ele m=n/b;
		for (int i=b+b-1; i>b;){
			ele j=m/(m/i+1),tmp=0;
			j=max(j,b);
			for (int k=1; k*k<=b; ++k)
				if (b%k==0){
					tmp+=calc(k,j-b,i-b);
					if (k*k!=b)
						tmp+=calc(b/k,j-b,i-b);
				}
			ans+=(ll)tmp*(m/i);
			i=j;
		}
	}
	printf("%lld\n",ans);
	return 0;
}
