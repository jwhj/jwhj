圆方树太神啦！

对于仙人掌中原来的点，建立一个圆点。对于每一个环，建立一个方点，然后把环上所有的点对应的圆点连到这个方点上面。对于其它的边，连接端点对应的圆点。这样就得到了仙人掌对应的圆方树。

圆方树有一些很好的性质，比如说：

  • 它是一棵树
  • 不管怎么换根，圆方树的形态不变化

对于树中所有圆圆边，边权为其在仙人掌中对应的边的边权。对于方圆/圆方边，边权为圆点对应的点到它所在环的最高点的最短路的长度。

考虑求u，v之间的最短路长度。若u，v在圆方树中的lca为圆点，两点在仙人掌中最短路的长度即为它们在圆方树中的路径长度。若u，v的lca为方点，说明它们在仙人掌上的lca是一个环。上一句话意会一下就可以了。那么就要考虑一下在lca那个环上怎么走。对于每条方圆边，还要保存一下它在仙人掌上对应的路径是顺时针还是逆时针的。

if (od[pu]!=od[pv])打成了if (od[pu]==od[pv])结果调了好久……我太弱了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 20010
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxn<<3],*ecnt;
ele n,m,Q,tcnt,top,scnt,sqcnt,stk[maxn],dfn[maxn],low[maxn],
d[maxn],D[maxn],tmp[maxn],size[maxn],seq[maxn],fa[maxn],dep[maxn],pos[maxn],r[maxn];
bool od[maxn];
edge *h[maxn],*h1[maxn],*hc[maxn];
inline ele _abs(ele x){
	return (x<0)?-x:x;
}
inline void addedge1(edge **h,ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
inline void addedge(edge **h,ele u,ele v,ele w){
	addedge1(h,u,v,w); addedge1(h,v,u,w);
}
void tarjan(ele p,ele i){
	low[i]=dfn[i]=tcnt++;
	stk[top++]=i;
	for (edge *j=h[i]; j; j=j->nxt){
		if (!~dfn[j->v]){
			d[j->v]=d[i]+j->w;
			tarjan(i,j->v);
			low[i]=min(low[i],low[j->v]);
			if (dfn[i]==low[j->v]){
				D[sqcnt]=d[stk[top-1]]-d[i]+tmp[stk[top-1]];
				addedge(h1,i,sqcnt,0);
				ele k;
				do{
					k=stk[top-1];
					--top;
					addedge(h1,sqcnt,k,min(d[k]-d[i],D[sqcnt]-d[k]+d[i]));
					od[k]=((d[k]-d[i])*2<=D[sqcnt])?0:1;
				}while (k!=j->v);
				sqcnt++;
			}
			else if (dfn[i]<low[j->v]) addedge(h1,i,j->v,j->w),--top;
		}
		else if (j->v!=p)
			if (dfn[j->v]<low[i]){
				low[i]=dfn[j->v];
				tmp[i]=j->w;
			}
	}
}
void dfs1(ele p,ele i){
	fa[i]=p;
	hc[i]=NULL;
	size[i]=1;
	for (edge *j=h1[i]; j; j=j->nxt)
		if (j->v!=p){
			d[j->v]=d[i]+j->w;
			dep[j->v]=dep[i]+1;
			dfs1(i,j->v);
			size[i]+=size[j->v];
			if (!hc[i] || size[hc[i]->v]<size[j->v]) hc[i]=j;
		}
}
void dfs2(ele p,ele i){
	pos[i]=scnt;
	seq[scnt++]=i;
	if (hc[i]) r[hc[i]->v]=r[i],dfs2(i,hc[i]->v);
	for (edge *j=h1[i]; j; j=j->nxt)
		if (j->v!=p && j!=hc[i]){
			r[j->v]=j->v;
			dfs2(i,j->v);
		}
}
inline ele lca(ele u,ele v){
	while (r[u]!=r[v]){
		if (dep[r[u]]>dep[r[v]]) swap(u,v);
		v=fa[r[v]];
	}
	return (dep[u]<dep[v])?u:v;
}
inline ele calc(ele x,ele w){
	while (fa[x]!=w){
		if (dep[w]<dep[r[x]]){
			x=r[x];
			if (fa[x]!=w) x=fa[x];
		}
		else x=seq[pos[w]+1];
	}
	return x;
}
inline ele qry(ele u,ele v){
	ele w=lca(u,v);
	if (w<n) return d[u]+d[v]-(d[w]<<1);
	ele pu=calc(u,w),pv=calc(v,w);
	ele l1=d[pu]-d[w],l2=d[pv]-d[w],L=d[u]+d[v]-(d[w]<<1)-l1-l2;;
	if (od[pu]!=od[pv]) return L+min(l1+l2,D[w]-l1-l2);
	else return L+min(_abs(l2-l1),D[w]-_abs(l2-l1));
}
int main(){
	scanf("%d%d%d",&n,&m,&Q);
	ecnt=ep; memset(h,0,sizeof(h)); memset(h1,0,sizeof(h1));
	for (int i=0; i<m; ++i){
		ele u,v,w;
		scanf("%d%d%d",&u,&v,&w); --u,--v;
		addedge(h,u,v,w);
	}
	sqcnt=n; top=0; tcnt=0;
	d[0]=0;
	memset(dfn,-1,sizeof(dfn));
	tarjan(-1,0);
	d[0]=0; dep[0]=0;
	dfs1(-1,0);
	scnt=0;
	dfs2(-1,0);
	while (Q--){
		ele u,v;
		scanf("%d%d",&u,&v); --u,--v;
		printf("%d\n",qry(u,v));
	}
	return 0;
}
