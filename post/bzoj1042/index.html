bzoj1042题解

Aug 8, 2017 10:02 · 86 words · 1 minute read 题解 bzoj 背包 容斥

看到题目第一个想法肯定是多重背包，但是这么做会超时。需要注意的是，只能判断可行性的O(VN)做法似乎不能用来计算方法总数，所以也不行。

再观察一下题目，会发现货币的种类很少，有很多个询问但是每个询问的货币种类是一样的。所以可以考虑容斥，也就是毫无限制的总方法数-1种超限的总方法数+2种超限的总方法数-3种超限的总方法数+4种超限的总方法数。

上面式子中每一项都可以用完全背包算出来，而对于每个询问货币种类一样，所以预处理一次完全背包就够了，不需要每次询问都算一次。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 5
#define maxs 100010
#define v calc
const ele INF=1e9;
ele tot,c[maxn],d[maxn],s,f[maxs];
inline ele calc(ele i,ele j,ele k,ele l){
	ele p=s-(d[0]+1)*c[0]*i-(d[1]+1)*c[1]*j-(d[2]+1)*c[2]*k-(d[3]+1)*(c[3])*l;
	return (p>=0)?f[p]:0;
}
int main(){
	for (int i=0; i<4; ++i) scanf("%lld",c+i);
	scanf("%lld",&tot);
	memset(f,0,sizeof(f));
	f[0]=1;
	for (int i=0; i<4; ++i)
		for (int j=c[i]; j<=maxs; ++j) f[j]+=f[j-c[i]];
	while (tot--){
		for (int i=0; i<4; ++i) scanf("%lld",d+i);
		scanf("%lld",&s);
		ele ans=
		v(0,0,0,0)
		-v(0,0,0,1)-v(0,0,1,0)-v(0,1,0,0)-v(1,0,0,0)
		+v(0,0,1,1)+v(0,1,0,1)+v(0,1,1,0)+v(1,0,0,1)+v(1,0,1,0)+v(1,1,0,0)
		-v(0,1,1,1)-v(1,0,1,1)-v(1,1,0,1)-v(1,1,1,0)
		+v(1,1,1,1);
		printf("%lld\n",ans);
	}
	return 0;
}
