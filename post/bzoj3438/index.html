bzoj3438题解

Mar 2, 2017 10:02 · 223 words · 2 minutes read 题解 bzoj 网络流

像这种题，有一堆东西，每个东西有两种选择，分别会有一定的贡献，一些选择的组合又会有一定的贡献的，可以用最小割模型来做。

对于此题，从s到每个作物连一条权为ai的边，每个作物到t连一条权为bi的边。如果割掉s到i的边，说明i种在B，否则说明i种在A。

如果一些作物都种在A会有额外收益c，那么新建一个点x，s到x连一条权为c的边，x到那些作物分别连一条权为INF的边。这样的话，如果这些作物中有一个i，它到t的边没有被割掉，那么s到x的边必须被割掉。于是所有没割掉的边的权值之和即为总收益，求出这个图的最小割，并用所有可能出现的收益的和减掉最小割就可以得到答案了。

据说这个模型可以解决未知数只能取0/1的线性规划？晚点研究一下。还是打单纯形吧。

写了一个dinic结果TLE了好几发，跟别人写的对比了之后发现我写得还是过于naive……有好多可以优化的地方。

代码：

#include <cstdio>
#include <cstring>
#include <cctype>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 3010
#define INF 1000000000
struct edge{
	ele v,r;
	edge *nxt,*rev;
}ep[2000000];
ele n,m,ecnt,dep[maxn];
edge *g[maxn],*cur[maxn];
inline void get(ele&a){
	char c;
	while (!isdigit(c=getchar()));
	a=c-'0';
	while (isdigit(c=getchar())) a=(a<<3)+(a<<1)+c-'0';
}
inline void addedge1(ele u,ele v,ele w){
	edge *p=&ep[ecnt++];
	p->v=v; p->r=w; p->nxt=g[u]; p->rev=ep+((ecnt-1)^1);
	g[u]=p;
}
inline void addedge(ele u,ele v,ele w){
	addedge1(u,v,w); addedge1(v,u,0);
}
inline bool bfs(ele s,ele t){
	static ele Q[maxn];
	ele qh=0,qt=1;
	Q[0]=s;
	memset(dep,0,sizeof(dep));
	dep[s]=1;
	while (qh<qt){
		for (edge *j=g[Q[qh]]; j; j=j->nxt)
			if (!dep[j->v] && j->r){
				dep[j->v]=dep[Q[qh]]+1;
				Q[qt++]=j->v;
				if (j->v==t) return true;
			}
		++qh;
	}
	return false;
}
ele dfs(ele i,ele t,ele p){
	if (i==t || !p) return p;
	ele ans=0;
	for (edge*&j=cur[i]; j; j=j->nxt)
		if (dep[i]+1==dep[j->v] && j->r){
			ele tmp=dfs(j->v,t,min(p,j->r));
			if (tmp){
				ans+=tmp;
				j->r-=tmp;
				j->rev->r+=tmp;
				p-=tmp;
				if (!p) break;
			}
		}
	if (p) dep[i]=-1;
	return ans;
}
inline ele dinic(ele s,ele t){
	ele ans=0;
	while (bfs(s,t)){
		memcpy(cur,g,sizeof(edge*)*(2+n+m+m));
		ans+=dfs(s,t,INF);
	}
	return ans;
}
int main(){
	get(n);
	ecnt=0;
	memset(g,0,sizeof(g));
	ele sum=0;
	for (int i=0; i<n; ++i){
		ele x;
		get(x);
		addedge(0,2+i,x);
		sum+=x;
	}
	for (int i=0; i<n; ++i){
		ele x;
		get(x);
		addedge(2+i,1,x);
		sum+=x;
	}
	scanf("%d",&m);
	for (int p=0; p<m; ++p){
		ele k,c1,c2;
		get(k); get(c1); get(c2);
		addedge(0,2+n+p*2,c1);
		addedge(2+n+p*2+1,1,c2);
		sum+=c1+c2;
		for (int i=0; i<k; ++i){
			ele j;
			get(j); --j;
			addedge(2+n+p*2,2+j,INF);
			addedge(2+j,2+n+p*2+1,INF);
		}
	}
	printf("%d\n",sum-dinic(0,1));
	return 0;
}
