bzoj2115题解

May 30, 2017 10:02 · 155 words · 1 minute read 题解 bzoj 线性基

任何一条从1到n的路径，异或上若干个环，就可以得到另外任何一条1到n的路径。

另一方面，假设我们已经有了一条路径，到达n后，再走到某个环上，绕环一圈，再回到n，就可以得到这个环的权值。

所以，我们只需要随便找一条路径，求出上面边权的异或和s，再把所有环找出来，让s异或上其中一些，使得权值最大就可以了。

求出环的权值的线性基，从大到小枚举，如果异或上这个数可以使答案更大，那就异或。利用线性基的性质，很容易从二进制的角度证明这个贪心的正确性。

找出所有的环可以像tarjan那样dfs，找出回边。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 50010
#define maxm 100010
#define W 64
struct edge{
	ele v,w;
	edge *nxt,*rev;
}ep[maxm<<1],*ecnt;
ele n,m,b[W],d[maxn];
edge *h[maxn];
bool vis[maxn];
inline void addedge(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u]; p->rev=ep+((ecnt-ep-1)^1);
	h[u]=p;
}
inline void add(ele x){
	for (int i=W-1; ~i; --i)
		if (x>>i&1){
			if (b[i]) x^=b[i];
			else{
				b[i]=x;
				for (int j=i-1; ~j; --j)
					if (x>>j) x^=b[j];
				for (int j=i+1; j<W; ++j)
					if (b[j]>>i) b[j]^=x;
				break;
			}
		}
}
void dfs(edge *p,ele i){
	vis[i]=true;
	for (edge *j=h[i]; j; j=j->nxt)
		if (!vis[j->v]){
			d[j->v]=d[i]^j->w;
			dfs(j->rev,j->v);
		}
		else if (j!=p){
			ele t=d[i]^d[j->v]^j->w;
			add(t);
		}
}
int main(){
	scanf("%lld%lld",&n,&m);
	ecnt=ep; memset(h,0,sizeof(h));
	for (int i=0; i<m; ++i){
		ele u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w); --u,--v;
		addedge(u,v,w); addedge(v,u,w);
	}
	memset(vis,0,sizeof(vis));
	memset(b,0,sizeof(b));
	for (int i=0; i<n; ++i)
		if (!vis[i]) d[i]=0,dfs(NULL,i);
	ele ans=d[n-1];
	for (int i=W-1; ~i; --i)
		if ((ans^b[i])>ans) ans^=b[i];
	printf("%lld\n",ans);
	return 0;
}
