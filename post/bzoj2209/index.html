bzoj2209题解

Feb 1, 2017 10:02 · 231 words · 2 minutes read 题解 bzoj 括号序列 treap 统计

把左括号看成-1，右括号看成1，那么一个区间中，前缀和最大值即为未匹配的右括号个数，后缀和最小值的相反数即为未匹配的左括号的个数。

设两者分别为a，b，如果$2|a,b$，那么答案为$\frac{a+b}{2}$，否则答案为$\frac{a+b}{2}+1$。

有一个细节需要注意，如果前缀和最大值为负，则右括号全部匹配完，如果后缀和最小值为正，则左括号全部匹配完。

代码：

#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
struct ANS{
	ele pmx,pmn,smx,smn,sum;
};
inline ANS operator+(ANS a,ANS b){
	ANS c;
	c.pmx=max(a.pmx,a.sum+b.pmx);
	c.pmn=min(a.pmn,a.sum+b.pmn);
	c.smx=max(b.smx,a.smx+b.sum);
	c.smn=min(b.smn,a.smn+b.sum);
	c.sum=a.sum+b.sum;
	return c;
}
inline void neg(ANS&a){
	swap(a.pmx,a.pmn);
	a.pmx*=-1; a.pmn*=-1;
	swap(a.smx,a.smn);
	a.smx*=-1; a.smn*=-1;
	a.sum*=-1;
}
inline void flip(ANS&a){
	swap(a.pmx,a.smx);
	swap(a.pmn,a.smn);
}
struct node{
	ANS a;
	ele p,x,s;
	bool neg,flip;
	node *l,*r;
}np[maxn];
inline void pushdown(node *x){
	if (x->neg){
		neg(x->a);
		x->x*=-1;
		if (x->l)
			x->l->neg^=1;
		if (x->r)
			x->r->neg^=1;
		x->neg=false;
	}
	if (x->flip){
		flip(x->a);
		swap(x->l,x->r);
		if (x->l)
			x->l->flip^=1;
		if (x->r)
			x->r->flip^=1;
		x->flip=false;
	}
}
inline void maintain(node *x){
	x->a={x->x,x->x,x->x,x->x,x->x};
	x->s=1;
	if (x->l)
		pushdown(x->l),x->a=x->l->a+x->a,x->s+=x->l->s;
	if (x->r)
		pushdown(x->r),x->a=x->a+x->r->a,x->s+=x->r->s;
}
node* merge(node *a,node *b){
	if (!a) return b;
	if (!b) return a;
	pushdown(a); pushdown(b);
	if (a->p>b->p){
		a->r=merge(a->r,b);
		maintain(a);
		return a;
	}
	else{
		b->l=merge(a,b->l);
		maintain(b);
		return b;
	}
}
void split(node *x,ele k,node*&a,node*&b){
	if (!x){
		a=b=NULL;
		return;
	}
	pushdown(x);
	ele s=(x->l)?(x->l->s+1):(1);
	if (k<s){
		split(x->l,k,a,b);
		x->l=b;
		maintain(x);
		b=x;
	}
	else if (k>s){
		split(x->r,k-s,a,b);
		x->r=a;
		maintain(x);
		a=x;
	}
	else{
		a=x; b=x->r;
		x->r=NULL;
		maintain(x);
	}
}
ele n,q;
char s[maxn];
node *root;
int main(){
	srand(2333);
	scanf("%d%d%s",&n,&q,s);
	root=NULL;
	for (int i=0; i<n; ++i){
		np[i].x=(s[i]=='(')?(-1):(1);
		np[i].p=rand();
		np[i].neg=np[i].flip=false;
		np[i].l=np[i].r=NULL;
		maintain(np+i);
		root=merge(root,np+i);
	}
	while (q--){
		ele op,u,v;
		scanf("%d%d%d",&op,&u,&v);
		node *a,*b,*c;
		split(root,v,a,c);
		split(a,u-1,a,b);
		if (op==0){
			ele x=max(b->a.pmx,0);
			ele y=-min(b->a.smn,0);
			if (x%2)
				printf("%d\n",(x+y)/2+1);
			else printf("%d\n",(x+y)/2);
		}
		else if (op==1)
			b->neg^=1;
		else if (op==2)
			b->flip^=1;
		root=merge(a,b);
		root=merge(root,c);
	}
	return 0;
}
