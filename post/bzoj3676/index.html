bzoj3676题解

Jan 27, 2017 10:02 · 210 words · 1 minute read 题解 bzoj 后缀数据结构 manacher

首先，串中至多有n种本质上不同的回文子串，即manacher执行过程中，每次回文子串向右延伸的最远位置变化的时候，才会出现本质上不同的回文子串。

于是我们可以枚举这些子串，并求出它们出现的次数。

使用后缀自动机来解决这个问题。通过dp可以求出后缀自动机每个节点对应的字符串的出现次数。所以对于某个子串，只需求出它在后缀自动机里面对应的节点就可以了。假设这个子串是s[l~r]，首先求出s[0~r]对应的子串，然后沿着后缀边走。我不知道这个做法时间复杂度对不对，反正AC了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define ll long long
using namespace std;
#define maxn 300010
#define alpha 26
struct node{
    ele len,c;
    node *nxt,*f,*l[alpha];
    inline void alloc(){
        f=NULL;
        memset(l,0,sizeof(l));
    }
}np[maxn<<1];
ele ncnt;
node *h[maxn];
struct SAM{
    inline void onnewnode(node *x){
        x->nxt=h[x->len];
        h[x->len]=x;
    }
    inline void alloc(){
        ncnt=0;
        root=&np[ncnt++];
        root->alloc();
        root->len=0;
        root->c=1;
        onnewnode(root);
        end=root;
    }
    inline node* add(ele c){
        node *p=&np[ncnt++],*q;
        p->alloc(); p->len=end->len+1; p->c=1;
        onnewnode(p);
        for (q=end; q && !q->l[c]; q=q->f)
            q->l[c]=p;
        if (!q)
            p->f=root;
        else if (q->len+1==q->l[c]->len)
            p->f=q->l[c];
        else{
            node *r=&np[ncnt++],*buf=q->l[c];
            *r=*buf;
            r->len=q->len+1;
            r->c=0;
            onnewnode(r);
            buf->f=p->f=r;
            for (; q && q->l[c]==buf; q=q->f)
                q->l[c]=r;
        }
        return end=p;
    }
    node *root,*end;
};
ele n,p[maxn<<1];
char s[maxn<<1],s1[maxn];
SAM sam;
node *seq[maxn];
int main(){
    scanf("%s",s1);
    n=strlen(s1);
    memset(h,0,sizeof(h));
    sam.alloc();
    for (int i=0; i<n; ++i)
        seq[i]=sam.add(s1[i]-'a');
    for (int i=sam.end->len; i>0; --i)
        for (node *j=h[i]; j; j=j->nxt)
            j->f->c+=j->c;
    ele m=0;
    s[m++]='!';
    for (int i=0; i<n; ++i){
        s[m++]='#';
        s[m++]=s1[i];
    }
    s[m++]='#';
    s[m++]='@';
    ele mx=0,id=0;
    p[0]=0;
    ll ans=0;
    for (int i=1; i<m; ++i){
        if (i>mx)
            p[i]=0;
        else p[i]=min(mx-i,p[id*2-i]);
        for (; s[i-p[i]-1]==s[i+p[i]+1]; ++p[i])
            ;
        if (i+p[i]>mx){
            for (int j=mx+1; j<=i+p[i]; ++j)
                if ('a'<=s[j] && s[j]<='z'){
                    ele k=i*2-j;
                    ele j1=j/2-1;
                    k=k/2-1;
                    ele L=j1-k+1;
                    node *p=seq[j1];
                    while (L<=p->f->len)
                        p=p->f;
                    ans=max(ans,(ll)p->c*L);
                }
            mx=i+p[i];
            id=i;
        }
    }
    printf("%lld\n",ans);
    return 0;
}
