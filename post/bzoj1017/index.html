bzoj1017题解

May 12, 2017 10:02 · 243 words · 2 minutes read 题解 bzoj DP

让f[i][j][k]代表只考虑以i为根的子树，i至少造j个，花费不超过k的最大收益。

转移的时候需要枚举每一棵子树分到多少金币，直接暴力枚举肯定是不可以的，观察一下会发现，这个子问题其实就是一个类似背包问题的的东西，可以让g[i][j]代表前i棵子树，花费不超过j的最大收益，然后DP。

细节还是比较多的……需要仔细调一下。

似乎还需要想办法卡一下常数才能AC。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 55
#define maxm 2010
#define maxl 110
const ele INF=1e9;
struct edge{
	ele v,w;
	edge *nxt;
}ep[maxn],*ecnt;
ele n,m,scnt,a[maxn],b[maxn],p[maxn],ind[maxn],seq[maxn],f[maxn][maxl][maxm],g[maxm],delta[maxn];
bool lf[maxn];
edge *h[maxn];
inline void addedge(ele u,ele v,ele w){
	edge *p=ecnt++;
	p->v=v; p->w=w; p->nxt=h[u];
	h[u]=p;
}
void dfs(ele i){
	seq[scnt++]=i;
	delta[i]=p[i];
	if (h[i]) b[i]=INF;
	for (edge *j=h[i]; j; j=j->nxt){
		dfs(j->v);
		delta[i]-=p[j->v]*j->w;
		a[i]+=a[j->v]*j->w;
		b[i]=min(b[i],b[j->v]/j->w);
	}
	b[i]=min(b[i],m/a[i]);
}
int main(){
	scanf("%d%d",&n,&m);
	ecnt=ep;
	memset(ind,0,sizeof(ind));
	bool flag=false;
	for (int i=0; i<n; ++i){
		char s[10];
		scanf("%d%s",p+i,s);
		if (s[0]=='B') scanf("%d%d",a+i,b+i),lf[i]=true,b[i]=min(b[i],m/a[i]);
		else{
			ele c;
			scanf("%d",&c);
			for (int j=0; j<c; ++j){
				ele v,w;
				scanf("%d%d",&v,&w); --v;
				addedge(i,v,w);
				++ind[v];
			}
			b[i]=maxl-1;
			lf[i]=false;
			flag=true;
		}
	}
	if (flag){
		scnt=0;
		ele r;
		for (int i=0; i<n; ++i)
			if (!ind[i]) dfs(r=i);
		for (int i=0; i<n; ++i)
			for (int j=0; j<maxl; ++j)
				for (int k=0; k<maxm; ++k) f[i][j][k]=-INF;
		for (int i=n-1; ~i; --i){
			ele x=seq[i];
			if (lf[x]){
				for (int j=0; j<=b[x]; ++j) f[x][j][j*a[x]]=p[x]*j;
			}
			else{
				for (int j=0; j<=b[x]; ++j){
					memset(g,0,sizeof(g));
					for (edge *p=h[x]; p; p=p->nxt){
						for (int k=0; k<=m; ++k){
							f[x][j][k]=-INF;
							for (int r=0; r<=k; ++r)
								if (j*p->w<maxl)
									f[x][j][k]=max(f[x][j][k],g[k-r]+f[p->v][j*p->w][r]);
						}
						memcpy(g,f[x][j],sizeof(ele)*(m+1));
					}
					for (int k=0; k<=m; ++k) f[x][j][k]+=delta[x]*j;
				}
			}
			for (int k=0; k<=m; ++k)	
				for (int j=b[x]-1; ~j; --j) f[x][j][k]=max(f[x][j][k],f[x][j+1][k]);
		}
		printf("%d\n",f[r][0][m]);
	}
	else{
		static ele f[maxm],g[maxm];
		memset(g,0,sizeof(g));
		for (int i=0; i<n; ++i){
			for (int j=0; j<=m; ++j){
				f[j]=-INF;
				for (int k=0; k<=j; ++k){
					ele c=min(b[i],k/a[i]);
					f[j]=max(f[j],g[j-k]+p[i]*c);
				}
			}
			memcpy(g,f,sizeof(ele)*(m+1));
		}
		printf("%d\n",f[m]);
	}
	return 0;
}
