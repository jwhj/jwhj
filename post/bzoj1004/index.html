bzoj1004题解

Dec 23, 2016 10:02 · 145 words · 1 minute read 题解 bzoj 数学 置换群

不会置换群的先去学一下吧，最好找本书看一看，网上很多地方没讲清楚，我比较懒不想写上来。

题目说，输入数据保证任意多次洗牌都可用这m种洗牌法中的一种代替，这说明输入的置换构成一个置换群。

因为三种颜色的使用次数有限制，所以不能用polya定理，要用burnside引理。

求每个置换不变元的个数时，先把置换拆成若干个循环。要在置换下不变，每个循环中必须全部同色，所以可以用背包来做。

最后的除法用逆元来处理。

代码：

#include <cstdio>
#include <cstring>
#define ele int
using namespace std;
#define maxn 70
ele n,m,s1,s2,s3,p,X[maxn];
inline void exgcd(ele a,ele b,ele&x,ele&y){
	if (!b){
		x=1;
		y=0;
		return;
	}
	exgcd(b,a%b,y,x);
	y-=(a/b)*x;
}
inline ele inv(ele a,ele p){
	ele x,y;
	exgcd(a,p,x,y);
	return ((x%=p)+=p)%=p;
}
inline ele dp(){
	ele h=0;
	static ele d[maxn];
	static bool vis[maxn];
	memset(vis,0,sizeof(vis));
	memset(d,0,sizeof(d));
	for (int i=0; i<n; ++i)
		if (!vis[i]){
			int j=i;
			while (!vis[j]){
				vis[j]=true;
				++d[h];
				j=X[j];
			}
			++h;
		}
	static ele f[maxn][maxn][maxn];
	f[0][0][0]=1;
	for (int i=0; i<h; ++i)
		for (int u=s1; u>=0; --u)
			for (int v=s2; v>=0; --v)
				for (int w=s3; w>=0; --w){
					f[u][v][w]=0;
					if (u>=d[i])
						(f[u][v][w]+=f[u-d[i]][v][w])%=p;
					if (v>=d[i])
						(f[u][v][w]+=f[u][v-d[i]][w])%=p;
					if (w>=d[i])
						(f[u][v][w]+=f[u][v][w-d[i]])%=p;
				}
	return f[s1][s2][s3];
}
int main(){
	scanf("%d%d%d%d%d",&s1,&s2,&s3,&m,&p);
	n=s1+s2+s3;
	ele ans=0;
	for (int i=0; i<m; ++i){
		for (int j=0; j<n; ++j)
			scanf("%d",X+j),--X[j];
		(ans+=dp())%=p;
	}
	for (int i=0; i<n; ++i)
		X[i]=i;
	(ans+=dp())%=p;
	(ans*=inv(m+1,p))%=p;
	printf("%d\n",ans);
	return 0;
}
