bzoj3196题解

Jan 27, 2017 10:02 · 295 words · 2 minutes read 题解 bzoj 分块

分块，每块维护一个快内元素排好序的数组，就可以做了。

代码：

#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 50010
#define maxs 3500
#define maxc 20
#define maxa 100000000
struct block{
	ele n,a[maxs],b[maxs];
};
ele n,q,S,bcnt,a[maxn];
block b[maxc];
inline ele getrank(ele l,ele r,ele k){
	ele i1=l/S,j1=l%S,i2=r/S,j2=r%S,ans=1;
	if (i1==i2){
		for (int i=j1; i<=j2; ++i)
			if (b[i1].a[i]<k)
				++ans;
	}
	else{
		for (int i=j1; i<b[i1].n; ++i)
			if (b[i1].a[i]<k)
				++ans;
		for (int i=0; i<=j2; ++i)
			if (b[i2].a[i]<k)
				++ans;
		for (int i=i1+1; i<i2; ++i){
			ele p=lower_bound(b[i].b,b[i].b+b[i].n,k)-b[i].b;
			ans+=p;
		}
	}
	return ans;
}
inline void upd(ele p,ele k){
	ele i1=p/S,j1=p%S;
	ele q=lower_bound(b[i1].b,b[i1].b+b[i1].n,b[i1].a[j1])-b[i1].b;
	b[i1].b[q]=k;
	while (q>0 && b[i1].b[q-1]>b[i1].b[q])
		swap(b[i1].b[q-1],b[i1].b[q]),--q;
	while (q<b[i1].n-1 && b[i1].b[q]>b[i1].b[q+1])
		swap(b[i1].b[q],b[i1].b[q+1]),++q;
	b[i1].a[j1]=k;
}
inline ele prev(ele l,ele r,ele k){
	ele i1=l/S,j1=l%S,i2=r/S,j2=r%S,ans=-1;
	if (i1==i2){
		for (int i=j1; i<=j2; ++i)
			if (b[i1].a[i]<k && b[i1].a[i]>ans)
				ans=b[i1].a[i];
	}
	else{
		for (int i=j1; i<b[i1].n; ++i)
			if (b[i1].a[i]<k && b[i1].a[i]>ans)
				ans=b[i1].a[i];
		for (int i=0; i<=j2; ++i)
			if (b[i2].a[i]<k && b[i2].a[i]>ans)
				ans=b[i2].a[i];
		for (int i=i1+1; i<i2; ++i){
			ele p=lower_bound(b[i].b,b[i].b+b[i].n,k)-b[i].b;
			if (p>0)
				ans=max(ans,b[i].b[p-1]);
		}
	}
	return ans;
}
inline ele succ(ele l,ele r,ele k){
	ele i1=l/S,j1=l%S,i2=r/S,j2=r%S,ans=maxa+1;
	if (i1==i2){
		for (int i=j1; i<=j2; ++i)
			if (b[i1].a[i]>k && b[i1].a[i]<ans)
				ans=b[i1].a[i];
	}
	else{
		for (int i=j1; i<b[i1].n; ++i)
			if (b[i1].a[i]>k && b[i1].a[i]<ans)
				ans=b[i1].a[i];
		for (int i=0; i<=j2; ++i)
			if (b[i2].a[i]>k && b[i2].a[i]<ans)
				ans=b[i2].a[i];
		for (int i=i1+1; i<i2; ++i){
			ele p=upper_bound(b[i].b,b[i].b+b[i].n,k)-b[i].b;
			if (p<b[i].n)
				ans=min(ans,b[i].b[p]);
		}
	}
	return ans;
}
int main(){
	scanf("%d%d",&n,&q);
	S=(ele)(sqrt(n)*log(n)/log(2)+0.5);
	for (int i=0; i<n; ++i)
		scanf("%d",a+i);
	bcnt=0;
	for (int i=0; i<n; i+=S,++bcnt){
		b[bcnt].n=min(S,n-i);
		for (int j=0; j<b[bcnt].n; ++j)
			b[bcnt].b[j]=b[bcnt].a[j]=a[i+j];
		sort(b[bcnt].b,b[bcnt].b+b[bcnt].n);
	}
	while (q--){
		ele op,u,v,w;
		scanf("%d%d%d",&op,&u,&v);
		if (op==3)
			--u;
		else{
			--u,--v;
			scanf("%d",&w);
		}
		if (op==1)
			printf("%d\n",getrank(u,v,w));
		else if (op==2){
			ele L=0,R=maxa+1;
			while (R-L>1){
				ele mid=(L+R)>>1;
				if (getrank(u,v,mid)<=w)
					L=mid;
				else R=mid;
			}
			printf("%d\n",L);
		}
		else if (op==3)
			upd(u,v);
		else if (op==4)
			printf("%d\n",prev(u,v,w));
		else if (op==5)
			printf("%d\n",succ(u,v,w));
	}
	return 0;
}
