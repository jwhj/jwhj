bzoj1034题解

Aug 6, 2017 10:02 · 97 words · 1 minute read 题解 bzoj 贪心

每次找出对手中最强的一个，用己方能赢的最弱的人去打，如果都不能赢，暂不分配对手。接下来，对于剩下的对手，能打平尽量打平。这样一定能得到最多的分数。

大概说一下为什么是对的。考虑当前某个对手a，按当前策略选出的对手是b。如果存在某个最优策略，己方输给了a，那么把a的对手调成b，一定不会更差。如果存在某个最优策略，己方与a打平，假设己方派出的是c，那么c比a弱且与b相当。考虑b，如果b输了或者平了，那么把a的对手改成b一定不会更差；如果b赢了，考虑b的对手d，如果d比a弱或者与a相当，同样可以把a的对手调成b，如果d比a强，由算法流程，为a挑对手的时候不可能调到b，矛盾。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
#define ele int
using namespace std;
#define maxn 100010
ele n,a[maxn],b[maxn];
bool vis[maxn];
multiset<ele> s;
inline ele calc(ele *a,ele *b){
	ele ans=0;
	s.clear();
	memset(vis,0,sizeof(vis));
	for (int i=0; i<n; ++i) s.insert(a[i]);
	for (int i=n-1; ~i; --i){
		multiset<ele>::iterator it=s.upper_bound(b[i]);
		if (it!=s.end()){
			vis[i]=true;
			ans+=2;
			s.erase(it);
		}
	}
	for (int i=n-1; ~i; --i){
		if (vis[i]) continue;
		multiset<ele>::iterator it=s.find(b[i]);
		if (it!=s.end()){
			vis[i]=true;
			ans+=1;
			s.erase(it);
		}
	}
	return ans;
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%d",a+i);
	for (int i=0; i<n; ++i) scanf("%d",b+i);
	sort(a,a+n); sort(b,b+n);
	printf("%d %d\n",calc(a,b),2*n-calc(b,a));
	return 0;
}
