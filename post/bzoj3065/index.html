bzoj3065题解

May 29, 2016 10:02 · 283 words · 2 minutes read 题解 bzoj 分块 块状链表

块状链表，每个块内部按升序排序，查询的时候二分套二分。

分块的时候每$\sqrt{n}\log n$分一个块，效率最高。

块状链表有一个比较短的写法：不对块进行分裂、合并，但每$\sqrt{n}$次操作后把整个链表暴力重构一次，摊还时间复杂度也是一样的。

时间复杂度：

查询$O(\sqrt{n}\log n)$

修改$O(\sqrt{n}\log n)$

插入$O(\sqrt{n}\log n)$

看上去很可怕，但据说这么做常数很小，就是A了……

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 70010
#define maxb 610
struct blk
{
	ele n,a[maxb],b[maxb];
};
ele n,q,B,bcnt,tmp[maxn];
blk a[maxb];
inline ele build(ele *b,blk *a)
{
	ele bcnt=0;
	for (int i=0; i<n; i+=B)
	{
		ele j=min(i+B,n);
		for (int k=i; k<j; ++k)
		a[bcnt].a[k-i]=a[bcnt].b[k-i]=b[k];
		a[bcnt].n=j-i;
		sort(a[bcnt].b,a[bcnt].b+a[bcnt].n);
		++bcnt;
	}
	return bcnt;
}
inline ele expt(blk *a,ele bcnt,ele *b)
{
	ele p=0;
	for (int i=0; i<bcnt; ++i)
		for (int j=0; j<a[i].n; ++j)
			b[p++]=a[i].a[j];
	return p;
}
inline void getpos(ele x,ele &i,ele &j)
{
	j=x;
	for (i=0; i<bcnt-1 && j>=a[i].n; ++i)
		j-=a[i].n;
}
inline ele lt(ele l,ele r,ele x)
{
	ele il,jl,ir,jr;
	getpos(l,il,jl); getpos(r,ir,jr);
	ele ans=0;
	if (il==ir)
	{
		for (int i=jl; i<=jr; ++i)
			if (a[il].a[i]<x) ++ans;
		return ans;
	}
	for (int i=jl; i<a[il].n; ++i)
		if (a[il].a[i]<x) ++ans;
	for (int i=0; i<=jr; ++i)
		if (a[ir].a[i]<x) ++ans;
	for (int i=il+1; i<ir; ++i)
	{
		ele k=lower_bound(a[i].b,a[i].b+a[i].n,x)-a[i].b;
		ans+=k;
	}
	return ans;
}
int main()
{
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%d",tmp+i);
	B=265;
	bcnt=build(tmp,a);
	scanf("%d",&q);
	ele lastans=0;
	for (int i=0; i<q; ++i)
	{
		char s[10];
		ele x,y,k;
		scanf("%s%d%d",s,&x,&y);
		x^=lastans; y^=lastans;
		if (s[0]=='Q')
		{
			--x,--y;
			scanf("%d",&k);
			k^=lastans;
			ele L=0,R=70001,mid;
			while (R-L>1)
			{
				mid=(L+R)>>1;
				if (lt(x,y,mid)<=k-1) L=mid; else R=mid;
			}
			printf("%d\n",lastans=L);
		}
		else if (s[0]=='M')
		{
			--x;
			ele i,j;
			getpos(x,i,j);
			ele p=lower_bound(a[i].b,a[i].b+a[i].n,a[i].a[j])-a[i].b;
			a[i].b[p]=y;
			for (int k=p; k<a[i].n-1; ++k)
				if (a[i].b[k]>a[i].b[k+1]) swap(a[i].b[k],a[i].b[k+1]);
				else break;
			for (int k=p; k>0; --k)
				if (a[i].b[k]<a[i].b[k-1]) swap(a[i].b[k],a[i].b[k-1]);
				else break;
			a[i].a[j]=y;
		}
		else if (s[0]=='I'){
			--x;
			ele i,j;
			getpos(x,i,j);
			++a[i].n;
			for (int k=a[i].n-1; k>j; --k) a[i].a[k]=a[i].a[k-1];
			a[i].a[j]=y;
			a[i].b[a[i].n-1]=y;
			for (int k=a[i].n-1; k>0; --k)
				if (a[i].b[k]<a[i].b[k-1]) swap(a[i].b[k],a[i].b[k-1]);
				else break;
		}
		if (i%B==0)
		{
			n=expt(a,bcnt,tmp);
			bcnt=build(tmp,a);
		}
	}
	return 0;
}
