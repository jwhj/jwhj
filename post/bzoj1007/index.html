bzoj1007题解

Dec 16, 2016 10:02 · 108 words · 1 minute read 题解 bzoj 计算几何 凸包

维护一个下凸壳再乱搞一下就可以了。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
#define real double
using namespace std;
#define maxn 50010
#define eps 1e-6
inline real rabs(real x){
    return (x<0)?(-x):(x);
}
struct line{
    real a,b;
    ele id;
    inline bool operator<(line y)const{
        if (rabs(a-y.a)>eps) return a<y.a;
        return b<y.b;
    }
};
ele n,q[maxn];
line l[maxn];
inline real func(line a,line b){
    return (b.b-a.b)/(a.a-b.a);
}
inline bool cmp(ele i,ele j){
    return l[i].id<l[j].id;
}
int main(){
    scanf("%d",&n);
    for (int i=0; i<n; ++i)
        scanf("%lf%lf",&l[i].a,&l[i].b),l[i].id=i+1;
    sort(l,l+n);
    ele cnt=0;
    for (int i=0; i<n; ++i){
        while (cnt && rabs(l[q[cnt-1]].a-l[i].a)<eps)
            --cnt;
        while (cnt>1 && func(l[q[cnt-2]],l[i])<func(l[q[cnt-2]],l[q[cnt-1]])+eps)
            --cnt;
        q[cnt++]=i;
    }
    sort(q,q+cnt,cmp);
    for (int i=0; i<cnt; ++i)
        printf("%d ",l[q[i]].id);
    return 0;
}
