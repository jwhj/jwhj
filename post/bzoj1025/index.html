bzoj1025题解

May 26, 2017 10:02 · 82 words · 1 minute read 题解 bzoj DP 数学

题意可以简化为：把n拆成若干个正整数，它们的lcm有多少种不同的可能。

直接考虑有多少种可能比较困难，考虑判断一个A是否是某一个拆分的lcm。

设$A=\prod p_i^{a_i}$，易证它是某一拆分的lcm当且仅当$\sum p_i^{a_i}\le n$。

让f[i][j]表示只考虑前i种素数，满足$\sum p_i^{a_i}\le j$的${a_i}$有多少个。

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 1010
ele n,pcnt,plst[maxn],f[maxn][maxn];
bool flag[maxn];
int main(){
	scanf("%lld",&n);
	pcnt=0;
	memset(flag,0,sizeof(flag));
	for (int i=2; i<=n; ++i){
		if (!flag[i]) plst[pcnt++]=i;
		for (int j=0; j<pcnt && i*plst[j]<=n; ++j){
			flag[i*plst[j]]=true;
			if (i%plst[j]==0) break;
		}
	}
	for (int j=0; j<=n; ++j) f[0][j]=1;
	for (int i=0; i<pcnt; ++i)
		for (int j=0; j<=n; ++j){
			f[i+1][j]=f[i][j];
			for (int k=plst[i]; k<=j; k*=plst[i]) f[i+1][j]+=f[i][j-k];
		}
	printf("%lld\n",f[pcnt][n]);
	return 0;
}
