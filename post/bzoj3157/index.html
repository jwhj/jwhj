bzoj3157题解

Apr 22, 2017 10:02 · 128 words · 1 minute read 题解 bzoj 数学

这里有一份比较好的题解。

有一个坑点就是最后记得要输出g[m]而不是f[m]，因为如果n=1的话f数组不会被赋值。

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 210
#define MOD 1000000007
ele n,m,top,stk[100],f[maxn],g[maxn],fac[maxn],ifac[maxn];
inline ele pw(ele a,ele x){
	ele ans=1,tmp=a;
	for (; x; x>>=1,tmp=tmp*tmp%MOD)
		if (x&1) ans=ans*tmp%MOD;
	return ans;
}
inline ele C(ele n,ele m){
	return fac[n]*ifac[m]%MOD*ifac[n-m]%MOD;
}
int main(){
	fac[0]=1;
	for (int i=1; i<maxn; ++i) fac[i]=fac[i-1]*i%MOD;
	for (int i=0; i<maxn; ++i) ifac[i]=pw(fac[i],MOD-2);
	scanf("%d%d",&n,&m);
	stk[0]=n;
	top=1;
	while (stk[top-1]!=1){
		ele t=stk[top-1];
		if (t&1) stk[top++]=t-1;
		else stk[top++]=(t>>1);
	}
	for (int k=0; k<=m; ++k) g[k]=m;
	while (top>1){
		if (stk[top-2]==stk[top-1]+1){
			for (int k=0; k<=m; ++k){
				f[k]=1;
				for (int j=0; j<=k; ++j){
					f[k]+=C(k,j)*g[j]%MOD;
					f[k]%=MOD;
				}
				f[k]*=m; f[k]%=MOD;
			}
		}
		else{
			for (int k=0; k<=m; ++k){
				f[k]=g[k];
				ele t=pw(m,stk[top-1]);
				for (int j=0; j<=k; ++j){
					f[k]+=C(k,j)*pw(stk[top-1],k-j)%MOD*g[j]%MOD*t%MOD;
					f[k]%=MOD;
				}
			}
		}
		memcpy(g,f,sizeof(ele)*(m+1));
		--top;
	}
	printf("%lld\n",g[m]);
	return 0;
}
