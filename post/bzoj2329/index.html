bzoj2329题解

Feb 13, 2017 10:02 · 246 words · 2 minutes read 题解 bzoj 括号序列 treap 统计

跟bzoj2209是差不多的，这里有那道题的题解。

不同的地方在于要多打一个标记，下传标记的时候要考虑一些细节。

代码：

#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 100010
struct ANS{
	ele pmn,pmx,smn,smx,sum;
};
inline ANS operator+(ANS a,ANS b){
	ANS c;
	c.sum=a.sum+b.sum;
	c.pmn=min(a.pmn,a.sum+b.pmn);
	c.pmx=max(a.pmx,a.sum+b.pmx);
	c.smn=min(b.smn,a.smn+b.sum);
	c.smx=max(b.smx,a.smx+b.sum);
	return c;
}
struct node{
	ANS a;
	ele s,p,st,x;
	bool flip,neg;
	node *l,*r;
}np[maxn];
inline void pushdown(node *x){
	if (!x) return;
	if (x->flip){
		swap(x->l,x->r);
		if (x->l) x->l->flip^=1;
		if (x->r) x->r->flip^=1;
		swap(x->a.pmn,x->a.smn);
		swap(x->a.pmx,x->a.smx);
		x->flip=false;
	}
	if (x->neg){
		swap(x->a.pmn,x->a.pmx);
		x->a.pmn*=-1; x->a.pmx*=-1;
		swap(x->a.smn,x->a.smx);
		x->a.smn*=-1; x->a.smx*=-1;
		x->a.sum*=-1; x->x*=-1;
		if (x->l) x->l->neg^=1,x->l->st*=-1;
		if (x->r) x->r->neg^=1,x->r->st*=-1;
		x->neg=false;
	}
	if (x->st){
		x->x=x->st;
		ele u=x->x,v=x->x*x->s;
		ele t1=min(u,v),t2=max(u,v);
		x->a=(ANS){t1,t2,t1,t2,v};
		if (x->l) x->l->st=x->st;
		if (x->r) x->r->st=x->st;
		x->st=0;
	}
}
inline void maintain(node *x){
	if (!x) return;
	x->a=(ANS){x->x,x->x,x->x,x->x,x->x};
	x->s=1;
	if (x->l) pushdown(x->l),x->a=x->l->a+x->a,x->s+=x->l->s;
	if (x->r) pushdown(x->r),x->a=x->a+x->r->a,x->s+=x->r->s;
}
node* merge(node *a,node *b){
	if (!b) return a;
	if (!a) return b;
	pushdown(a); pushdown(b);
	if (a->p>b->p){
		a->r=merge(a->r,b);
		maintain(a);
		return a;
	}
	else{
		b->l=merge(a,b->l);
		maintain(b);
		return b;
	}
}
void split(node *x,ele k,node*&a,node*&b){
	if (!x){ a=b=NULL; return; }
	pushdown(x);
	ele s=(x->l)?(x->l->s+1):(1);
	if (k<s){
		split(x->l,k,a,b);
		x->l=b;
		maintain(x);
		b=x;
	}
	else if (k>s){
		split(x->r,k-s,a,b);
		x->r=a;
		maintain(x);
		a=x;
	}
	else{
		b=x->r;
		x->r=NULL;
		maintain(x);
		a=x;
	}
}
ele n,q;
char s[maxn];
node *root;
int main(){
	srand(2333);
	scanf("%d%d%s",&n,&q,s);
	root=NULL;
	for (int i=0; i<n; ++i){
		np[i].x=(s[i]=='(')?(-1):(1);
		np[i].l=np[i].r=NULL;
		np[i].st=0;
		np[i].flip=np[i].neg=false;
		maintain(np+i);
		np[i].p=rand();
		root=merge(root,np+i);
	}
	while (q--){
		char op[10],s1[5];
		ele l,r;
		scanf("%s%d%d",op,&l,&r);
		node *a,*b,*c;
		split(root,r,b,c);
		split(b,l-1,a,b);
		pushdown(b);
		if (op[0]=='R'){
			scanf("%s",s1);
			b->st=(s1[0]=='(')?(-1):(1);
		}
		else if (op[0]=='S')
			b->flip^=1;
		else if (op[0]=='I')
			b->neg^=1;
		else{
			ele x=max(b->a.pmx,0);
			ele y=max(-b->a.smn,0);
			printf("%d\n",(x&1)?(((x+y)>>1)+1):((x+y)>>1));
		}
		root=merge(a,b);
		root=merge(root,c);
	}
	return 0;
}
