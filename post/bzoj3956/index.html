bzoj3956题解

Sep 4, 2016 10:02 · 208 words · 1 minute read 题解 bzoj 统计 单调栈/队列 乱搞

假设(i,j)是一个好点对，且$a_i\lt a_j$，那么$a_j$只能是$a_i$右侧第一个大于$a_i$的数。

同样地，如果$a_i\gt a_j$，那么$a_i$只能是$a_j$左侧第一个大于$a_i$的数。

所以，如果一个数作为好点对中的较小数，那么至多只有两种情况：左边一个，右边一个。

用单调栈可以$O(n)$地统计出每个数左右第一个比它大的数，接着就可以得出以每个数为起点（不一定要是较小数）的好点对个数，以及以每个数为终点（也不一定要是较小数）的好点对个数。

考虑询问l,r，求出[l,r]中最大数的编号x（有多个则任取一个)，好点对显然不会跨越x，所以答案为左端点在[l,x)中的好点对数+右端点在(x,r]中的好点对数。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 300010
const ele size=1<<20;
const ele M=size>>1;
struct data
{
	ele x,i;
	inline bool operator<(data b)const
	{
		if (x!=b.x) return x>b.x;
		return i<b.i;
	}
};
ele n,m,ty,a[maxn],f[maxn],g[maxn],z[size],idx[size];
data stack[maxn];
inline void maintain(ele *z,ele *idx,ele i)
{
	if (z[i<<1]>z[i<<1|1]) z[i]=z[i<<1],idx[i]=idx[i<<1];
	else z[i]=z[i<<1|1],idx[i]=idx[i<<1|1];
}
inline ele query(ele *z,ele *idx,ele l,ele r)
{
	ele _max=-1e9,i;
	for (l=l+M,r=r+M+2; l^r^1; l>>=1,r>>=1)
	{
		if (~l&1 && z[l^1]>_max) _max=z[l^1],i=idx[l^1];
		if (r&1 && z[r^1]>_max) _max=z[r^1],i=idx[r^1];
	}
	return i;
}
int main()
{
	scanf("%d%d%d",&n,&m,&ty);
	for (int i=0; i<n; ++i) scanf("%d",a+i);
	stack[0]=(data){a[0],0};
	ele cnt=1;
	memset(f,0,sizeof(f));
	for (int i=1; i<n; ++i)
	{
		data p=(data){a[i],i};
		ele k=lower_bound(stack,stack+cnt,p)-stack-1;
		if (k>=0) ++f[stack[k].i],++g[i];
		while (cnt>0 && stack[cnt-1].x<=a[i]) --cnt;
		stack[cnt++]=p;
	}
	for (int i=0; i<=n-i-1; ++i) swap(a[i],a[n-i-1]);
	stack[0]=(data){a[0],0};
	cnt=1;
	g[0]=0;
	for (int i=1; i<n; ++i)
	{
		data p=(data){a[i],i};
		ele k=lower_bound(stack,stack+cnt,p)-stack-1;
		if (k>=0 && stack[k].x!=a[i]) ++g[n-1-stack[k].i],++f[n-1-i];
		while (cnt>0 && stack[cnt-1].x<=a[i]) --cnt;
		stack[cnt++]=p;
	}
	for (int i=0; i<=n-i-1; ++i) swap(a[i],a[n-i-1]);
	f[n]=g[n]=0;
	for (int i=n-1; i>=0; --i) f[i]+=f[i+1],g[i]+=g[i+1];
	for (int i=0; i<n; ++i) z[i+M+1]=a[i],idx[i+M+1]=i;
	for (int i=M-1; i>0; --i) maintain(z,idx,i);
	ele last=0;
	while (m--)
	{
		ele l,r;
		scanf("%d%d",&l,&r);
		if (ty)
		{
			l=(l+last-1)%n+1;
			r=(r+last-1)%n+1;
			if (l>r) swap(l,r);
		}
		--l,--r;
		ele i=query(z,idx,l,r);
		printf("%d\n",last=f[l]-f[i]+g[i+1]-g[r+1]);
	}
	return 0;
}
