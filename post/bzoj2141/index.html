bzoj2141题解

Jul 2, 2016 10:02 · 292 words · 2 minutes read 题解 bzoj 分块

感觉数据好小啊……考虑使用分块，每块里面排好序。

交换hi和hj的时候，只有i,j之间的数会对答案造成影响。具体来讲，对于每个$p \in(i,j)$，如果$hp \gt hi$，答案+1，如果$h_p \lt h_i$，答案-1。对于$h_j$有类似的讨论。每个块排了序，两边的块暴力计算，中间的二分。

如果块的大小为$O(\sqrt n)$，那么每次的复杂度是$O(\sqrt n\log n)$的，但是注意到，对于每次询问，修改要花$O(\sqrt n)$，计算答案要花$O(\sqrt n\log n)$的时间，不是很均衡。如果把块的大小增大到$O(\sqrt{n\log n})$，时间复杂度就能降下来一点。

最后这个算法时间复杂度为$O(n\sqrt{n\log n})$。

代码：

#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 20010
#define maxs 550
#define maxc 100
#define S 140
struct BIT
{
	inline void alloc(ele n)
	{
		this->n=n;
		memset(a,0,sizeof(a));
	}
	inline void add(ele i,ele x)
	{
		while (i<=n)
		{
			a[i]+=x;
			i+=i&(-i);
		}
	}
	inline ele prefix(ele i)
	{
		ele ans=0;
		while (i>0)
		{
			ans+=a[i];
			i-=i&(-i);
		}
		return ans;
	}
	inline ele sum(ele l,ele r)
	{
		if (l>r) return 0;
		return prefix(r)-prefix(l-1);
	}
	ele n,a[maxn];
};
struct block
{
	ele n,a[maxs],b[maxs];
};
ele n,m,len,B,ans,h[maxn],arr[maxn];
BIT bit;
block a[maxc];
inline void build(ele n,ele *a,block *b)
{
	for (int i=0,j=0; i<n; i+=B,++j)
	{
		int k;
		for (k=0; k<B && i+k<n; ++k)
			b[j].a[k]=b[j].b[k]=a[i+k];
		b[j].n=k;
		sort(b[j].b,b[j].b+b[j].n);
	}
}
inline void scan(ele l,ele r,ele x,ele k)
{
	if (l>r) return;
	ele i1=l/B,j1=l%B,i2=r/B,j2=r%B;
	if (i1==i2)
	{
		for (int j=j1; j<=j2; ++j)
			if (a[i1].a[j]<x) ans-=k;
			else if (a[i1].a[j]>x) ans+=k;
		return;
	}
	for (int j=j1; j<a[i1].n; ++j)
		if (a[i1].a[j]<x) ans-=k;
		else if (a[i1].a[j]>x) ans+=k;
	for (int j=0; j<=j2; ++j)
		if (a[i2].a[j]<x) ans-=k;
		else if (a[i2].a[j]>x) ans+=k;
	for (int i=i1+1; i<i2; ++i)
	{
		ele p=lower_bound(a[i].b,a[i].b+a[i].n,x)-a[i].b;
		ans-=p*k;
		p=upper_bound(a[i].b,a[i].b+a[i].n,x)-a[i].b;
		ans+=(a[i].n-p)*k;
	}
}
inline void upd(ele k,ele x)
{
	ele i=k/B,j=k%B;
	ele p=lower_bound(a[i].b,a[i].b+a[i].n,a[i].a[j])-a[i].b;
	a[i].b[p]=x;
	ele q=p;
	while (q>0 && a[i].b[q-1]>a[i].b[q])
		swap(a[i].b[q-1],a[i].b[q]),--q;
	q=p;
	while (q+1<a[i].n && a[i].b[q]>a[i].b[q+1])
		swap(a[i].b[q],a[i].b[q+1]),++q;
	a[i].a[j]=x;
}
int main()
{
	scanf("%lld",&n);
	B=sqrt((double)n*log(n)/log(2.0));
	for (int i=0; i<n; ++i) scanf("%lld",h+i),arr[i]=h[i];
	sort(arr,arr+n);
	len=unique(arr,arr+n)-arr;
	for (int i=0; i<n; ++i)
		h[i]=lower_bound(arr,arr+len,h[i])-arr+1;
	ans=0;
	bit.alloc(len);
	for (int i=0; i<n; ++i)
	{
		ans+=bit.sum(h[i]+1,len);
		bit.add(h[i],1);
	}
	printf("%lld\n",ans);
	build(n,h,a);
	scanf("%lld",&m);
	while (m--)
	{
		ele i,j;
		scanf("%lld%lld",&i,&j);
		--i,--j;
		if (i>j) swap(i,j);
		if (h[i]>h[j]) --ans;
		else if (h[i]<h[j]) ++ans;
		scan(i+1,j-1,h[i],1);
		scan(i+1,j-1,h[j],-1);
		upd(i,h[j]);
		upd(j,h[i]);
		swap(h[i],h[j]);
		printf("%lld\n",ans);
	}
	return 0;
}
