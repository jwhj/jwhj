bzoj3673&3674题解

Sat 13 August 2016

并查集的实现基于数组，如果可以把数组可持久化的话，问题就解决了。

可以建立一棵线段树当数组用，线段树是很容易可持久化的。这样复杂度多处一个$O(\log n)$。当然可持久化数组可以做到$O(\log\log n)$，不过我不会。

3674代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 200010
struct node
{
    ele f;
    node *l,*r;
}pool[maxn<<6];
ele ncnt;
node *build(ele l,ele r)
{
    node *p=&pool[ncnt++];
    if (l==r)
    {
        p->f=l;
        p->l=p->r=NULL;
    }
    else
    {
        ele mid=(l+r)>>1;
        p->l=build(l,mid);
        p->r=build(mid+1,r);
    }
    return p;
}
node *upd(node *x,ele a,ele b,ele i,ele k)
{
    node *p=&pool[ncnt++];
    *p=*x;
    if (a==b){p->f=k; return p;}
    ele mid=(a+b)>>1;
    if (i<=mid) p->l=upd(x->l,a,mid,i,k);
    else p->r=upd(x->r,mid+1,b,i,k);
    return p;
}
ele query(node *x,ele a,ele b,ele i)
{
    if (a==b) return x->f;
    ele mid=(a+b)>>1;
    if (i<=mid) return query(x->l,a,mid,i);
    else return query(x->r,mid+1,b,i);
}
ele n,m;
node *t[maxn],*a;
ele getf(ele x)
{
    ele tmp=query(a,0,n-1,x);
    if (x==tmp) return x;
    tmp=getf(tmp);
    a=upd(a,0,n-1,x,tmp);
    return tmp;
}
inline void uni(ele x,ele y)
{
    ele u=getf(x),v=getf(y);
    if (u==v) return;
    a=upd(a,0,n-1,u,v);
}
int main()
{
    scanf("%d%d",&n,&m);
    ncnt=0;
    a=t[0]=build(0,n-1);
    ele lastans=0;
    for (int i=1; i<=m; ++i)
    {
        ele cmd,u,v;
        scanf("%d%d",&cmd,&u);
        u^=lastans;
        if (cmd!=2) scanf("%d",&v),v^=lastans;
        if (cmd==1)
            uni(u,v);
        else if (cmd==2) a=t[u];
        else if (cmd==3)
            if (getf(u)==getf(v)) lastans=1,puts("1");
            else lastans=0,puts("0");
        t[i]=a;
    }
    return 0;
}
