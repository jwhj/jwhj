暴力的方法当然是很容易想的，期望得分25。

考虑一种简单一些的情况：玩家都在序列而不是树上跑，而且所有玩家都在往右跑。

假设某个玩家从l跑到r，那么他会在满足$j \in [l,r]$且$j-w_j=l$的点j上被观测到。

于是，我们可以将所有的玩家按照l排序，把序列上的所有点按照$j-w_j$排序，每次加入l=p的所有玩家，用树状数组统计序列上每个点被覆盖了几次，据此计算所有$j-w_j=p$的点的答案，最后把l=p的玩家删去。这样每个玩家被加入一次，删去一次，序列上每个点被计算一次，时间复杂度为$O(n\log n)$。

考虑复杂一点的情况，有的玩家往右跑，有的往左跑。

此时可以把这两种玩家分开处理，往右跑的处理方法同上，对于往左跑的玩家，假设他从r跑到l，那么他会在满足$j \in [l,r]$且$j+w_j=r$的点j上被观测到。于是往左跑的玩家也可以用类似上面的办法处理。

现在需要把序列上的做法弄到树上。

一个简单的想法是使用树链剖分，不过码量可能会有一点大，时间复杂度$O(n\log^2 n)$。

其实可以不用树链剖分。考虑一个玩家从u跑到v，u和v的最近公共祖先为w。那么其实可以把它拆成u向上跑到w和w向下跑到v两部分。把上面的$j\pm w_j$改成$dep_j\pm w_j$就可以用类似的方法来做了。

此时需要一个数据结构，支持链上加一个数以及单点查询，可以树上差分再用树状数组来维护。这样时间复杂度是$O(n\log n)$的。

这个算法经测试是95分，最后一个点TLE应该是倍增求lca常数太大的原因，晚点优化一下试试看。

代码：

#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele long long
using namespace std;
#define maxn 300008
#define maxk 20
#define INF 1000000000
struct edge{
    ele v;
    edge *nxt;
}ep[maxn<<1];
struct BIT{
    inline void alloc(ele n){
        this->n=n;
        memset(a,0,sizeof(a));
    }
    inline void add(ele i,ele k){
        for (int j=i; j<=n; j+=j&(-j))
            a[j]+=k;
    }
    inline ele query(ele i){
        ele ans=0;
        for (int j=i; j>0; j-=j&(-j))
            ans+=a[j];
        return ans;
    }
    inline ele query(ele l,ele r){
        if (l>r) return 0;
        return query(r)-query(l-1);
    }
    ele n,a[maxn];
};
struct ASK{
    ele u,v,w;
    inline bool operator<(ASK b)const{
        if (w!=b.w) return w<b.w;
        if (u!=b.u) return u<b.u;
        return v<b.v;
    }
};
ele n,m,ecnt,scnt,ans[maxn],tmp[maxn],w[maxn],s[maxn],t[maxn],size[maxn],fa[maxn][maxk],dep[maxn],seq[maxn],pos[maxn];
edge *h[maxn];
ASK q[maxn],q1[maxn];
BIT b;
inline void addedge(ele u,ele v){
    edge *p=&ep[ecnt++];
    p->v=v; p->nxt=h[u];
    h[u]=p;
}
void dfs(ele p,ele i){
    pos[i]=scnt;
    seq[scnt++]=i;
    size[i]=1;
    fa[i][0]=p;
    for (edge *j=h[i]; j; j=j->nxt)
        if (j->v!=p){
            dep[j->v]=dep[i]+1;
            dfs(i,j->v);
            size[i]+=size[j->v];
        }
}
inline ele lca(ele u,ele v){
    if (dep[u]>dep[v]) swap(u,v);
    ele k=dep[v]-dep[u];
    for (int i=0; k; k>>=1,++i)
        if (k&1)
            v=fa[v][i];
    if (u==v) return u;
    for (int i=maxk-1; i>=0; --i)
        if (fa[u][i]!=fa[v][i])
            u=fa[u][i],v=fa[v][i];
    return fa[u][0];
}
inline bool cmp1(ele i,ele j){
    return w[i]+dep[i]<w[j]+dep[j];
}
inline bool cmp2(ele i,ele j){
    return dep[i]-w[i]<dep[j]-w[j];
}
int main(){
    scanf("%lld%lld",&n,&m);
    ecnt=0;
    memset(h,0,sizeof(h));
    for (int i=0; i<n-1; ++i){
        ele u,v;
        scanf("%lld%lld",&u,&v);
        --u,--v;
        addedge(u,v); addedge(v,u);
    }
    for (int i=0; i<n; ++i)
        scanf("%lld",w+i);
    for (int i=0; i<m; ++i)
        scanf("%lld%lld",s+i,t+i),--s[i],--t[i];
    dep[0]=0;
    scnt=0;
    dfs(-1,0);
    for (int i=1; i<maxk; ++i)
        for (int j=0; j<maxn; ++j)
            if (fa[j][i-1]==-1)
                fa[j][i]=-1;
            else fa[j][i]=fa[fa[j][i-1]][i-1];
    memset(ans,0,sizeof(ans));
    for (int i=0; i<m; ++i){
        ele p=lca(s[i],t[i]);
        q[i].u=p;
        q[i].v=s[i];
        q[i].w=dep[s[i]];
        if (p==t[i])
            q1[i].w=-INF;
        else{
            q1[i].v=t[i];
            ele u=t[i],k=dep[t[i]]-dep[p]-1;
            for (int j=0; k; k>>=1,++j)
                if (k&1)
                    u=fa[u][j];
            q1[i].u=u;
            q1[i].w=dep[u]-(dep[s[i]]-dep[p]+1);
        }
    }
    sort(q,q+m);
    for (int i=0; i<n; ++i)
        tmp[i]=i;
    sort(tmp,tmp+n,cmp1);
    ele j=0;
    b.alloc(n);
    for (int i=0; i<n; ++i){
        ele t1=w[tmp[i]]+dep[tmp[i]];
        for (; j<m && q[j].w<t1; ++j)
            ;
        ele k=j;
        for (; k<m && q[k].w==t1; ++k)
            ;
        for (int p=j; p<k; ++p){
            b.add(pos[q[p].v]+1,1);
            if (fa[q[p].u][0]!=-1)
                b.add(pos[fa[q[p].u][0]]+1,-1);
        }
        for (; i<n && w[tmp[i]]+dep[tmp[i]]==t1; ++i){
            ele t2=b.query(pos[tmp[i]]+1,pos[tmp[i]]+size[tmp[i]]);
            ans[tmp[i]]+=t2;
        }
        for (int p=j; p<k; ++p){
            b.add(pos[q[p].v]+1,-1);
            if (fa[q[p].u][0]!=-1)
                b.add(pos[fa[q[p].u][0]]+1,1);
        }
        --i;
        j=k;
    }
    sort(q1,q1+m);
    for (int i=0; i<n; ++i)
        tmp[i]=i;
    sort(tmp,tmp+n,cmp2);
    j=0;
    b.alloc(n);
    for (int i=0; i<n; ++i){
        ele t1=dep[tmp[i]]-w[tmp[i]];
        for (; j<m && q1[j].w<t1; ++j)
            ;
        ele k=j;
        for (; k<m && q1[k].w==t1; ++k)
            ;
        for (int p=j; p<k; ++p){
            b.add(pos[q1[p].v]+1,1);
            if (fa[q1[p].u][0]!=-1)
                b.add(pos[fa[q1[p].u][0]]+1,-1);
        }
        for (; i<n && dep[tmp[i]]-w[tmp[i]]==t1; ++i){
            ele t2=b.query(pos[tmp[i]]+1,pos[tmp[i]]+size[tmp[i]]);
            ans[tmp[i]]+=t2;
        }
        for (int p=j; p<k; ++p){
            b.add(pos[q1[p].v]+1,-1);
            if (fa[q1[p].u][0]!=-1)
                b.add(pos[fa[q1[p].u][0]]+1,1);
        }
        --i;
        j=k;
    }
    for (int i=0; i<n-1; ++i)
        printf("%lld ",ans[i]);
    printf("%lld\n",ans[n-1]);
    return 0;
}
