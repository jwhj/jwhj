bzoj1875题解

Sun 11 September 2016

如果没有不会立刻沿着刚刚走来的路走回这个条件，可以参考这道题的做法。

现在加上了这个条件，可以把每条边拆成两条有向边，然后把这些边作为点建一个新图。边$e_1=(u_1,v_1)$对应的点有边到达边$e_2=(u_2,v_2)$对应的点当且仅当$v_1=u_2$且$e_1,e_2$不由同一条边拆出来。

然后就可以用矩乘来做了。

忘了异或的优先级比不等于要低结果调了好久……其实可以直接i^j^1的……

代码：

#include <cstdio>
#include <cstring>
#define ele long long
using namespace std;
#define maxn 30
#define maxm 130
#define MOD 45989
struct edge
{
    ele u,v;
}e[maxm];
struct MAT
{
    ele n,m,a[maxm][maxm];
};
inline MAT operator*(MAT a,MAT b)
{
    static MAT ans;
    ans.n=a.n; ans.m=b.m;
    for (int i=0; i<a.n; ++i)
        for (int j=0; j<b.m; ++j)
        {
            ans.a[i][j]=0;
            for (int k=0; k<a.m; ++k)
                ans.a[i][j]+=a.a[i][k]*b.a[k][j],ans.a[i][j]%=MOD;
        }
    return ans;
}
inline MAT pow(MAT a,ele x)
{
    static MAT ans;
    ans.n=ans.m=a.n;
    memset(ans.a,0,sizeof(ans.a));
    for (int i=0; i<a.n; ++i) ans.a[i][i]=1;
    for (static MAT tmp=a; x; x>>=1,tmp=tmp*tmp)
        if (x&1) ans=ans*tmp;
    return ans;
}
ele n,m,t,A,B;
MAT C,C1;
int main()
{
    scanf("%lld%lld%lld%lld%lld",&n,&m,&t,&A,&B);
    for (int i=0; i<m; ++i)
        scanf("%lld%lld",&e[i<<1].u,&e[i<<1].v),
        e[i<<1|1].u=e[i<<1].v,e[i<<1|1].v=e[i<<1].u;
    C.n=C.m=m<<1|1;
    memset(C.a,0,sizeof(C.a));
    for (int i=0; i<(m<<1); ++i)
        for (int j=0; j<(m<<1); ++j)
            if (e[i].v==e[j].u && i^j^1) C.a[i][j]=1;
    C=pow(C,t-1);
    C1.n=C1.m=m<<1|1;
    memset(C1.a,0,sizeof(C1.a));
    for (int i=0; i<(m<<1); ++i)
        if (e[i].u==A) C1.a[m<<1][i]=1;
    C=C1*C;
    ele ans=0;
    for (int i=0; i<(m<<1); ++i)
        if (e[i].v==B)
            ans+=C.a[m<<1][i],ans%=MOD;
    printf("%lld\n",ans);
    return 0;
}
